181  void UpdateAI(const uint32 diff) override
182  {
183  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
184  return;
185 
186  // Summoning auras handling
187  if (m_summonTimer)
188  {
189  if (m_summonTimer < diff)
190  {
191  switch (m_summonStep) {
192  case STEP_TRAINEE:
193 
194  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_TRAINEE, CAST_TRIGGERED) == CAST_OK)
195  {
196  m_summonTimer = 45 * IN_MILLISECONDS;
197  ++m_summonStep;
198  }
199  break;
200  case STEP_KNIGHT:
201  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_KNIGHT, CAST_TRIGGERED) == CAST_OK)
202  {
203  m_summonTimer = 55 * IN_MILLISECONDS;
204  ++m_summonStep;
205  }
206  break;
207  case STEP_RIDER:
208  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_MOUNTED_KNIGHT, CAST_TRIGGERED) == CAST_OK)
209  {
210  m_summonTimer = 0;
211  m_summonStep = 0;
212  }
213  break;
214  default:
215  break;
216  }
217  }
218  else
219  m_summonTimer -= diff;
220  }
221 
222  switch (m_phase)
223  {
224  case PHASE_SPEECH:
225  if (m_speechTimer < diff)
226  {
227  switch (m_speech)
228  {
229  case 1: DoScriptText(SAY_SPEECH_1, m_creature); m_speechTimer = 4 * IN_MILLISECONDS; break;
230  case 2: DoScriptText(SAY_SPEECH_2, m_creature); m_speechTimer = 6 * IN_MILLISECONDS; break;
231  case 3: DoScriptText(SAY_SPEECH_3, m_creature); m_speechTimer = 5 * IN_MILLISECONDS; break;
232  case 4: DoScriptText(SAY_SPEECH_4, m_creature); m_phase = PHASE_BALCONY; break;
233  }
234  m_speech++;
235  }
236  else
237  m_speechTimer -= diff;
238 
239  // No break here
240 
241  case PHASE_BALCONY: // Do nothing but wait to teleport down: summoning is handled by instance script
242  if (m_phaseTimer < diff)
243  {
244  m_phase = PHASE_TELEPORT_DOWN;
245  m_phaseTimer = 0;
246  }
247  else
248  m_phaseTimer -= diff;
249 
250  break;
251 
252  case PHASE_TELEPORT_DOWN:
253  if (m_phaseTimer < diff)
254  {
255  if (DoCastSpellIfCan(m_creature, SPELL_TELEPORT_RIGHT, CAST_TRIGGERED) == CAST_OK)
256  {
257  m_phase = m_instance ? PHASE_TELEPORTING : PHASE_STOP_TELEPORTING;
258 
259  DoScriptText(SAY_TELEPORT, m_creature);
260 
261  // Remove Immunity
262  m_creature->RemoveAurasDueToSpell(SPELL_IMMUNE_ALL);
263  SetMeleeEnabled(true);
264  SetCombatMovement(true);
265  DoResetThreat();
266  m_creature->SetInCombatWithZone();
267  }
268  }
269  else
270  m_phaseTimer -= diff;
271 
272  break;
273 
274  case PHASE_TELEPORTING: // Phase is only reached if m_instance is valid
275  if (m_teleportTimer < diff)
276  {
277  uint32 teleportSpellId = m_instance->IsInRightSideGothikArea(m_creature) ? SPELL_TELEPORT_LEFT : SPELL_TELEPORT_RIGHT;
278  if (DoCastSpellIfCan(m_creature, teleportSpellId) == CAST_OK)
279  {
280  m_teleportTimer = urand(30, 45) * IN_MILLISECONDS; // Teleports between 30 seconds and 45 seconds.
281  m_shadowboltTimer = 2 * IN_MILLISECONDS;
282  ++m_teleportCount;
283  }
284  }
285  else
286  m_teleportTimer -= diff;
287 
288  // Second time that Gothik teleports back from dead side to living side or less than 30% HP: open the central gate
289  if (m_teleportCount >= 4 || m_creature->GetHealthPercent() <= 30.0f)
290  {
291  m_phase = PHASE_STOP_TELEPORTING;
292  m_instance->SetData(TYPE_GOTHIK, SPECIAL);
293  }
294  // no break here
295 
296  case PHASE_STOP_TELEPORTING:
297  if (m_harvestSoulTimer < diff)
298  {
299  if (DoCastSpellIfCan(m_creature, SPELL_HARVESTSOUL) == CAST_OK)
300  m_harvestSoulTimer = 15 * IN_MILLISECONDS;
301  }
302  else
303  m_harvestSoulTimer -= diff;
304 
305  if (m_shadowboltTimer)
306  {
307  if (m_shadowboltTimer <= diff)
308  m_shadowboltTimer = 0;
309  else
310  m_shadowboltTimer -= diff;
311  }
312  // Shadowbolt cooldown finished, cast when ready
313  else if (!m_creature->IsNonMeleeSpellCasted(true))
314  {
315  // Select valid target
316  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SHADOWBOLT, SELECT_FLAG_IN_LOS))
317  DoCastSpellIfCan(target, SPELL_SHADOWBOLT);
318  }
319  break;
320  }
321  }

329 bool EffectDummyCreature_spell_anchor(Unit* /*caster*/, uint32 spellId, SpellEffectIndex effIndex, Creature* creatureTarget, ObjectGuid /*originalCasterGuid*/)
330 {
331  if (effIndex != EFFECT_INDEX_0 || creatureTarget->GetEntry() != NPC_SUB_BOSS_TRIGGER)
332  return true;
333 
334  instance_naxxramas* instance = (instance_naxxramas*)creatureTarget->GetInstanceData();
335 
336  if (!instance)
337  return true;
338 
339  switch (spellId)
340  {
341  case SPELL_A_TO_ANCHOR_1: // trigger mobs at high right side
342  case SPELL_B_TO_ANCHOR_1:
343  case SPELL_C_TO_ANCHOR_1:
344  {
345  if (Creature* pAnchor2 = instance->GetClosestAnchorForGothik(creatureTarget, false))
346  {
347  uint32 triggered = SPELL_A_TO_ANCHOR_2;
348 
349  if (spellId == SPELL_B_TO_ANCHOR_1)
350  triggered = SPELL_B_TO_ANCHOR_2;
351  else if (spellId == SPELL_C_TO_ANCHOR_1)
352  triggered = SPELL_C_TO_ANCHOR_2;
353 
354  creatureTarget->CastSpell(pAnchor2, triggered, TRIGGERED_OLD_TRIGGERED);
355  }
356 
357  return true;
358  }
359  case SPELL_A_TO_ANCHOR_2: // trigger mobs at high left side
360  case SPELL_B_TO_ANCHOR_2:
361  case SPELL_C_TO_ANCHOR_2:
362  {
363  CreatureList listTargets;
364  instance->GetGothikSummonPoints(listTargets, false);
365 
366  if (!listTargets.empty())
367  {
368  CreatureList::iterator itr = listTargets.begin();
369  uint32 position = urand(0, listTargets.size() - 1);
370  advance(itr, position);
371 
372  if (Creature* target = (*itr))
373  {
374  uint32 triggered = SPELL_A_TO_SKULL;
375 
376  if (spellId == SPELL_B_TO_ANCHOR_2)
377  triggered = SPELL_B_TO_SKULL;
378  else if (spellId == SPELL_C_TO_ANCHOR_2)
379  triggered = SPELL_C_TO_SKULL;
380 
381  creatureTarget->CastSpell(target, triggered, TRIGGERED_OLD_TRIGGERED);
382  }
383  }
384  return true;
385  }
386  case SPELL_A_TO_SKULL: // final destination trigger mob
387  case SPELL_B_TO_SKULL:
388  case SPELL_C_TO_SKULL:
389  {
390  uint32 summonSpellId = SPELL_SUMMON_SPECTRAL_TRAINEE;
391 
392  if (spellId == SPELL_B_TO_SKULL)
393  summonSpellId = SPELL_SUMMON_SPECTRAL_KNIGHT;
394  else if (spellId == SPELL_C_TO_SKULL)
395  summonSpellId = SPELL_SUMMON_SPECTRAL_RIVENDARE;
396 
397  creatureTarget->CastSpell(creatureTarget, summonSpellId, TRIGGERED_OLD_TRIGGERED);
398 
399  return true;
400  }
401  case SPELL_RESET_GOTHIK_EVENT:
402  {
403  creatureTarget->RemoveAurasDueToSpell(SPELL_SUMMON_TRAINEE);
404  creatureTarget->RemoveAurasDueToSpell(SPELL_SUMMON_KNIGHT);
405  creatureTarget->RemoveAurasDueToSpell(SPELL_SUMMON_MOUNTED_KNIGHT);
406  return true;
407  }
408 
409  }
410 
411  return true;

448 struct CheckGothikSide : public SpellScript
449 {
450  void OnEffectExecute(Spell* spell, SpellEffectIndex /* effIdx */) const override
451  {
452  if (Unit* caster = spell->GetCaster())
453  {
454  if (instance_naxxramas* instance = (instance_naxxramas*)caster->GetInstanceData())
455  {
456  // If central gate is open, attack any one (central gate is only closed during IN_PROGRESS)
457  if (instance->GetData(TYPE_GOTHIK) != IN_PROGRESS)
458  {
459  ((Creature*)caster)->SetInCombatWithZone();
460  caster->AI()->AttackClosestEnemy();
461  }
462  // Else look for a target in the side the summoned NPC is
463  else
464  {
465  Map::PlayerList const& playersList = instance->instance->GetPlayers();
466  if (playersList.isEmpty())
467  return;
468 
469  for (const auto& playerInList : playersList)
470  {
471  if (Player* player = playerInList.getSource())
472  {
473  if (!player->IsAlive())
474  return;
475 
476  if (spell->m_spellInfo->Id == SPELL_CHECK_UNRELENTING_SIDE && instance->IsInRightSideGothikArea(player))
477  caster->AI()->AttackClosestEnemy();
478  else if (spell->m_spellInfo->Id == SPELL_CHECK_SPECTRAL_SIDE && !instance->IsInRightSideGothikArea(player))
479  caster->AI()->AttackClosestEnemy();
480  else
481  return;
482  }
483  }
484  }
485  }
486  }
487  }
