31 void RemoveShadowEmbraceIfNecessary(Aura* aura)
32 {
33  bool hasOtherShadowEmbraceCausingAuras = false;
34  Unit::AuraList const& auraPeriodicDmg = aura->GetTarget()->GetAurasByType(SPELL_AURA_PERIODIC_DAMAGE);
35  for (Unit::AuraList::const_iterator itr = auraPeriodicDmg.begin(); itr != auraPeriodicDmg.end(); ++itr)
36  {
37  if ((*itr)->GetCasterGuid() == aura->GetCasterGuid() && std::find(std::begin(causesShadowEmbraceSpellList), std::end(causesShadowEmbraceSpellList), (*itr)->GetId()) != std::end(causesShadowEmbraceSpellList))
38  {
39  hasOtherShadowEmbraceCausingAuras = true;
40  break;
41  }
42  }
43 
44  // if this is the final Shadow Embrace causing DoT being removed, also remove Shadow Embrace if found
45  if (!hasOtherShadowEmbraceCausingAuras)
46  {
47  uint32 shadowEmbraceSpellId = 0;
48  Unit::AuraList const& auraPctDmgDone = aura->GetTarget()->GetAurasByType(SPELL_AURA_MOD_DAMAGE_PERCENT_DONE);
49  for (Unit::AuraList::const_iterator itr = auraPctDmgDone.begin(); itr != auraPctDmgDone.end(); ++itr)
50  {
51  if ((*itr)->GetCasterGuid() == aura->GetCasterGuid() && std::find(std::begin(shadowEmbraceSpellList), std::end(shadowEmbraceSpellList), (*itr)->GetId()) != std::end(shadowEmbraceSpellList))
52  {
53  shadowEmbraceSpellId = (*itr)->GetId();
54  break;
55  }
56  }
57  if (shadowEmbraceSpellId)
58  aura->GetTarget()->RemoveAurasDueToSpell(shadowEmbraceSpellId);
59  }

278  if (aura->GetEffIndex() != EFFECT_INDEX_1)
279  return SPELL_AURA_PROC_OK;

281  // if damage is more than need
282  if (mod->m_amount <= (int32)procData.damage)
283  {
284  // remember guid before aura delete
285  ObjectGuid casterGuid = procData.triggeredByAura->GetCasterGuid();
286 
287  // Remove aura (before cast for prevent infinite loop handlers)
288  procData.victim->RemoveAurasByCasterSpell(procData.triggeredByAura->GetId(), procData.triggeredByAura->GetCasterGuid());
289 
290  // Cast finish spell (triggeredByAura already not exist!)
291  if (Unit* caster = procData.triggeredByAura->GetCaster())
292  caster->CastSpell(procData.victim, SPELL_SEED_DAMAGE, TRIGGERED_IGNORE_GCD | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL | TRIGGERED_IGNORE_CASTER_AURA_STATE | TRIGGERED_HIDE_CAST_IN_COMBAT_LOG);
293  return SPELL_AURA_PROC_OK; // no hidden cooldown
294  }
295
