76  uint32 GetSubsequentActionTimer(uint32 id)
77  {
78  switch (id)

80  case GLUTH_MORTAL_WOUND: return 10u * IN_MILLISECONDS;

129  void ExecuteAction(uint32 action) override
130  {
131  if (!m_instance)
132  return;
133 
134  if (m_creature->HasAuraType(SPELL_AURA_MOD_STUN))
135  {
136  DelayCombatAction(action, 500u);
137  return;
138  }
139 
140  switch (action)
141  {
142  case GLUTH_MORTAL_WOUND:
143  {
144  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_MORTALWOUND) == CAST_OK)
145  ResetCombatAction(action, GetSubsequentActionTimer(action));
146  return;
147  }
148  case GLUTH_ENRAGE:
149  {
150  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
151  {
152  DoScriptText(EMOTE_BOSS_GENERIC_ENRAGED, m_creature);
153  ResetCombatAction(action, GetSubsequentActionTimer(action));
154  }
155  return;
156  }
157  case GLUTH_ROAR:
158  {
159  if (DoCastSpellIfCan(m_creature, SPELL_TERRIFYING_ROAR) == CAST_OK)
160  ResetCombatAction(action, GetSubsequentActionTimer(action));
161  return;
162  }
163  case GLUTH_DECIMATE:
164  {
165  if (DoCastSpellIfCan(m_creature, SPELL_DECIMATE, CAST_TRIGGERED) == CAST_OK)
166  ResetCombatAction(action, GetSubsequentActionTimer(action));
167  return;
168  }
169  case GLUTH_BERSERK:
170  {
171  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
172  ResetCombatAction(action, GetSubsequentActionTimer(action));
173  return;
174  }
175  }

179 // Reduce all players and Zombie Chow NPCs HP to 5% max HP
180 struct Decimate : public SpellScript
181 {
182  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx ) const override
183  {
184  if (effIdx == EFFECT_INDEX_0)
185  {
186  if (Unit* unitTarget = spell->GetUnitTarget())
187  {
188  // Return if not player, pet nor Zombie Chow NPC
189  if (unitTarget->GetTypeId() == TYPEID_UNIT && !unitTarget->IsControlledByPlayer() && unitTarget->GetEntry() != NPC_ZOMBIE_CHOW)
190  return;
191 
192  int32 damage = unitTarget->GetHealth() - unitTarget->GetMaxHealth() * 0.05f;
193  if (damage > 0)
194  spell->GetCaster()->CastCustomSpell(unitTarget, SPELL_DECIMATE_DAMAGE, &damage, nullptr, nullptr, TRIGGERED_INSTANT_CAST);
195  }
196  }
197  }

218 struct ZombieChowSearch : public SpellScript
219 {
220  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx ) const override
221  {
222  if (effIdx == EFFECT_INDEX_0)
223  {
224  if (Unit* unitTarget = spell->GetUnitTarget())
225  {
226  if (!unitTarget->IsAlive())
227  return;
228 
229  Unit* caster = spell->GetCaster();
230 
231  caster->SetTarget(nullptr);
232  caster->SetFacingToObject(unitTarget);
233  if (caster->CastSpell(unitTarget, SPELL_ZOMBIE_CHOW_SEARCH_KILL, TRIGGERED_NONE) == SPELL_CAST_OK) // Zombie Chow Search - Insta kill, single target
234  {
235  DoScriptText(EMOTE_DEVOURS_ZOMBIE_CHOW, spell->GetCaster(), unitTarget);
236  caster->CastSpell(caster, SPELL_ZOMBIE_CHOW_SEARCH_HEAL, TRIGGERED_INSTANT_CAST);
237  }
238  }
