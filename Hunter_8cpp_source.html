25  {
26  if (data.effIdx == EFFECT_INDEX_2)
27  {
28  int32 auraValue = 0;
29  if (data.aura)
30  {
31  if (Aura* otherAura = data.aura->GetHolder()->m_auras[EFFECT_INDEX_1])
32  auraValue = otherAura->GetAmount(); // fetch ranged aura AP
33  }
34  else if (data.caster) // or newly calculate it
35  auraValue = data.caster->CalculateSpellEffectValue(data.target, data.spellProto, EFFECT_INDEX_1, nullptr, true, false);
36 
37  if (Unit* caster = data.caster)
38  {
39  Unit::AuraList const& classScriptAuras = caster->GetAurasByType(SPELL_AURA_OVERRIDE_CLASS_SCRIPTS);
40  for (auto& aura : classScriptAuras)
41  {
42  switch (aura->GetModifier()->m_miscvalue)
43  {
44  case 5236:
45  case 5237:
46  case 5238:
47  case 5239:
48  case 5240:
49  value = aura->GetModifier()->m_amount * auraValue / 100;

51  }
52  }
53  }
54  }

56  }

86 
87  // additional damage from pet to pet target

89  if (!pet->GetVictim())
