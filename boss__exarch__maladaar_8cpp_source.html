170  void SpellHitTarget(Unit* target, const SpellEntry* spellInfo) override
171  {
172  if (spellInfo->Id == SPELL_STOLEN_SOUL && target->IsPlayer())

174  }
175 
176  void OnSpellCast(SpellEntry const* spellInfo, Unit* target) override
177  {
178  if (spellInfo->Id == SPELL_STOLEN_SOUL && urand(0, 1))
179  DoBroadcastText(urand(0, 1) ? SAY_ROAR : SAY_SOUL_CLEAVE, m_creature);
180  }
181 
182  void ExecuteAction(uint32 action) override
183  {
184  if (action == MALADAAR_SUMMON_AVATAR && m_creature->GetHealthPercent() < 25.0f)
185  {
186  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_AVATAR) == CAST_OK)
187  {
188  DoBroadcastText(SAY_SUMMON, m_creature);
189  SetActionReadyStatus(action, false);
190  }
191  }
192  }
