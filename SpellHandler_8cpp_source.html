33 void WorldSession::HandleUseItemOpcode(WorldPacket& recvPacket)
34 {
35  uint8 bagIndex, slot;
36  uint8 spell_index; // item spell index which should be used
37  uint8 cast_count; // next cast if exists (single or not)
38  ObjectGuid itemGuid;
39 
40  recvPacket >> bagIndex >> slot >> spell_index >> cast_count >> itemGuid;
41 
42  // TODO: add targets.read() check
43  Player* pUser = _player;
44 
45  // ignore for remote control state
46  if (!pUser->IsSelfMover())
47  {
48  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
49  return;
50  }
51 
52  Item* pItem = pUser->GetItemByPos(bagIndex, slot);
53  if (!pItem)
54  {
55  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
56  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
57  return;
58  }
59 
60  if (pItem->GetObjectGuid() != itemGuid)
61  {
62  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
63  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
64  return;
65  }
66 
67  DETAIL_LOG("WORLD: CMSG_USE_ITEM packet, bagIndex: %u, slot: %u, spell_index: %u , Item: %u, data length = %u", bagIndex, slot, spell_index, pItem->GetEntry(), (uint32)recvPacket.size());
68 
69  ItemPrototype const* proto = pItem->GetProto();
70  if (!proto)
71  {
72  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
73  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, nullptr);
74  return;
75  }
76 
77  // some item classes can be used only in equipped state
78  if (proto->InventoryType != INVTYPE_NON_EQUIP && !pItem->IsEquipped())
79  {
80  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
81  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, nullptr);
82  return;
83  }
84 
85  InventoryResult msg = pUser->CanUseItem(pItem);
86  if (msg != EQUIP_ERR_OK)
87  {
88  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
89  pUser->SendEquipError(msg, pItem, nullptr);
90  return;
91  }
92 
93  // not allow use item from trade (cheat way only)
94  if (pItem->IsInTrade())
95  {
96  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
97  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, nullptr);
98  return;
99  }
100 
101  // only allow conjured consumable, bandage, poisons (all should have the 2^21 item flag set in DB)
102  if (proto->Class == ITEM_CLASS_CONSUMABLE &&
103  !(proto->Flags & ITEM_FLAG_IGNORE_DEFAULT_ARENA_RESTRICTIONS) &&
104  pUser->InArena())
105  {
106  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
107  pUser->SendEquipError(EQUIP_ERR_NOT_DURING_ARENA_MATCH, pItem, nullptr);
108  return;
109  }
110 
111  if (pUser->IsInCombat())
112  {
113  for (const auto& Spell : proto->Spells)
114  {
115  if (SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(Spell.SpellId))
116  {
117  if (IsNonCombatSpell(spellInfo))
118  {
119  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
120  pUser->SendEquipError(EQUIP_ERR_NOT_IN_COMBAT, pItem, nullptr);
121  return;
122  }
123  }
124  }
125  }
126 
127  // check also BIND_WHEN_PICKED_UP and BIND_QUEST_ITEM for .additem or .additemset case by GM (not binded at adding to inventory)
128  if (pItem->GetProto()->Bonding == BIND_WHEN_USE || pItem->GetProto()->Bonding == BIND_WHEN_PICKED_UP || pItem->GetProto()->Bonding == BIND_QUEST_ITEM)
129  {
130  if (!pItem->IsSoulBound())
131  {
132  pItem->SetState(ITEM_CHANGED, pUser);
133  pItem->SetBinding(true);
134  }
135  }
136 
137  SpellCastTargets targets;
138 
139  recvPacket >> targets.ReadForCaster(pUser);
140 
141  targets.Update(pUser);
142 
143  if (!pItem->IsTargetValidForItemUse(targets.getUnitTarget()))
144  {
145  // free gray item after use fail
146  pUser->SendEquipError(EQUIP_ERR_NONE, pItem, nullptr);
147 
148  // search spell for spell error
149  uint32 spellid = 0;
150  for (int i = 0; i < MAX_ITEM_PROTO_SPELLS; ++i)
151  {
152  if (proto->Spells[i].SpellTrigger == ITEM_SPELLTRIGGER_ON_USE)
153  {
154  spellid = proto->Spells[i].SpellId;
155  break;
156  }
157  }
158 
159  // send spell error
160  if (SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellid))
161  Spell::SendCastResult(_player, spellInfo, cast_count, SPELL_FAILED_BAD_TARGETS);
162  return;
163  }
164 
165  _player->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_ITEM_USE);
166 
167  // Note: If script stop casting it must send appropriate data to client to prevent stuck item in gray state.
168  if (!sScriptDevAIMgr.OnItemUse(pUser, pItem, targets))
169  {
170  // no script or script not process request by self
171  pUser->CastItemUseSpell(pItem, targets, cast_count, spell_index);
172  }
173 }

181 void WorldSession::HandleOpenItemOpcode(WorldPacket& recvPacket)
182 {
183  DETAIL_LOG("WORLD: CMSG_OPEN_ITEM packet, data length = " SIZEFMTD, recvPacket.size());
184 
185  uint8 bagIndex, slot;
186 
187  recvPacket >> bagIndex >> slot;
188 
189  DETAIL_LOG("bagIndex: %u, slot: %u", bagIndex, slot);
190 
191  Player* pUser = _player;
192 
193  // ignore for remote control state
194  if (!pUser->IsSelfMover())
195  return;
196 
197  Item* pItem = pUser->GetItemByPos(bagIndex, slot);
198  if (!pItem)
199  {
200  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
201  return;
202  }
203 
204  ItemPrototype const* proto = pItem->GetProto();
205  if (!proto)
206  {
207  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, nullptr);
208  return;
209  }
210 
211  // locked item
212  uint32 lockId = proto->LockID;
213  if (lockId && !pItem->HasFlag(ITEM_FIELD_FLAGS, ITEM_DYNFLAG_UNLOCKED))
214  {
215  LockEntry const* lockInfo = sLockStore.LookupEntry(lockId);
216 
217  if (!lockInfo)
218  {
219  pUser->SendEquipError(EQUIP_ERR_ITEM_LOCKED, pItem, nullptr);
220  sLog.outError("WORLD::OpenItem: item [guid = %u] has an unknown lockId: %u!", pItem->GetGUIDLow(), lockId);
221  return;
222  }
223 
224  // required picklocking
225  if (lockInfo->Skill[1] || lockInfo->Skill[0])
226  {
227  pUser->SendEquipError(EQUIP_ERR_ITEM_LOCKED, pItem, nullptr);
228  return;
229  }
230  }
231 
232  if (pItem->HasFlag(ITEM_FIELD_FLAGS, ITEM_DYNFLAG_WRAPPED))// wrapped?
233  {
234  QueryResult* result = CharacterDatabase.PQuery("SELECT entry, flags FROM character_gifts WHERE item_guid = '%u'", pItem->GetGUIDLow());
235  if (result)
236  {
237  Field* fields = result->Fetch();
238  uint32 entry = fields[0].GetUInt32();
239  uint32 flags = fields[1].GetUInt32();
240 
241  pItem->SetGuidValue(ITEM_FIELD_GIFTCREATOR, ObjectGuid());
242  pItem->SetEntry(entry);
243  pItem->SetUInt32Value(ITEM_FIELD_FLAGS, flags);
244  pItem->SetState(ITEM_CHANGED, pUser);
245  delete result;
246  }
247  else
248  {
249  sLog.outError("Wrapped item %u don't have record in character_gifts table and will deleted", pItem->GetGUIDLow());
250  pUser->DestroyItem(pItem->GetBagSlot(), pItem->GetSlot(), true);
251  return;
252  }
253 
254  static SqlStatementID delGifts ;
255 
256  SqlStatement stmt = CharacterDatabase.CreateStatement(delGifts, "DELETE FROM character_gifts WHERE item_guid = ?");
257  stmt.PExecute(pItem->GetGUIDLow());
258  }
259  else
260  {
261  Loot*& loot = pItem->m_loot;
262  if (!loot)
263  loot = new Loot(pUser, pItem, LOOT_PICKPOCKETING);
264 
265  loot->ShowContentTo(pUser);
266  }
267 }

269 void WorldSession::HandleGameObjectUseOpcode(WorldPacket& recv_data)
270 {
271  ObjectGuid guid;
272 
273  recv_data >> guid;
274 
275  DEBUG_LOG("WORLD: Received opcode CMSG_GAMEOBJ_USE guid: %s", guid.GetString().c_str());
276 
277  // ignore for remote control state
278  if (!_player->IsSelfMover())
279  return;
280 
281  if (_player->IsBeingTeleported())
282  return;
283 
284  GameObject* obj = _player->GetMap()->GetGameObject(guid);
285  if (!obj)
286  return;
287 
288  if (!obj->IsAtInteractDistance(_player))
289  return;
290 
291  // Additional check preventing exploits (ie loot despawned chests)
292  if (!obj->IsSpawned())
293  {
294  sLog.outError("HandleGameObjectUseOpcode: CMSG_GAMEOBJ_USE for despawned GameObject (Entry %u), didn't expect this to happen.", obj->GetEntry());
295  return;
296  }
297 
298  // Never expect this opcode for some type GO's
299  if (obj->GetGoType() == GAMEOBJECT_TYPE_GENERIC)
300  {
301  sLog.outError("HandleGameObjectUseOpcode: CMSG_GAMEOBJ_USE for not allowed GameObject type %u (Entry %u), didn't expect this to happen.", obj->GetGoType(), obj->GetEntry());
302  return;
303  }
304 
305  if (obj->HasFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED)) // we should not allow use of a locked GO
306  return;
307 
308  if (obj->HasFlag(GAMEOBJECT_FLAGS, GO_FLAG_IN_USE))
309  return;
310 
311  // Never expect this opcode for non intractable GO's
312  if (obj->HasFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT))
313  {
314  sLog.outError("HandleGameObjectUseOpcode: CMSG_GAMEOBJ_USE for GameObject (Entry %u) with non intractable flag (Flags %u), didn't expect this to happen.", obj->GetEntry(), obj->GetUInt32Value(GAMEOBJECT_FLAGS));
315  return;
316  }
317 
318  // client checks this but needs recheck
319  if (obj->GetGOInfo()->CannotBeUsedUnderImmunity() && _player->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE))
320  return;
321 
322  // code meant to be in CanUseNow
323  if (obj->GetGoType() == GAMEOBJECT_TYPE_CHAIR)
324  {
325  float x, y;
326  std::tie(x, y) = obj->GetClosestChairSlotPosition(_player);
327  if (_player->GetDistance(x, y, obj->GetPositionZ(), DIST_CALC_NONE) > 3.f * 3.f)
328  return;
329  }
330 
331  obj->Use(_player);
332 }

334 void WorldSession::HandleCastSpellOpcode(WorldPacket& recvPacket)
335 {
336  uint32 spellId;
337  uint8 cast_count;
338  recvPacket >> spellId;
339  recvPacket >> cast_count;
340 
341  // ignore for remote control state (for player case)
342  Unit* mover = _player->GetMover();
343  if (mover != _player && mover->GetTypeId() == TYPEID_PLAYER)
344  {
345  recvPacket.rpos(recvPacket.wpos()); // prevent spam at ignore packet
346  return;
347  }
348 
349  DEBUG_LOG("WORLD: CMSG_CAST_SPELL, spellId - %u, cast_count: %u data length = " SIZEFMTD,
350  spellId, cast_count, recvPacket.size());
351 
352  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellId);
353  if (!spellInfo)
354  {
355  sLog.outError("WORLD: unknown spell id %u", spellId);
356  recvPacket.rpos(recvPacket.wpos()); // prevent spam at ignore packet
357  return;
358  }
359 
360  Unit* caster = mover;
361  if (mover->GetTypeId() == TYPEID_PLAYER)
362  {
363  // not have spell in spellbook or spell passive and not casted by client
364  if (!((Player*)mover)->HasActiveSpell(spellId) || IsPassiveSpell(spellInfo))
365  {
366  sLog.outError("World: Player %u casts spell %u which he shouldn't have", mover->GetGUIDLow(), spellId);
367  // cheater? kick? ban?
368  recvPacket.rpos(recvPacket.wpos()); // prevent spam at ignore packet
369  return;
370  }
371  }
372  else
373  {
374  bool isPassive = IsPassiveSpell(spellInfo);
375  // not have spell in spellbook or spell passive and not casted by client
376  if (!mover->HasSpell(spellId) || isPassive)
377  {
378  if (!_player->HasSpell(spellId) || isPassive)
379  {
380  // cheater? kick? ban?
381  recvPacket.rpos(recvPacket.wpos()); // prevent spam at ignore packet
382  return;
383  }
384  else
385  caster = _player;
386  }
387  }
388 
389  // client provided targets
390  SpellCastTargets targets;
391 
392 #ifdef BUILD_PLAYERBOT
393  recvPacket >> targets.ReadForCaster(mover);
394 #else
395  recvPacket >> targets.ReadForCaster(_player);
396 #endif
397 
398  // auto-selection buff level base at target level (in spellInfo)
399  if (Unit* target = targets.getUnitTarget())
400  {
401  // if rank not found then function return nullptr but in explicit cast case original spell can be casted and later failed with appropriate error message
402  if (SpellEntry const* actualSpellInfo = sSpellMgr.SelectAuraRankForLevel(spellInfo, target->GetLevel()))
403  spellInfo = actualSpellInfo;
404  }
405 
406  if (HasMissingTargetFromClient(spellInfo))
407  targets.setUnitTarget(mover->GetTarget());
408 
409  if (_player->HasQueuedSpell())
410  return;
411 
412  bool handled = false;
413  Spell* spell = new Spell(caster, spellInfo, TRIGGERED_NONE);
414  spell->m_cast_count = cast_count; // set count of casts
415  spell->m_clientCast = true;
416  if (caster->HasGCD(spellInfo) || !caster->IsSpellReady(*spellInfo))
417  {
418  if (caster->HasGCDOrCooldownWithinMargin(*spellInfo))
419  {
420  handled = true;
421  _player->SetQueuedSpell(spell);
422  GetMessager().AddMessage([guid = caster->GetObjectGuid(), isPlayer = caster != mover, targets = targets](WorldSession* session) mutable
423  {
424  if (session->GetPlayer()) // in case of logout
425  {
426  // in case of mind control end
427  if ((isPlayer && session->GetPlayer()->GetObjectGuid() == guid) || (!isPlayer && session->GetPlayer()->GetMover()->GetObjectGuid() == guid))
428  session->GetPlayer()->CastQueuedSpell(targets);
429  else
430  session->GetPlayer()->ClearQueuedSpell();
431  }
432  });
433  }
434  }
435 
436  if (!handled)
437  spell->SpellStart(&targets);
438 }

458 void WorldSession::HandleCancelAuraOpcode(WorldPacket& recvPacket)
459 {
460  uint32 spellId;
461  recvPacket >> spellId;
462 
463  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellId);
464  if (!spellInfo)
465  return;
466 
467  if (spellInfo->HasAttribute(SPELL_ATTR_NO_AURA_CANCEL))
468  return;
469 
470  if (IsPassiveSpell(spellInfo))
471  return;
472 
473  SpellAuraHolder* holder = _player->GetSpellAuraHolder(spellId);
474 
475  if (!holder)
476  return;
477 
478  if (!holder->IsPositive())
479  {
480  // ignore for remote control state
481  if (!_player->IsSelfMover())
482  {
483  // except own aura spells
484  bool allow = false;
485  for (unsigned int k : spellInfo->EffectApplyAuraName)
486  {
487  if (k == SPELL_AURA_MOD_POSSESS ||
488  k == SPELL_AURA_MOD_POSSESS_PET)
489  {
490  allow = true;
491  break;
492  }
493  }
494 
495  // this also include case when aura not found
496  if (!allow)
497  return;
498  }
499  else
500  return;
501  }
502 
503  // channeled spell case (it currently casted then)
504  if (IsChanneledSpell(spellInfo))
505  {
506  if (Spell* curSpell = _player->GetCurrentSpell(CURRENT_CHANNELED_SPELL))
507  if (curSpell->m_spellInfo->Id == spellId)
508  _player->InterruptSpell(CURRENT_CHANNELED_SPELL);
509  return;
510  }
511 
512  // not own area auras can't be cancelled (note: maybe need to check for aura on holder and not general on spell)
513  if (holder && holder->GetCasterGuid() != _player->GetObjectGuid() && HasAreaAuraEffect(holder->GetSpellProto()))
514  return;
515 
516  // non channeled case
517  _player->RemoveAurasDueToSpellByCancel(spellId);
518 }

613 void WorldSession::HandleGetMirrorimageData(WorldPacket& recv_data)

616
