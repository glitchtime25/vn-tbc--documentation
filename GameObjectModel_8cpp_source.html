42 void LoadGameObjectModelList()
43 {
44  FILE* model_list_file = fopen((sWorld.GetDataPath() + "vmaps/" + VMAP::GAMEOBJECT_MODELS).c_str(), "rb");
45  if (!model_list_file)
46  return;
47 
48  uint32 name_length, displayId;
49  char buff[500];
50  while (!feof(model_list_file))
51  {
52  fread(&displayId, sizeof(uint32), 1, model_list_file);
53  fread(&name_length, sizeof(uint32), 1, model_list_file);
54 
55  if (name_length >= sizeof(buff))
56  {
57  sLog.outDebug("File %s seems to be corrupted", VMAP::GAMEOBJECT_MODELS);
58  break;
59  }
60 
61  fread(&buff, sizeof(char), name_length, model_list_file);
62  Vector3 v1, v2;
63  fread(&v1, sizeof(Vector3), 1, model_list_file);
64  fread(&v2, sizeof(Vector3), 1, model_list_file);
65 
66  model_list.insert(ModelList::value_type(displayId, GameobjectModelData(std::string(buff, name_length), AABox(v1, v2))));
67  }
68  fclose(model_list_file);
69 }

77 bool GameObjectModel::initialize(const GameObject* const pGo, const GameObjectDisplayInfoEntry* const pDisplayInfo)
78 {
79  ModelList::const_iterator it = model_list.find(pDisplayInfo->Displayid);
80  if (it == model_list.end())
81  return false;
82 
83  G3D::AABox mdl_box(it->second.bound);
84  // ignore models with no bounds
85  if (mdl_box == G3D::AABox::zero())
86  {
87  sLog.outDebug("Model %s has zero bounds, loading skipped", it->second.name.c_str());
88  return false;
89  }
90 
91  iModel = ((VMAP::VMapManager2*)VMAP::VMapFactory::createOrGetVMapManager())->acquireModelInstance(sWorld.GetDataPath() + "vmaps/", it->second.name);
92 
93  if (!iModel)
94  return false;
95 
96  if (it->second.name.find(".m2") != std::string::npos)
97  iModel->setModelFlags(VMAP::MOD_M2);
98 
99  name = it->second.name;
100  iPos = Vector3(pGo->GetPositionX(), pGo->GetPositionY(), pGo->GetPositionZ());
101  collision_enabled = true;
102  iScale = pGo->GetObjectScale();
103  iInvScale = 1.f / iScale;
104 
105  G3D::Matrix3 iRotation = G3D::Matrix3::fromEulerAnglesZYX(pGo->GetOrientation(), 0, 0);
106  iInvRot = iRotation.inverse();
107  // transform bounding box:
108  mdl_box = AABox(mdl_box.low() * iScale, mdl_box.high() * iScale);
109  AABox rotated_bounds;
110  for (int i = 0; i < 8; ++i)
111  rotated_bounds.merge(iRotation * mdl_box.corner(i));
112 
113  this->iBound = rotated_bounds + iPos;
114 
115 #ifdef SPAWN_CORNERS
116  // test:
117  for (int i = 0; i < 8; ++i)
118  {
119  Vector3 pos(iBound.corner(i));
120  if (Creature* c = const_cast<GameObject*>(pGo)->SummonCreature(24440, pos.x, pos.y, pos.z, 0, TEMPSPAWN_MANUAL_DESPAWN, 0))
121  {

147 bool GameObjectModel::intersectRay(const G3D::Ray& ray, float& MaxDist, bool StopAtFirstHit, bool ignoreM2Model) const
148 {
149  if (!collision_enabled)
150  return false;
151 
152  float time = ray.intersectionTime(iBound);
153  if (time == G3D::inf())
154  return false;
155 
156  // child bounds are defined in object space:
157  Vector3 p = iInvRot * (ray.origin() - iPos) * iInvScale;
158  Ray modRay(p, iInvRot * ray.direction());
159  float distance = MaxDist * iInvScale;
160  bool hit = iModel->IntersectRay(modRay, distance, StopAtFirstHit, ignoreM2Model);
161  if (hit)
162  {
163  distance *= iScale;
164  MaxDist = distance;
165  }
166  return hit;
167 }
