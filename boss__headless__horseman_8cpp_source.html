255  // function to handle toss head phase
256  void DoTossHead()
257  {
258  // in the first transition; spawn the head
259  if (m_creature->HasAura(SPELL_BODY_STAGE_1))
260  m_creature->SummonCreature(NPC_HEAD_OF_HORSEMAN, m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
261 
262  // make head available
263  if (Creature* head = m_creature->GetMap()->GetCreature(m_headGuid))
264  {
265  head->CastSpell(m_creature, SPELL_HEAD_REPOSITIONS, TRIGGERED_OLD_TRIGGERED); // also should use SPELL_COMMAND_HEAD_REPOSITIONS
266  DoCastSpellIfCan(head, SPELL_SEND_HEAD, CAST_TRIGGERED);
267  }
268 
269  // only from second transition we start whirlwind
270  if (m_creature->HasAura(SPELL_BODY_STAGE_2) || m_creature->HasAura(SPELL_BODY_STAGE_3))
271  DoCastSpellIfCan(m_creature, SPELL_WHIRLWIND, CAST_TRIGGERED);
272 
273  // remove head visual and set transition phase auras
274  m_creature->RemoveAurasDueToSpell(SPELL_HEAD_VISUAL);
275  DoCastSpellIfCan(m_creature, SPELL_BODY_REGEN_PROC, CAST_TRIGGERED);
276  DoCastSpellIfCan(m_creature, SPELL_BODY_REGEN, CAST_TRIGGERED);
277  DoCastSpellIfCan(m_creature, SPELL_BODY_REGEN_CONFUSE, CAST_TRIGGERED);
278 
279  m_fightPhase = PHASE_HEAD_TOSS;
280  m_bHeadRequested = false;
281  }
282 
283  // function to handle the head rejoin
284  void DoRejoinHead()
285  {
286  // remove transition auras and set the head visual
287  m_creature->RemoveAurasDueToSpell(SPELL_BODY_REGEN_CONFUSE);
288  m_creature->RemoveAurasDueToSpell(SPELL_BODY_REGEN);
289  m_creature->RemoveAurasDueToSpell(SPELL_BODY_REGEN_PROC);
290  m_creature->RemoveAurasDueToSpell(SPELL_WHIRLWIND);
291 
292  DoScriptText(SAY_REJOINED, m_creature);
293  DoCastSpellIfCan(m_creature, SPELL_BODY_HEAD_VISUAL, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
294 
295  // switch from phase 1 to phase 2
296  if (m_creature->HasAura(SPELL_BODY_STAGE_1))
297  {
298  m_fightPhase = PHASE_CONFLAGRATION;
299  m_creature->RemoveAurasDueToSpell(SPELL_BODY_STAGE_1);
300  DoCastSpellIfCan(m_creature, SPELL_BODY_STAGE_2, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
301  }
302  // switch from phase 2 to phase 3 or repeat phase 3
303  else if (m_creature->HasAura(SPELL_BODY_STAGE_2) || m_creature->HasAura(SPELL_BODY_STAGE_3))
304  {
305  m_fightPhase = PHASE_PUMPKINS;
306  m_creature->RemoveAurasDueToSpell(SPELL_BODY_STAGE_2);
307  DoCastSpellIfCan(m_creature, SPELL_BODY_STAGE_3, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
308  DoCastSpellIfCan(m_creature, SPELL_SUMMON_PUMPKIN, CAST_TRIGGERED);
309  }
310  }

312  void UpdateAI(const uint32 uiDiff) override
313  {
314  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
315  return;
316 
317  switch (m_fightPhase)
318  {
319  case PHASE_PUMPKINS:
320 
321  if (m_uiPumpkinTimer < uiDiff)
322  {
323  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_PUMPKIN) == CAST_OK)
324  m_uiPumpkinTimer = urand(35000, 40000);
325  }
326  else
327  m_uiPumpkinTimer -= uiDiff;
328 
329  // no break;
330  case PHASE_CONFLAGRATION:
331 
332  // conflagration not happening during pumpkin phase
333  if (m_fightPhase != PHASE_PUMPKINS)
334  {
335  if (m_uiConflagrationTimer < uiDiff)
336  {
337  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
338  {
339  if (DoCastSpellIfCan(pTarget, SPELL_CONFLAGRATION) == CAST_OK)
340  {
341  DoScriptText(SAY_CONFLAGRATION, m_creature);
342  m_uiConflagrationTimer = urand(15000, 20000);
343  }
344  }
345  }
346  else
347  m_uiConflagrationTimer -= uiDiff;
348  }
349 
350  // no break;
351  case PHASE_HORSEMAN:
352 
353  // cleave - all phases
354  if (m_uiCleaveTimer < uiDiff)
355  {
356  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_HORSEMAN_CLEAVE) == CAST_OK)
357  m_uiCleaveTimer = 5000;
358  }
359  else
360  m_uiCleaveTimer -= uiDiff;
361 
362  DoMeleeAttackIfReady();
363  break;
364  case PHASE_HEAD_TOSS:
365  // rejoin head by force at 100% hp
366  if (!m_bHeadRequested && m_creature->GetHealthPercent() == 100.0f)
367  {
368  if (Creature* pHead = m_creature->GetMap()->GetCreature(m_headGuid))
369  SendAIEvent(AI_EVENT_CUSTOM_B, m_creature, pHead);
370 
371  m_bHeadRequested = true;
372  }
373  break;
374  }
375  }
