35 void WorldSession::SendGMTicket(const GMTicket& ticket, time_t now/* = time(nullptr)*/) const
36 {
37  float daysTicketAge = sTicketMgr.GetTicketAgeDays(now, ticket.GetCreatedAt());
38  float daysOldestTicketAge = sTicketMgr.GetOldestTicketAgeDays();
39  float daysLastUpdated = sTicketMgr.GetOldestTicketAgeLastUpdateDays(now);
40 
41  WorldPacket data(SMSG_GMTICKET_GETTICKET, 4);
42  data << uint32(GMTICKET_STATUS_HASTEXT);
43  data << ticket.GetText(); // Ticket text: data, should never exceed 1999 bytes
44  data << uint8(ticket.GetCategory()); // Ticket category
45  data << float(daysTicketAge); // Days passed since the ticket got in the queue
46  data << float(daysOldestTicketAge); // Days passed since oldest ticket got in the queue
47  data << float(daysLastUpdated); // Days passed since the last update to oldestTicketTime
48  data << uint8(ticket.GetStatus()); // Escalated status of the ticket
49  data << uint8(ticket.IsSeen()); // Ticket has been read by a gm
50  SendPacket(data);
51 }

69 void WorldSession::HandleGMTicketCreateOpcode(WorldPacket& recv_data)
70 {
71  if (sTicketMgr.GetSystemStatus() != GMTICKET_QUEUE_STATUS_ENABLED || !m_ticketSquelchTimer.Passed())
72  {
73  recv_data.read_skip(recv_data.size());
74  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_ERROR);
75  return;
76  }
77 
78  uint8 category = 0; // TBC+: "<Uncategorized>"
79  uint32 map;
80  float x, y, z;
81  std::string message;
82 
83  recv_data >> map >> x >> y >> z;
84  recv_data >> message;
85 
86  recv_data.read_skip<uint32>(); // TBC+: need response?
87 
88  uint32 chatDataLineCount;
89  recv_data >> chatDataLineCount;
90 
91  uint32 chatDataSizeInflated;
92  recv_data >> chatDataSizeInflated;
93 
94  if (size_t chatDataSizeDeflated = (recv_data.size() - recv_data.rpos()))
95  recv_data.read_skip(chatDataSizeDeflated); // Compressed chat data
96 
97  Player* player = GetPlayer();
98  const ObjectGuid& guid = player->GetObjectGuid();
99 
100  if (!CheckChatMessage(message))
101  {
102  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_ERROR);
103  return;
104  }
105 
106  // Check if open ticket already exists
107  if (GMTicket* existing = sTicketMgr.GetTicketByPlayer(guid))
108  {
109  m_ticketSquelchTimer.Reset(5000);
110  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_ALREADY_EXIST);
111  }
112  else
113  {
114  // First, try to recycle stored abandoned ticket as new: to avoid new/abandon db spam
115  if (GMTicket* abandoned = sTicketMgr.GetTicketByPlayer(player->GetObjectGuid(), GMTICKET_STATE_ABANDONED))
116  {
117  abandoned->Recycle(GetPlayer(), message, category);
118 
119  if (sTicketMgr.Recycle(abandoned))
120  {
121  m_ticketSquelchTimer.Reset(5000);
122  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_SUCCESS);
123  }
124  else
125  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_ERROR);
126  }
127  else
128  {
129  GMTicket* ticket = new GMTicket(GetPlayer(), message, category);
130 
131  if (sTicketMgr.Add(ticket))
132  {
133  m_ticketSquelchTimer.Reset(5000);
134  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_SUCCESS);
135  }
136  else
137  {
138  delete ticket;
139  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_ERROR);
140  }
141  }
142  }
143 }

186 void WorldSession::HandleGMSurveySubmitOpcode(WorldPacket& recv_data)
187 {
188  if (sTicketMgr.GetSystemStatus() != GMTICKET_QUEUE_STATUS_ENABLED || !m_ticketSquelchTimer.Passed())
189  {
190  recv_data.read_skip(recv_data.size());
191  return;
192  }
193 
194  m_ticketSquelchTimer.Reset(5000);
195 
196  // GM survey is shown after SMSG_GM_TICKET_STATUS_UPDATE with status 3
197 
198  uint32 surveyId;
199  recv_data >> surveyId; // Survey ID: found in GMSurveySurveys.dbc
200 
201  // Amount of question-answer pairs has hardcoded a limit of 10
202  std::map<uint32, uint8> answers;
203 
204  for (uint8 i = 0; i < MAX_GMSURVEY_QUESTIONS; ++i)
205  {
206  uint32 questionId;
207  recv_data >> questionId; // Question ID: questions found in GMSurveyQuestions.dbc
208 
209  // Detect end of sequence
210  if (!questionId)
211  break;
212 
213  uint8 answer;
214  recv_data >> answer; // Rating: hardcoded limit of 0-5 in pre-Wrath, ranges defined in GMSurveyAnswers.dbc Wrath+
215 
216  answers.insert({questionId, answer});
217 
218  std::string unused;
219  recv_data >> unused; // Answer comment: Unused in stock UI, can be only set by calling Lua function
220 
221  // utf8limit(unused, 11458); // Answer comment max sizes in bytes: Vanilla - 8106:8110, TBC - 11459:11463, Wrath - 582:586
222  }
223 
224  std::string comment;
225  recv_data >> comment; // Comment
226 
227  if (!CheckChatMessage(comment))
228  return;
229 
230  utf8limit(comment, 11469); // Comment max sizes in bytes: Vanilla - 8117:8121, TBC - 11470:11474, Wrath - 593:597
231 
232  // Perform some sanity checks on the input:
233 
234  GMSurveyCurrentSurveyEntry const* currentSurvey = sGMSurveyCurrentSurveyStore.LookupEntry(uint32(m_sessionDbcLocale));
235 
236  // Check if survey id is correct
237  if (!currentSurvey || currentSurvey->surveyID != surveyId)
238  return;
239 
240  GMSurveyEntry const* survey = sGMSurveySurveysStore.LookupEntry(surveyId);
241 
242  // Check if survey entry with provided id exists
243  if (!survey)
244  return;
245 
246  // Check if all provided question ids match the survey entry
247  {
248  for (auto itr = answers.begin(); itr != answers.end(); ++itr)
249  {
250  bool exists = false;
251 
252  for (uint8 j = 0; (j < MAX_GMSURVEY_QUESTIONS && survey->questionID[j] && !exists); ++j)
253  exists = (survey->questionID[j] == itr->first);
254 
255  if (!exists)
256  return;
257 
258  // Pre-WotLK: answers have hardcoded range of 0-5
259  if (itr->second > 5)
260  return;
261  }
262  }
263 
264  // We only store closed tickets in memory if they are surveyed, so we lookup one:
265  if (GMTicket* surveyed = sTicketMgr.GetTicketByPlayer(GetPlayer()->GetObjectGuid(), GMTICKET_STATE_CLOSED))
266  {
267  GMSurveyResult* result = new GMSurveyResult(*survey, surveyed->GetId(), answers, comment);
268  sTicketMgr.Survey(surveyed, result);
269  delete result;
270  }
271 }
