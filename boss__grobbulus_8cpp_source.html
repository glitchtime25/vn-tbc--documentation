98  {

104  uint32 GetSubsequentActionTimer(uint32 id)
105  {
106  switch (id)
107  {
108  // Mutagen Injection is used more often when below 30% HP
109  case GROBBULUS_INJECTION: return m_creature->GetHealthPercent() > 30.0f ? urand(7, 13) * IN_MILLISECONDS : urand(3, 7) * IN_MILLISECONDS;
110  case GROBBULUS_POISON_CLOUD: return 15u * IN_MILLISECONDS;
111  case GROBBULUS_SLIME_SPRAY: return urand(20, 30) * IN_MILLISECONDS;
112  default: return 0;
113  }
114  }

116  void ExecuteAction(uint32 action) override
117  {
118  switch (action)
119  {
120  case GROBBULUS_SLIME_STREAM:
121  {
122  if (!m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
123  DoCastSpellIfCan(m_creature, SPELL_SLIME_STREAM, CAST_TRIGGERED);
124  ResetCombatAction(action, 3u * IN_MILLISECONDS);
125  break;
126  }
127  case GROBBULUS_INJECTION:
128  {
129  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MUTATING_INJECTION, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_AURA))
130  {
131  if (DoCastSpellIfCan(target, SPELL_MUTATING_INJECTION, CAST_TRIGGERED) == CAST_OK)
132  ResetCombatAction(action, GetSubsequentActionTimer(action));
133  }
134  break;
135  }
136  case GROBBULUS_POISON_CLOUD:
137  {
138  if (DoCastSpellIfCan(m_creature, SPELL_POISON_CLOUD) == CAST_OK)
139  ResetCombatAction(action, GetSubsequentActionTimer(action));
140  break;
141  }
142  case GROBBULUS_SLIME_SPRAY:
143  {
144  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SLIME_SPRAY) == CAST_OK)
145  ResetCombatAction(action, GetSubsequentActionTimer(action));
146  break;
147  }
148  case GROBBULUS_BERSERK:
149  {
150  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
151  DisableCombatAction(action);
152  break;
153  }
154  }
155  }

158 struct MutatingInjection : public AuraScript
159 {
160  void OnApply(Aura* aura, bool apply) const override
161  {
162  if (!apply)
163  {
164  Unit* target = aura->GetTarget();
165  if (!target)
166  return;
167 
168  if (aura->GetRemoveMode() == AURA_REMOVE_BY_EXPIRE)
169  // Embalming Cloud
170  target->CastSpell(target, SPELL_EMBALMING_CLOUD, TRIGGERED_OLD_TRIGGERED, nullptr, aura);
171  else // Removed by dispel
172  // Mutagen Explosion
173  target->CastSpell(target, SPELL_MUTAGEN_EXPLOSION, TRIGGERED_OLD_TRIGGERED, nullptr, aura);
174  // Poison Cloud
175  target->CastSpell(target, SPELL_POISON_CLOUD, TRIGGERED_OLD_TRIGGERED, nullptr, aura);
176  }
177  }
