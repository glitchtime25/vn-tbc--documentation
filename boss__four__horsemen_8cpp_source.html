103 // Coordinates at the entrance of the Four Horsemen room. Beyond these, the Horsemen reset (leashing)
104 static const Location2DPoint resetCoords[3] = {

109 struct boss_horsmenAI : public ScriptedAI
110 {
111  boss_horsmenAI(Creature* creature) : ScriptedAI(creature)
112  {
113  m_instance = (ScriptedInstance*)creature->GetInstanceData();
114  m_creature->GetCombatManager().SetLeashingCheck([&](Unit*, float x, float y, float /*z*/)
115  {
116  return (x > resetCoords[0].x && y < resetCoords[0].y) ||
117  (x > resetCoords[1].x && y < resetCoords[1].y) ||
118  (x > resetCoords[2].x && y < resetCoords[2].y);
119  });
120  Reset();
121  }

123  ScriptedInstance* m_instance;
124 
125  uint32 m_markTimer;
126  uint32 m_markCounter;
127  uint32 m_horsmenIndex; // Used to differentiate each of the Four Horsemen in the children classes
128 
129  float m_healthPercentageCheck;
130 
131  void Reset() override
132  {
133  m_markTimer = 20 * IN_MILLISECONDS;
134  m_markCounter = 0;
135  m_horsmenIndex = INDEX_BLAUMEUX; // Default: Lady Blaumeux (1: Highlord Morgraine, 2: Thane Korth'azz, 3: Sir Zeliek)
136 
137  m_healthPercentageCheck = 50.0f;
138  }
139

141  {
142  DoScriptText(aggroSayList[m_horsmenIndex], m_creature);
143 
144  if (m_instance)

146  }
147 
148  void KilledUnit(Unit* /*victim*/) override
149  {
150  DoScriptText(killSayList[m_horsmenIndex], m_creature);
151  }
152 
153  void JustDied(Unit* /*killer*/) override
154  {
155  DoScriptText(deathSayList[m_horsmenIndex], m_creature);
156  DoCastSpellIfCan(m_creature, spiritSpellList[m_horsmenIndex], CAST_TRIGGERED);
157 
158  if (m_instance)

160  }
161 
162  void EnterEvadeMode() override

165

168  }

170  virtual void UpdateHorsmenAI(const uint32 /*diff*/) {}
171 
172  void UpdateAI(const uint32 diff) override
173  {
174  // Do nothing if no target
175  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
176  return;
177 
178  // Call Horsemen specific virtual function to cast special spells
179  UpdateHorsmenAI(diff);
180 
181  if (m_creature->GetHealthPercent() <= m_healthPercentageCheck)
182  {
183  if (DoCastSpellIfCan(m_creature, SPELL_SHIELDWALL) == CAST_OK)
184  m_healthPercentageCheck -= 30.0f;
185  }
186 
187  if (m_markTimer < diff)
188  {
189  if (DoCastSpellIfCan(m_creature, markSpellList[m_horsmenIndex]) == CAST_OK)
190  {
191  m_markCounter++;
192  m_markTimer = 12 * IN_MILLISECONDS;
193  }
194  }
195  else
196  m_markTimer -= diff;
197 
198  DoMeleeAttackIfReady();
199  }
200 };
