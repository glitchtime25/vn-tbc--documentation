164 bool AreaTrigger_at_murkdeep(Player* pPlayer, AreaTriggerEntry const* /*pAt*/)
165 {
166  // Handle Murkdeep event start
167  // The area trigger summons 3 Greymist Coastrunners; The rest of the event is handled by world map scripts
168  if (pPlayer->IsAlive() && !pPlayer->IsGameMaster() && pPlayer->GetQuestStatus(QUEST_WANTED_MURKDEEP) == QUEST_STATUS_INCOMPLETE)
169  {
170  ScriptedMap* pScriptedMap = (ScriptedMap*)pPlayer->GetInstanceData();
171  if (!pScriptedMap)
172  return false;
173 
174  // If Murkdeep is already spawned, skip the rest
175  if (pScriptedMap->GetSingleCreatureFromStorage(NPC_MURKDEEP, true))
176  return true;
177 
178  // Check if there are already coastrunners (dead or alive) around the area
179  if (GetClosestCreatureWithEntry(pPlayer, NPC_GREYMIST_COASTRUNNNER, 60.0f, false, false))
180  return true;
181 
182  float fX, fY, fZ;
183  for (uint8 i = 0; i < 3; ++i)
184  {
185  // Spawn locations are defined in World Maps Scripts.h
186  pPlayer->GetRandomPoint(aSpawnLocations[POS_IDX_MURKDEEP_SPAWN][0], aSpawnLocations[POS_IDX_MURKDEEP_SPAWN][1], aSpawnLocations[POS_IDX_MURKDEEP_SPAWN][2], 5.0f, fX, fY, fZ);
187 
188  if (Creature* pTemp = pPlayer->SummonCreature(NPC_GREYMIST_COASTRUNNNER, fX, fY, fZ, aSpawnLocations[POS_IDX_MURKDEEP_SPAWN][3], TEMPSPAWN_DEAD_DESPAWN, 0))
189  {
190  pTemp->SetWalk(false);
191  pTemp->GetRandomPoint(aSpawnLocations[POS_IDX_MURKDEEP_MOVE][0], aSpawnLocations[POS_IDX_MURKDEEP_MOVE][1], aSpawnLocations[POS_IDX_MURKDEEP_MOVE][2], 5.0f, fX, fY, fZ);
192  pTemp->GetMotionMaster()->MovePoint(0, fX, fY, fZ);
193  }
194  }
195  }
196 
197  return false;
198 }

296 bool AreaTrigger_at_huldar_miran(Player* pPlayer, AreaTriggerEntry const* /*pAt*/)
297 {
298  // Player is deaed, a GM, quest complete or no quest, do nothing
299  if (!pPlayer->IsAlive() || pPlayer->IsGameMaster() ||
300  pPlayer->GetQuestStatus(QUEST_RESUPPLYING_THE_EXCAVATION) == QUEST_STATUS_COMPLETE ||
301  pPlayer->GetQuestStatus(QUEST_RESUPPLYING_THE_EXCAVATION) == QUEST_STATUS_NONE)
302  return false;
303 
304  ScriptedMap* pScriptedMap = (ScriptedMap*)pPlayer->GetInstanceData();
305  if (!pScriptedMap)
306  return false;
307 
308  Creature* m_miran = GetClosestCreatureWithEntry(pPlayer, NPC_MIRAN, 60.0f, true);
309  Creature* m_huldar = GetClosestCreatureWithEntry(pPlayer, NPC_HULDAR, 60.0f, true);
310  Creature* m_saean = GetClosestCreatureWithEntry(pPlayer, NPC_SAEAN, 60.0f, true);
311 
312  // Quest NPCs not availble, do noting
313  if (!m_miran || !m_huldar)
314  return false;
315 
316  DoScriptText(SAY_MIRAN_AMBUSH, m_miran);
317 
318  // complete quest
319  pPlayer->CompleteQuest(QUEST_RESUPPLYING_THE_EXCAVATION);
320  pPlayer->SendQuestCompleteEvent(QUEST_RESUPPLYING_THE_EXCAVATION);
321 
322  // Quest NPCs in combat, skip the rest, prevent double spawns
323  if (m_miran->IsInCombat() || m_huldar->IsInCombat())
324  return true;
325 
326  // Check if Saean is spawned and set his faction to hostile - summon him if not spawned
327  if (m_saean)
328  m_saean->SetFactionTemporary(FACTION_HOSTILE, TEMPFACTION_RESTORE_RESPAWN);
329  else
330  {
331  m_saean = m_huldar->SummonCreature(NPC_SAEAN, m_miranAmbushSpawns[0].m_fX, m_miranAmbushSpawns[0].m_fY, m_miranAmbushSpawns[0].m_fZ, m_miranAmbushSpawns[0].m_fO, TEMPSPAWN_CORPSE_TIMED_DESPAWN, 25000);
332  if (m_saean)
333  m_saean->SetFactionTemporary(FACTION_HOSTILE, TEMPFACTION_RESTORE_RESPAWN);
334  }
335 
336  // Check if any Dark Iron Ambusher are already spawned or dead, if so, do nothing
337  if (m_saean && !GetClosestCreatureWithEntry(pPlayer, NPC_DARK_IRON_AMBUSHER, 60.0f, false, false))
338  {
339  m_saean->SummonCreature(NPC_DARK_IRON_AMBUSHER, m_miranAmbushSpawns[1].m_fX, m_miranAmbushSpawns[1].m_fY, m_miranAmbushSpawns[1].m_fZ, m_miranAmbushSpawns[1].m_fO, TEMPSPAWN_CORPSE_TIMED_DESPAWN, 25000);
340  m_saean->SummonCreature(NPC_DARK_IRON_AMBUSHER, m_miranAmbushSpawns[2].m_fX, m_miranAmbushSpawns[2].m_fY, m_miranAmbushSpawns[2].m_fZ, m_miranAmbushSpawns[2].m_fO, TEMPSPAWN_CORPSE_TIMED_DESPAWN, 25000);
341  }
342 
343  return true;
344 }
