101 void instance_scarlet_monastery::SetData(uint32 uiType, uint32 uiData)
102 {
103  if (uiType == TYPE_MOGRAINE_AND_WHITE_EVENT)
104  {
105  if (uiData == IN_PROGRESS)
106  DoUseDoorOrButton(GO_WHITEMANE_DOOR);
107  if (uiData == FAIL)
108  {
109  Creature* pWhitemane = GetSingleCreatureFromStorage(NPC_WHITEMANE);
110  if(!pWhitemane)
111  return;
112  Creature* pMograine = GetSingleCreatureFromStorage(NPC_MOGRAINE);
113  if(!pMograine)
114  return;
115  if(pWhitemane->IsAlive() && pMograine->IsAlive())
116  {
117  pWhitemane->SetRespawnDelay(30, true);
118  pWhitemane->ForcedDespawn();
119  pMograine->SetRespawnDelay(30, true);
120  pMograine->ForcedDespawn();
121  DoUseDoorOrButton(GO_WHITEMANE_DOOR);
122  m_auiEncounter[0] = NOT_STARTED;
123  return;
124  }
125  if(!pMograine->IsAlive() && pWhitemane->IsAlive())
126  {
127  pWhitemane->SetRespawnDelay(30, true);
128  pWhitemane->ForcedDespawn();
129  DoUseDoorOrButton(GO_WHITEMANE_DOOR);
130  m_auiEncounter[0] = uiData;
131  return;
132  }
133  if(!pWhitemane->IsAlive() && pMograine->IsAlive())
134  {
135  pMograine->ForcedDespawn();
136  m_auiEncounter[0] = uiData;
137  return;
138  }
139  }
140 
141  m_auiEncounter[0] = uiData;
142  }
143  else if (uiType == TYPE_ASHBRINGER_EVENT)
144  {
145  if (uiData == IN_PROGRESS)
146  {
147  DoUseDoorOrButton(GO_CHAPEL_DOOR);
148 
149  Creature* whitemane = GetSingleCreatureFromStorage(NPC_WHITEMANE, true);
150  if (whitemane && whitemane->IsAlive() && !whitemane->IsInCombat())
151  whitemane->ForcedDespawn();
152 
153  for (auto scarletCathedralNpcGuid : m_sAshbringerFriendlyGuids)
154  if (Creature* scarletNpc = instance->GetCreature(scarletCathedralNpcGuid))
155  if (scarletNpc->IsAlive() && !scarletNpc->IsInCombat())
156  scarletNpc->setFaction(35);
157  }
158  m_auiEncounter[1] = uiData;
159  }
160 }
