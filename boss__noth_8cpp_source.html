149  void Aggro(Unit* /*who*/) override

169  void SummonedCreatureJustDied(Creature* summoned) override

204  void DoSummonPlaguedChampions(bool isThirdWave)
205  {
206  // Four Plagued Champions are to be summoned, one in each of the three corners and the fourth randomly along the western wall
207  // On the third wave, we use specific spell to also summon a Plagued Construct in West part and North-East corner
208  if (DoCastSpellIfCan(m_creature, spellSummonPlaguedChampionSWList[urand(0, 1)], CAST_TRIGGERED) == CAST_OK)
209  ++m_wavesNpcsCount; // We count the addition of a new NPC on each spell cast because JustSummoned() can possibly be called more than once per summoned NPC, making the count there incorrect
210  if (DoCastSpellIfCan(m_creature, spellSummonPlaguedChampionNWList[urand(0, 1)], CAST_TRIGGERED) == CAST_OK)
211  ++m_wavesNpcsCount;
212  if (DoCastSpellIfCan(m_creature, (isThirdWave ? SPELL_SUMMON_CONSTR02 : spellSummonPlaguedChampionNEList[urand(0, 1)]), CAST_TRIGGERED) == CAST_OK)
213  (isThirdWave ? m_wavesNpcsCount += 2 : ++m_wavesNpcsCount);
214  if (DoCastSpellIfCan(m_creature, (isThirdWave ? SPELL_SUMMON_CONSTR01 : spellSummonPlaguedChampionWestList[urand(0, 3)]), CAST_TRIGGERED) == CAST_OK)
215  (isThirdWave ? m_wavesNpcsCount += 2 : ++m_wavesNpcsCount);
216  }
217 
218  void DoSummonPlaguedGuardians()
219  {
220  // Two Plagued Guardians are to be summoned, one in the south-west corner and one in one of the two other corners
221  if (DoCastSpellIfCan(m_creature, spellSummonPlaguedGuardianNENWList[urand(0, 1)], CAST_TRIGGERED) == CAST_OK)
222  ++m_wavesNpcsCount;
223  if (DoCastSpellIfCan(m_creature, spellSummonPlaguedGuardianSWList[urand(0, 1)], CAST_TRIGGERED) == CAST_OK)
224  ++m_wavesNpcsCount;
225  }

227  void HandleBalconyPhase()
228  {
229  SetCombatMovement(false);
230  if (DoCastSpellIfCan(m_creature, SPELL_TELEPORT) == CAST_OK)
231  {
232  DisableCombatAction(NOTH_CURSE);
233  DisableCombatAction(NOTH_BLINK);
234  DoCastSpellIfCan(m_creature, SPELL_IMMUNE_ALL, CAST_TRIGGERED); // Prevent players from damaging Noth when he is on the balcony
235  SetMeleeEnabled(false);
236  StopTargeting(true);
237  m_creature->SetTarget(nullptr);
238  ResetCombatAction(NOTH_SUMMON, 10u * IN_MILLISECONDS); // Summon first wave 10 seconds after teleport
239  ++m_phaseSub;
240  switch (m_phaseSub)
241  {
242  case PHASE_SKELETON_1: ResetTimer(NOTH_GROUND_PHASE, 70u * IN_MILLISECONDS); break;
243  case PHASE_SKELETON_2: ResetTimer(NOTH_GROUND_PHASE, 97u * IN_MILLISECONDS); break;
244  case PHASE_SKELETON_3: ResetTimer(NOTH_GROUND_PHASE, 120u * IN_MILLISECONDS); break;
245  }
246  m_phase = PHASE_BALCONY;
247  }
248  else
249  {
250  SetCombatMovement(true); // Restore combat movement on cast failure
251  ResetTimer(NOTH_BALCONY_PHASE, 1000u);
252  }
253  }
254 
255  void HandleGroundPhase()
256  {
257  if (DoCastSpellIfCan(m_creature, SPELL_TELEPORT_RETURN, CAST_TRIGGERED) == CAST_OK)
258  {
259  m_creature->RemoveAurasDueToSpell(SPELL_IMMUNE_ALL);
260  SetMeleeEnabled(true);
261  SetCombatMovement(true);
262  HandleTargetRestoration();
263  ResetCombatAction(NOTH_SUMMON, 5u * IN_MILLISECONDS); // 5 seconds before summoning Plagued Warriors again
264  ResetCombatAction(NOTH_CURSE, 28u * IN_MILLISECONDS);
265  ResetCombatAction(NOTH_BLINK, 25u * IN_MILLISECONDS);
266  m_subWavesCount = 0;
267  switch (m_phaseSub)
268  {
269  case PHASE_SKELETON_1: ResetTimer(NOTH_BALCONY_PHASE, 110u * IN_MILLISECONDS); break;
270  case PHASE_SKELETON_2: ResetTimer(NOTH_BALCONY_PHASE, 180u * IN_MILLISECONDS); break;
271  case PHASE_SKELETON_3: DisableTimer(NOTH_BALCONY_PHASE); break;
272  }
273  m_phase = PHASE_GROUND;
274  }
275  else
276  {
277  ResetTimer(NOTH_GROUND_PHASE, 1000u);
278  }
279  }
280 
281  void ExecuteAction(uint32 action) override
282  {
283  switch (action)
284  {
285  case NOTH_CURSE:
286  {
287  DoCastSpellIfCan(m_creature, SPELL_CURSE_PLAGUEBRINGER);
288  ResetCombatAction(action, 28u * IN_MILLISECONDS);
289  break;
290  }
291  case NOTH_BLINK:
292  {
293  DoCastSpellIfCan(m_creature, SPELL_CRIPPLE);
294  if (DoCastSpellIfCan(m_creature, spellBlinkList[urand(0, 3)]) == CAST_OK)
295  {
296  DoResetThreat();
297  ResetCombatAction(action, 25u * IN_MILLISECONDS);
298  }
299  break;
300  }
301  case NOTH_SUMMON:
302  {
303  if (m_phase == PHASE_GROUND)
304  {
305  DoScriptText(SAY_SUMMON, m_creature);
306  for (const uint32 spell : { SPELL_SUMMON_WARRIOR_1, SPELL_SUMMON_WARRIOR_2, SPELL_SUMMON_WARRIOR_3 })
307  DoCastSpellIfCan(m_creature, spell, CAST_TRIGGERED);
308  ResetCombatAction(action, 30 * IN_MILLISECONDS);
309  }
310  else
311  {
312  switch (m_phaseSub)
313  {
314  case PHASE_SKELETON_1:
315  {
316  DoSummonPlaguedChampions(false);
317  ResetCombatAction(action, 30 * IN_MILLISECONDS);
318  break;
319  }
320  case PHASE_SKELETON_2:
321  {
322  DoSummonPlaguedChampions(false);
323  DoSummonPlaguedGuardians();
324  ResetCombatAction(action, 45 * IN_MILLISECONDS);
325  break;
326  }
327  case PHASE_SKELETON_3:
328  {
329  DoSummonPlaguedChampions(true); // Set parameter to true because we need to indicate this is the third wave and Plagued Construct need to be also summoned
330  DoSummonPlaguedGuardians();
331  ResetCombatAction(action, 60 * IN_MILLISECONDS);
332  break;
333  }
334  }
335  ++m_subWavesCount; // Keep track of which wave we currently are
336  }
337  break;
338  }
339  default:
340  break;
341  }
342  }
