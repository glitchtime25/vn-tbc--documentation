36 void MapManager::LoadTransports()
37 {
38  sTransportMgr.LoadTransportTemplates();
39 
40  QueryResult* result = WorldDatabase.Query("SELECT entry, name, period FROM transports");
41 
42  uint32 count = 0;
43 
44  if (!result)
45  {
46  BarGoLink bar(1);
47  bar.step();
48  sLog.outString(">> Loaded %u transports", count);
49  sLog.outString();
50  return;
51  }
52 
53  BarGoLink bar(result->GetRowCount());
54 
55  do
56  {
57  bar.step();
58 
59  Field* fields = result->Fetch();
60 
61  uint32 entry = fields[0].GetUInt32();
62  std::string name = fields[1].GetCppString();
63  uint32 period = fields[2].GetUInt32();
64 
65  const GameObjectInfo* goinfo = ObjectMgr::GetGameObjectInfo(entry);
66 
67  if (!goinfo)
68  {
69  sLog.outErrorDb("Transport ID:%u, Name: %s, will not be loaded, gameobject_template missing", entry, name.c_str());
70  continue;
71  }
72 
73  if (goinfo->type != GAMEOBJECT_TYPE_MO_TRANSPORT)
74  {
75  sLog.outErrorDb("Transport ID:%u, Name: %s, will not be loaded, gameobject_template type wrong", entry, name.c_str());
76  continue;
77  }
78 
79  TransportTemplate* transportTemplate = sTransportMgr.GetTransportTemplate(entry);
80  if (!transportTemplate)
81  {
82  sLog.outErrorDb("Transport ID:%u, Name: %s, will not be loaded, transport template missing (core generated)", entry, name.c_str());
83  continue;
84  }
85 
86  const MapEntry* pMapInfo = sMapStore.LookupEntry(transportTemplate->keyFrames.begin()->Node->mapid);
87  if (!pMapInfo)
88  continue;
89 
90  transportTemplate->pathTime = period;
91  transportTemplate->keyFrames.back().DepartureTime = period;
92 
93  m_transportsByMap[pMapInfo->MapID].push_back(transportTemplate);
94 
95  ++count;
96  }
97  while (result->NextRow());
98  delete result;
99 
100  // check transport data DB integrity
101  result = WorldDatabase.Query("SELECT gameobject.guid,gameobject.id,transports.name FROM gameobject,transports WHERE gameobject.id = transports.entry");
102  if (result) // wrong data found
103  {
104  do
105  {
106  Field* fields = result->Fetch();
107 
108  uint32 guid = fields[0].GetUInt32();
109  uint32 entry = fields[1].GetUInt32();
110  std::string name = fields[2].GetCppString();
111  sLog.outErrorDb("Transport %u '%s' have record (GUID: %u) in `gameobject`. Transports MUST NOT have any records in `gameobject` or its behavior will be unpredictable/bugged.", entry, name.c_str(), guid);
112  }
113  while (result->NextRow());
114 
115  delete result;
116  }
117 
118  sLog.outString(">> Loaded %u transports", count);

212 void Transport::TeleportTransport(uint32 newMapid, float x, float y, float z, float /*o*/)
213 {
214  Map* oldMap = GetMap();
215  Relocate(x, y, z);
216 
217  bool mapChange = GetMapId() != newMapid;
218 
219  auto& passengers = GetPassengers();
220  for (m_passengerTeleportIterator = passengers.begin(); m_passengerTeleportIterator != passengers.end();)
221  {
222  WorldObject* obj = (*m_passengerTeleportIterator++);
223 
224  if (!obj->IsUnit())
225  return; // should never happen on tbc
226 
227  Unit* passengerUnit = static_cast<Unit*>(obj);
228 
229  Position pos = passengerUnit->m_movementInfo.GetTransportPos();
230 
231  switch (obj->GetTypeId())
232  {
233  case TYPEID_UNIT:
234  {
235  if (mapChange)
236  {
237  RemovePassenger(passengerUnit);
238  if (obj->IsCreature() && !static_cast<Creature*>(obj)->IsPet())
239  passengerUnit->AddObjectToRemoveList();
240  }
241  break;
242  }
243  case TYPEID_PLAYER:
244  {
245  Player* player = static_cast<Player*>(obj);
246  if (player->IsDead() && !player->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_GHOST))
247  player->ResurrectPlayer(1.0);
248  player->TeleportTo(newMapid, pos.x, pos.y, pos.z, pos.o, TELE_TO_NOT_LEAVE_TRANSPORT, nullptr, this);
249  break;
250  }
251  }
252  }
253 
254  // we need to create and save new Map object with 'newMapid' because if not done -> lead to invalid Map object reference...
255  // player far teleport would try to create same instance, but we need it NOW for transport...
256  // correct me if I'm wrong O.o
257  if (mapChange)
258  {
259  RemoveModelFromMap();
260  oldMap->RemoveTransport(this);
261  UpdateForMap(oldMap, false);
262  ResetMap();
263 
264  Map* newMap = sMapMgr.CreateMap(newMapid, this);
265  newMap->GetMessager().AddMessage([transport = this](Map* map)
266  {
267  transport->SetMap(map);
268  map->AddTransport(transport);
269  transport->AddModelToMap();
270  transport->UpdateForMap(map, true);
271  });
272  }
273  else
274  UpdateModelPosition();
275 }

355 void Transport::Update(const uint32 /*diff*/)
356 {
357  uint32 const positionUpdateDelay = 50;
358 
359  if (GetKeyFrames().size() <= 1)
360  return;
361 
362  uint32 currentMsTime = GetMap()->GetCurrentMSTime() - m_movementStarted;
363  if (m_pathProgress >= currentMsTime) // map transition and update happened in same tick due to MT
364  return;
365 
366  const uint32 diff = currentMsTime - m_pathProgress;
367 
368  if (IsMoving() || !m_pendingStop)
369  m_pathProgress = currentMsTime;
370 
371  uint32 pathProgress = m_pathProgress % GetPeriod();
372  while (true)
373  {
374  if (pathProgress >= m_currentFrame->ArriveTime && pathProgress < m_currentFrame->DepartureTime)
375  {
376  SetMoving(false);
377  break; // its a stop frame and we are waiting
378  }
379 
380  // not waiting anymore
381  SetMoving(true);
382 
383  if (pathProgress >= m_currentFrame->DepartureTime && pathProgress < m_currentFrame->NextArriveTime)
384  break; // found current waypoint
385 
386  DoEventIfAny(*m_currentFrame->Node, true);
387 
388  MoveToNextWayPoint();
389 
390  DoEventIfAny(*m_currentFrame->Node, false);
391 
392  // first check help in case client-server transport coordinates de-synchronization
393  if (m_currentFrame->Node->mapid != GetMapId() || m_currentFrame->IsTeleportFrame())
394  {
395  TeleportTransport(m_nextFrame->Node->mapid, m_nextFrame->Node->x, m_nextFrame->Node->y, m_nextFrame->Node->z, m_nextFrame->InitialOrientation);
396  return;
397  }
398 
399  if (m_currentFrame == GetKeyFrames().begin())
400  DETAIL_FILTER_LOG(LOG_FILTER_TRANSPORT_MOVES, " ************ BEGIN ************** %s", GetName());
401 
402  DETAIL_FILTER_LOG(LOG_FILTER_TRANSPORT_MOVES, "%s moved to %f %f %f %d", GetName(), m_currentFrame->Node->x, m_currentFrame->Node->y, m_currentFrame->Node->z, m_currentFrame->Node->mapid);
403  }
404 
405  // Set position
406  m_positionChangeTimer.Update(diff);
407  if (m_positionChangeTimer.Passed())
408  {
409  m_positionChangeTimer.Reset(positionUpdateDelay);
410  if (IsMoving() && pathProgress)
411  {
412  float t = CalculateSegmentPos(float(pathProgress) * 0.001f);
413  G3D::Vector3 pos, dir;
414  m_currentFrame->Spline->evaluate_percent(m_currentFrame->Index, t, pos);
415  m_currentFrame->Spline->evaluate_derivative(m_currentFrame->Index, t, dir);
416  UpdatePosition(pos.x, pos.y, pos.z, atan2(dir.y, dir.x) + M_PI);
417  }
418  }
419 }

421 float Transport::CalculateSegmentPos(float now)
422 {
423  KeyFrame const& frame = *m_currentFrame;
424  float const speed = float(m_goInfo->moTransport.moveSpeed);
425  float const accel = float(m_goInfo->moTransport.accelRate);
426  float timeSinceStop = frame.TimeFrom + (now - (1.0f / IN_MILLISECONDS) * frame.DepartureTime);
427  float timeUntilStop = frame.TimeTo - (now - (1.0f / IN_MILLISECONDS) * frame.DepartureTime);
428  float segmentPos, dist;
429  float accelTime = m_transportTemplate.accelTime;
430  float accelDist = m_transportTemplate.accelDist;
431  // calculate from nearest stop, less confusing calculation...
432  if (timeSinceStop < timeUntilStop)
433  {
434  if (timeSinceStop < accelTime)
435  dist = 0.5f * accel * timeSinceStop * timeSinceStop;
436  else
437  dist = accelDist + (timeSinceStop - accelTime) * speed;
438  segmentPos = dist - frame.DistSinceStop;
439  }
440  else
441  {
442  if (timeUntilStop < m_transportTemplate.accelTime)
443  dist = 0.5f * accel * timeUntilStop * timeUntilStop;
444  else
445  dist = accelDist + (timeUntilStop - accelTime) * speed;
446  segmentPos = frame.DistUntilStop - dist;
447  }
448 
449  return segmentPos / frame.NextDistFromPrev;
450 }

577 void GenericTransport::CalculatePassengerPosition(float& x, float& y, float& z, float* o, float transX, float transY, float transZ, float transO)
578 {
579  float inx = x, iny = y, inz = z;
580  if (o)
581  *o = MapManager::NormalizeOrientation(transO + *o);
582 
583  x = transX + inx * std::cos(transO) - iny * std::sin(transO);
584  y = transY + iny * std::cos(transO) + inx * std::sin(transO);
585  z = transZ + inz;
586 }

588 void GenericTransport::CalculatePassengerOffset(float& x, float& y, float& z, float* o, float transX, float transY, float transZ, float transO)
589 {
590  if (o)
591  *o = MapManager::NormalizeOrientation(*o - transO);
592 
593  z -= transZ;
594  y -= transY; // y = searchedY * std::cos(o) + searchedX * std::sin(o)
595  x -= transX; // x = searchedX * std::cos(o) + searchedY * std::sin(o + pi)
596  float inx = x, iny = y;
597  y = (iny - inx * std::tan(transO)) / (std::cos(transO) + std::sin(transO) * std::tan(transO));
598  x = (inx + iny * std::tan(transO)) / (std::cos(transO) + std::sin(transO) * std::tan(transO));
599 }
