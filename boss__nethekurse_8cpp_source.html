123  switch (urand(0, 2))

125  case 0: DoBroadcastText(SAY_PEON_DIE_1, m_creature); break;

142  void DoTauntPeons()
143  {
144  if (m_peonKilledCount >= 4)
145  return;
146 
147  std::list<Creature*> felConverts;
148  GuidVector m_felConverts;
149 
150  GetCreatureListWithEntryInGrid(felConverts, m_creature, NPC_FEL_ORC_CONVERT, 40.0f);
151 
152  for (Creature* convert : felConverts)
153  m_felConverts.push_back(convert->GetObjectGuid());
154 
155  if (m_felConverts.size() == 0)
156  return;
157 
158  switch (urand(0, 2))
159  {
160  case 0:
161  DoCastSpellIfCan(nullptr, SPELL_SHADOW_SEAR);
162  DoBroadcastText(SAY_TAUNT_1, m_creature);
163  break;
164  case 1:
165  DoCastSpellIfCan(nullptr, SPELL_DEATH_COIL_2);
166  DoBroadcastText(SAY_TAUNT_2, m_creature);
167  break;
168  case 2:
169  DoCastSpellIfCan(nullptr, SPELL_TARGET_FISSURES);
170  DoBroadcastText(SAY_TAUNT_3, m_creature);
171  break;
172  }
173 
174  ResetTimer(NETHEKURSE_TAUNT_PEONS, urand(30000, 35000));
175  }

258

261  Creature* nethekurse = m_instance->GetSingleCreatureFromStorage(NPC_NETHEKURSE);
262  if (nethekurse && m_creature->IsWithinDist(nethekurse, 45.0f))
263  {

265  pKurseAI->DoYellForPeonAggro();
266 
267  if (m_instance->GetData(TYPE_NETHEKURSE) == IN_PROGRESS)
268  return;
269  m_instance->SetData(TYPE_NETHEKURSE, IN_PROGRESS);
