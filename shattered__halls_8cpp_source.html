133 void instance_shattered_halls::SetData(uint32 type, uint32 data)
134 {
135  switch (type)
136  {
137  case TYPE_NETHEKURSE:
138  m_auiEncounter[type] = data;
139  if (data == DONE)
140  {
141  DoUseDoorOrButton(GO_NETHEKURSE_DOOR);
142  DoUseOpenableObject(GO_NETHEKURSE_ENTER_DOOR, true);
143  }
144  break;
145  case TYPE_OMROGG:
146  m_auiEncounter[type] = data;
147  break;
148  case TYPE_BLADEFIST:
149  m_auiEncounter[type] = data;
150  if (data == DONE)
151  {
152  // Make executioner attackable only after the final boss is dead
153  if (Creature* pExecutioner = GetSingleCreatureFromStorage(NPC_EXECUTIONER, true))
154  pExecutioner->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE | UNIT_FLAG_SPAWNING);
155  }
156  break;
157  case TYPE_EXECUTION:
158  m_auiEncounter[type] = data;
159  if (data == IN_PROGRESS && !GetSingleCreatureFromStorage(NPC_EXECUTIONER, true))
160  {
161  if (Player* player = GetPlayerInMap())
162  {
163  // summon the 3 npcs for execution
164  for (uint8 i = 2; i < 5; ++i)
165  player->SummonCreature(m_team == ALLIANCE ? aSoldiersLocs[i].m_uiAllianceEntry : aSoldiersLocs[i].m_uiHordeEntry, aSoldiersLocs[i].m_fX, aSoldiersLocs[i].m_fY, aSoldiersLocs[i].m_fZ, aSoldiersLocs[i].m_fO, TEMPSPAWN_DEAD_DESPAWN, 0);
166 
167  // Summon the executioner; Note: according to wowhead he shouldn't be targetable until Kargath encounter is finished
168  if (Creature* pExecutioner = player->SummonCreature(NPC_EXECUTIONER, afExecutionerLoc[0], afExecutionerLoc[1], afExecutionerLoc[2], afExecutionerLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0, true))
169  pExecutioner->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE | UNIT_FLAG_SPAWNING);
170 
171  // cast the execution spell
172  DoCastGroupDebuff(SPELL_KARGATH_EXECUTIONER_1);
173  }
174  }
175  if (data == DONE)
176  {
177  // If the officer is already killed, then skip the quest completion
178  if (m_executionStage)
179  break;
180 
181  // Complete quest 9524 or 9525
182  if (Creature* officer = GetSingleCreatureFromStorage(m_team == ALLIANCE ? NPC_OFFICER_ALLIANCE : NPC_OFFICER_HORDE))
183  {
184  Map::PlayerList const& lPlayers = instance->GetPlayers();
185  for (const auto& lPlayer : lPlayers)
186  {
187  if (Player* player = lPlayer.getSource())
188  player->KilledMonsterCredit(officer->GetEntry(), officer->GetObjectGuid());
189  }
190  }
191  }
192  break;
193  case TYPE_GAUNTLET:
194  if (m_auiEncounter[type] == DONE) // do not allow any exploits
195  return;
196 
197  switch (data)
198  {
199  case FAIL: // Called on wipe/players left/Boss Evade
200  FailGauntlet();
201  break;
202  case DONE: // Called on boss kill
203  EndGauntlet();
204  break;
205  case SPECIAL: // Called on Boss Aggro
206  StopGauntlet();
207  break;
208  case IN_PROGRESS:
209  DoInitialGets();
210  break;
211  default:
212  break;
213  }
214 
215  m_auiEncounter[type] = data;
216  break;
217  }
218 
219  if (data == DONE)
220  {
221  OUT_SAVE_INST_DATA;
222 
223  std::ostringstream saveStream;
224 
225  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
226  m_strInstData = saveStream.str();
227 
228  SaveToDB();
229  OUT_SAVE_INST_DATA_COMPLETE;
230  }
231 }

424 void instance_shattered_halls::Update(uint32 diff)
425 {
426  if (m_auiEncounter[TYPE_GAUNTLET] == IN_PROGRESS)
427  {
428  if (!GetPlayerInMap(true))
429  {
430  SetData(TYPE_GAUNTLET, FAIL);
431  }
432  else
433  {
434  for (auto& blaze : m_blazeTimers)
435  {
436  if (blaze.second <= diff)
437  {
438  blaze.second = 2000;
439  if (GameObject* blazeGo = instance->GetGameObject(blaze.first))
440  blazeGo->CastSpell(nullptr, nullptr, SPELL_FLAMES, TRIGGERED_NONE, nullptr, nullptr, blazeGo->GetObjectGuid());
441  }
442  else
443  blaze.second -= diff;
444  }
445  }
446 
447  if (m_gauntletStopped)
448  return;
449 
450  if (m_initialWavesSpawned)
451  {
452  if (m_porungDoneYelling)
453  {
454  if (m_waveTimer < diff) // Periodic waves
455  {
456  if (m_zealotOneOrTwo) // second zealot, long delay
457  {
458  DoSummonSHZealot();
459  m_waveTimer = WAVE_TIMER;
460  m_zealotOneOrTwo = false;
461  }
462  else // first zealot, short delay
463  {
464  DoSummonSHZealot();
465  m_waveTimer = DELAY_350_MILLI;
466  m_zealotOneOrTwo = true;
467  }
468  }
469  else
470  m_waveTimer -= diff;
471 
472  if (m_shootFlamingArrowTimer_1 < diff) // Left Archer
473  {
474  DoBeginArcherAttack(true);
475  m_shootFlamingArrowTimer_1 = ARCHER_SHOOT_DELAY + urand(0, 2000);
476  }
477  else
478  m_shootFlamingArrowTimer_1 -= diff;
479

481  {
482  DoBeginArcherAttack(false);
483  m_shootFlamingArrowTimer_2 = ARCHER_SHOOT_DELAY + urand(0, 2000);
484  }
485  else
486  m_shootFlamingArrowTimer_2 -= diff;
487  }
488  else // Not done yelling
489  {
490  if (m_porungYellDelay < diff)
491  {
492  switch (m_porungYellNumber)
493  {
494  case 0:
495  DoScriptText(PORUNG_FORM_RANKS_YELL, m_porung);

502  DoBeginArcherAttack(false);
503  m_porungYellDelay = PORUNG_YELL_DELAY_2;
504  m_porungYellNumber += 1;
505  break;
506  case 2:
507  DoScriptText(PORUNG_AIM_YELL, m_porung);
508  m_porungYellDelay = PORUNG_YELL_DELAY_2;
509  m_porungYellNumber += 1;
510  break;
511  case 3:
512  DoScriptText(PORUNG_FIRE_YELL, m_porung);
513  m_porungDoneYelling = true;
514  break;
515  }
516  }
517  else
518  m_porungYellDelay -= diff;
519  }
520  }
521  else // not done spawning first waves
522  {
523  if (m_initialWaves_Delay < diff)
524  {
525  switch (m_numInitialWaves)
526  {
527  case 0:
528  case 1:
529  case 2:
530  DoSummonInitialWave();
531  m_numInitialWaves++;
532  m_initialWaves_Delay = DELAY_350_MILLI;
533  break;
534  case 3:
535  DoSummonInitialWave();
536  m_initialWavesSpawned = true;
537  break;
538  }
539  }
540  else
541  m_initialWaves_Delay -= diff;
542  }
543  }
544 
545  if (m_auiEncounter[TYPE_EXECUTION] != IN_PROGRESS)
546  return;
547 
548  if (m_executionTimer < diff)
549  {
550  switch (m_executionStage)
551  {
552  case 0:
553  // Kill the officer
554  if (Creature* pSoldier = GetSingleCreatureFromStorage(m_team == ALLIANCE ? NPC_OFFICER_ALLIANCE : NPC_OFFICER_HORDE))
555  pSoldier->Suicide();
556 
557  // Make Kargath yell

559 
560  // Set timer for the next execution
561  DoCastGroupDebuff(SPELL_KARGATH_EXECUTIONER_2);
562  m_executionTimer = 10 * MINUTE * IN_MILLISECONDS;
563  break;
564  case 1:
565  if (Creature* pSoldier = GetSingleCreatureFromStorage(m_team == ALLIANCE ? NPC_SOLDIER_ALLIANCE_2 : NPC_SOLDIER_HORDE_2))
566  pSoldier->Suicide();
567 
568  DoCastGroupDebuff(SPELL_KARGATH_EXECUTIONER_3);
569  m_executionTimer = 15 * MINUTE * IN_MILLISECONDS;
570  break;
571  case 2:

573  pSoldier->Suicide();
574

576  m_executionTimer = 0;
577  break;
