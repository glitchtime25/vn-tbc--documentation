225  void HandleBladeDance()
226  {
227  if (m_bladeDanceTargetCount == 0)
228  {
229  // stop bladedance
230  SetCombatScriptStatus(false);
231  SetCombatMovement(true, true);
232  SetMeleeEnabled(true);
233  if (!m_isRegularMode)
234  ResetCombatAction(WARCHIEF_KARGATH_CHARGE, 500);
235  }
236  else
237  {
238  // move in bladedance
239  SpellCastResult result = SPELL_NOT_FOUND;
240  do
241  {
242  uint32 index = urand(0, m_bladeDanceTargetGuids.size() - 1);
243  if (Unit* target = m_creature->GetMap()->GetCreature(m_bladeDanceTargetGuids[index]))
244  // Until neutral target type TARGET_UNIT stops rolling for hit, need to force it
245  // should not send SMSG_SPELL_START but triggered spell cast bypasses checks which are necessary
246  result = m_creature->CastSpell(target, SPELL_BLADE_DANCE_CHARGE, TRIGGERED_IGNORE_HIT_CALCULATION);
247  m_bladeDanceTargetGuids.erase(m_bladeDanceTargetGuids.begin() + index);
248  --m_bladeDanceTargetCount;
249  } while (result != SPELL_CAST_OK && m_bladeDanceTargetCount > 0);
250  if (result != SPELL_CAST_OK)
251  {
252  ResetTimer(WARCHIEF_KARGATH_COMBAT_BLADE_DANCE, 1);
253  m_bladeDanceTargetCount = 0;
254  }
255  }
256  }
257 
258  void ExecuteAction(uint32 action) override
259  {
260  switch (action)
261  {
262  case WARCHIEF_KARGATH_BLADE_DANCE:
263  {
264  ResetTimer(WARCHIEF_KARGATH_COMBAT_BLADE_DANCE, 1);
265  SetCombatScriptStatus(true);
266  SetCombatMovement(false);
267  SetMeleeEnabled(false);
268  m_bladeDanceTargetGuids.clear();
269  m_creature->CastSpell(nullptr, SPELL_BLADE_DANCE_TARGETING, TRIGGERED_NONE);
270  m_bladeDanceTargetCount = std::min(m_bladeDanceTargetGuids.size(), size_t(TARGET_NUM));
271  ResetCombatAction(action, 30000);
272  break;
273  }
274  case WARCHIEF_KARGATH_CHARGE:
275  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
276  DoCastSpellIfCan(target, SPELL_CHARGE_H);
277  break;
278  }
279  }
