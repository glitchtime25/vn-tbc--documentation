378  void UpdateAI(const uint32 diff) override
379  {
380  if (m_areBothDead) // This is the case while fighting Thaddius
381  {
382  if (m_shockOverloadTimer) // Timer before triggering phase 2 with spell Shock Overload
383  {
384  if (m_shockOverloadTimer <= diff)
385  {
386  if (DoCastSpellIfCan(m_creature, SPELL_TRIGGER_TESLAS) == CAST_OK)
387  m_shockOverloadTimer = 0;
388  }
389  else
390  m_shockOverloadTimer -= diff;
391  }
392  return;
393  }
394 
395  if (m_isFakingDeath)
396  {
397  if (m_reviveTimer < diff)
398  {
399  if (Creature* pOther = GetOtherAdd())
400  {
401  if (boss_thaddiusAddsAI* otherAI = dynamic_cast<boss_thaddiusAddsAI*>(pOther->AI()))
402  {
403  if (!otherAI->IsCountingDead()) // Raid was to slow to kill the second add
404  {
405  DoResetThreat();
406  SetCombatMovement(false);
407  m_holdTimer = 1500;
408  Reset();
409  }
410  else
411  {
412  m_areBothDead = true; // Now both adds are counting dead
413  otherAI->m_areBothDead = true;
414  // Set both Teslas to overload
415  m_shockOverloadTimer = 14 * IN_MILLISECONDS;
416  }
417  }
418  }
419  }
420  else
421  m_reviveTimer -= diff;
422  return;
423  }
424 
425  // If no target do nothing
426  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
427  return;
428 
429  if (m_holdTimer) // A short timer preventing combat movement after revive
430  {
431  if (m_holdTimer <= diff)
432  {
433  SetCombatMovement(true);
434  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
435  m_holdTimer = 0;
436  }
437  else
438  m_holdTimer -= diff;
439  }
440 
441  if (m_warStompTimer < diff)
442  {
443  if (DoCastSpellIfCan(m_creature, SPELL_WARSTOMP) == CAST_OK)
444  m_warStompTimer = urand(8 * IN_MILLISECONDS, 10 * IN_MILLISECONDS);
445  }
446  else
447  m_warStompTimer -= diff;
448 
449  UpdateAddAI(diff); // For Add Specific Abilities
450 
451  DoMeleeAttackIfReady();
452  }

454  void DamageTaken(Unit* dealer, uint32& damage, DamageEffectType /*damagetype*/, SpellEntry const* /*spellInfo*/) override
455  {

457  return;

613 struct PolarityShift : public SpellScript
614 {
615  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx ) const override
616  {
617  if (effIdx == EFFECT_INDEX_0)
618  {
619  if (Unit* unitTarget = spell->GetUnitTarget())
620  {
621  unitTarget->RemoveAurasDueToSpell(SPELL_POSITIVE_CHARGE);
622  unitTarget->RemoveAurasDueToSpell(SPELL_NEGATIVE_CHARGE);
623 
624  uint64 scriptValue = spell->GetScriptValue();
625 
626  // 28059 : Positive Charge, 28084 : Negative Charge
627  switch (scriptValue)
628  {
629  case 0: // first target random
630  scriptValue = urand(0, 1) ? SPELL_POSITIVE_CHARGE : SPELL_NEGATIVE_CHARGE;
631  spell->SetScriptValue(scriptValue);
632  unitTarget->CastSpell(unitTarget, scriptValue, TRIGGERED_INSTANT_CAST);
633  break;
634  case SPELL_POSITIVE_CHARGE: // second target the other
635  spell->SetScriptValue(1);
636  unitTarget->CastSpell(unitTarget, SPELL_NEGATIVE_CHARGE, TRIGGERED_INSTANT_CAST);
637  break;
638  case SPELL_NEGATIVE_CHARGE:
639  spell->SetScriptValue(1);
640  unitTarget->CastSpell(unitTarget, SPELL_POSITIVE_CHARGE, TRIGGERED_INSTANT_CAST);
641  break;
642  default: // third and later random
643  unitTarget->CastSpell(unitTarget, urand(0, 1) ? SPELL_POSITIVE_CHARGE : SPELL_NEGATIVE_CHARGE, TRIGGERED_INSTANT_CAST);
644  break;
645  }
646  }
647  }
648  }

675 struct ThaddiusCharge : public AuraScript

677  void OnApply(Aura* aura, bool apply) const override
678  {
679  if (apply)
680  {
681  Unit* target = aura->GetTarget();
682  if (!target)
683  return;
684 
685  // On Polarity Shift, remove the previous damage buffs
686  uint32 buffAuraIds[2] = { SPELL_POSITIVE_CHARGE_BUFF , SPELL_NEGATIVE_CHARGE_BUFF };
687  for (auto buffAura: buffAuraIds)
688  target->RemoveAurasDueToSpell(buffAura);
689  }
690  }
691 
692  void OnPeriodicTrigger(Aura* aura, PeriodicTriggerData& /*data*/) const override
693  {
694  if (Unit* target = aura->GetTarget())
695  {
696 
697  uint32 buffAuraId = aura->GetId() == SPELL_POSITIVE_CHARGE ? SPELL_POSITIVE_CHARGE_BUFF : SPELL_NEGATIVE_CHARGE_BUFF;
698  float range = 13.f; // Static value from DBC files. As the value is the same for both spells we can hardcode it instead of accessing is through sSpellRadiusStore
699 
700  uint32 curCount = 0;
701  PlayerList playerList;
702  GetPlayerListWithEntryInWorld(playerList, target, range);
703  for (Player* player : playerList)
704  if (target != player && player->HasAura(aura->GetId()))
705  ++curCount;
706 
707  // Remove previous buffs in case we have less targets of the same charge near use than in previous tick
708  target->RemoveAurasDueToSpell(buffAuraId);
709  for (uint32 i = 0; i < curCount; i++)
710  target->CastSpell(target, buffAuraId, TRIGGERED_OLD_TRIGGERED);
711  }
712  }
