126  void OnSpellCast(SpellEntry const* spellInfo, Unit* target) override
127  {
128  switch (spellInfo->Id)
129  {
130  case SPELL_PLANT_WHITE:
131  case SPELL_PLANT_GREEN:
132  case SPELL_PLANT_BLUE:
133  case SPELL_PLANT_RED: AddSeedlingCooldowns(spellInfo->Id); break;
134  case SPELL_TREE_FORM:
135  DoBroadcastText(urand(0, 1) ? SAY_TREE_1 : SAY_TREE_2, m_creature);
136  m_creature->CastSpell(nullptr, SPELL_SUMMON_FRAYER, TRIGGERED_IGNORE_CURRENT_CASTED_SPELL | TRIGGERED_IGNORE_GCD);
137  m_creature->SetSpellList(m_isRegularMode ? SPELL_LIST_TREE_FORM : SPELL_LIST_TREE_FORM_H);
138  break;
139  }
140  }
