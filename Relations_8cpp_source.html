107 static inline ReputationRank GetFactionReaction(FactionTemplateEntry const* thisTemplate, FactionTemplateEntry const* otherTemplate)
108 {
109  MANGOS_ASSERT(thisTemplate)
110  MANGOS_ASSERT(otherTemplate)
111 
112  // Original logic begins
113 
114  if (otherTemplate->factionGroupMask & thisTemplate->enemyGroupMask)
115  return REP_HOSTILE;
116 
117  if (thisTemplate->enemyFaction[0] && otherTemplate->faction)
118  {
119  for (unsigned int i : thisTemplate->enemyFaction)
120  {
121  if (i == otherTemplate->faction)
122  return REP_HOSTILE;
123  }
124  }
125 
126  if (otherTemplate->factionGroupMask & thisTemplate->friendGroupMask)
127  return REP_FRIENDLY;
128 
129  if (thisTemplate->friendFaction[0] && otherTemplate->faction)
130  {
131  for (unsigned int i : thisTemplate->friendFaction)
132  {
133  if (i == otherTemplate->faction)
134  return REP_FRIENDLY;
135  }
136  }
137 
138  if (thisTemplate->factionGroupMask & otherTemplate->friendGroupMask)
139  return REP_FRIENDLY;
140 
141  if (otherTemplate->friendFaction[0] && thisTemplate->faction)
142  {
143  for (unsigned int i : otherTemplate->friendFaction)
144  {
145  if (i == thisTemplate->faction)
146  return REP_FRIENDLY;
147  }
148  }
149  return REP_NEUTRAL;
150 }

161 static ReputationRank GetFactionReaction(FactionTemplateEntry const* thisTemplate, Unit const* unit)
162 {
163  MANGOS_ASSERT(unit)
164 
165  // Original logic begins
166 
167  if (thisTemplate)
168  {
169  if (const FactionTemplateEntry* unitFactionTemplate = unit->GetFactionTemplateEntry())
170  {
171  if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
172  {
173  if (const Player* unitPlayer = unit->GetControllingPlayer())
174  {
175  if (unitPlayer->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_CONTESTED_PVP) && thisTemplate->IsContestedGuardFaction())
176  return REP_HOSTILE;
177 
178  if (const ReputationRank* rank = unitPlayer->GetReputationMgr().GetForcedRankIfAny(thisTemplate))
179  return (*rank);
180 
181  if (!unit->HasFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_IGNORE_REPUTATION))
182  {
183  const FactionEntry* thisFactionEntry = sFactionStore.LookupEntry<FactionEntry>(thisTemplate->faction);
184  if (thisFactionEntry && thisFactionEntry->HasReputation())
185  {
186  const ReputationMgr& reputationMgr = unitPlayer->GetReputationMgr();
187  return reputationMgr.GetRank(thisFactionEntry);
188  }
189  }
190  }
191 
192  }
193  // Default fallback if player-specific checks didn't catch anything: facton to faction
194  return GetFactionReaction(thisTemplate, unitFactionTemplate);
195  }
196  }
197  return REP_NEUTRAL;

207 ReputationRank Unit::GetReactionTo(Unit const* unit) const
208 {
209  MANGOS_ASSERT(unit)
210 
211  // Original logic begins
212 
213  if (this == unit)
214  return REP_FRIENDLY;
215 
216  if (HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
217  {
218  const Player* thisPlayer = GetControllingPlayer();
219 
220  if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
221  {
222  const Player* unitPlayer = unit->GetControllingPlayer();
223 
224  if (!thisPlayer || !unitPlayer)
225  return REP_NEUTRAL;
226 
227  if (thisPlayer == unitPlayer)
228  return REP_FRIENDLY;
229 
230  if (thisPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) && unitPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) && thisPlayer->GetGuidValue(PLAYER_DUEL_ARBITER) == unitPlayer->GetGuidValue(PLAYER_DUEL_ARBITER))
231  return thisPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) != unitPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) ? REP_HOSTILE : REP_FRIENDLY;
232 
233  // Pre-WotLK group check: always, replaced with faction template check in WotLK
234  if (thisPlayer->IsInGroup(unitPlayer))
235  return REP_FRIENDLY;
236 
237  // Pre-WotLK FFA check, known limitation: FFA doesn't work with totem elementals both client-side and server-side
238  if (thisPlayer->IsPvPFreeForAll() && unitPlayer->IsPvPFreeForAll())
239  return REP_HOSTILE;
240  }
241 
242  if (thisPlayer)
243  {
244  if (const FactionTemplateEntry* unitFactionTemplate = unit->GetFactionTemplateEntry())
245  {
246  if (const ReputationRank* rank = thisPlayer->GetReputationMgr().GetForcedRankIfAny(unitFactionTemplate))
247  return (*rank);
248 
249  const FactionEntry* unitFactionEntry = sFactionStore.LookupEntry<FactionEntry>(unitFactionTemplate->faction);
250 
251  // If the faction has reputation ranks available, "at war" and contested PVP flags decide outcome
252  if (!this->HasFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_IGNORE_REPUTATION) && unitFactionEntry && unitFactionEntry->HasReputation())
253  {
254  if (thisPlayer->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_CONTESTED_PVP) && unitFactionTemplate->IsContestedGuardFaction())
255  return REP_HOSTILE;
256 
257  return thisPlayer->GetReputationMgr().IsAtWar(unitFactionEntry) ? REP_HOSTILE : REP_FRIENDLY;
258  }
259  }
260  }
261  }
262  // Default fallback if player-specific checks didn't catch anything: facton to unit
263  ReputationRank reaction = GetFactionReaction(GetFactionTemplateEntry(), unit);
264 
265  // Persuation support
266  if (reaction > REP_HOSTILE && reaction < REP_HONORED && (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PERSUADED) || GetPersuadedGuid() == unit->GetObjectGuid()))
267  {
268  if (const FactionTemplateEntry* unitFactionTemplate = unit->GetFactionTemplateEntry())
269  {
270  const FactionEntry* unitFactionEntry = sFactionStore.LookupEntry<FactionEntry>(unitFactionTemplate->faction);
271  if (unitFactionEntry && unitFactionEntry->HasReputation())
272  reaction = ReputationRank(int32(reaction) + 1);
273  }
274  }
275  return reaction;
276 }

360 bool Unit::CanAttack(const Unit* unit) const
361 {
362  MANGOS_ASSERT(unit)
363 
364  // Original logic
365 
366  // TBC+: Arena Tournament commenatator
367  if (GetTypeId() == TYPEID_PLAYER && static_cast<const Player*>(this)->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_COMMENTATOR))
368  return false;
369 
370  // Creatures cannot attack player ghosts, unless it is a specially flagged ghost creature
371  if (GetTypeId() == TYPEID_UNIT && unit->GetTypeId() == TYPEID_PLAYER && static_cast<const Player*>(unit)->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_GHOST))
372  {
373  if (!(static_cast<const Creature*>(this)->GetCreatureInfo()->CreatureTypeFlags & CREATURE_TYPEFLAGS_GHOST_VISIBLE))
374  return false;
375  }
376 
377  // We can't attack unit when at least one of these flags is present on it:
378  const uint32 mask = (UNIT_FLAG_SPAWNING | UNIT_FLAG_NOT_ATTACKABLE_1 | UNIT_FLAG_UNTARGETABLE | UNIT_FLAG_TAXI_FLIGHT | UNIT_FLAG_UNINTERACTIBLE);
379  if (unit->HasFlag(UNIT_FIELD_FLAGS, mask))
380  return false;
381 
382  // Cross-check immunity and sanctuary flags: this <-> unit
383  const bool thisPlayerControlled = HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED);
384  if (thisPlayerControlled)
385  {
386  if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER))
387  return false;
388 
389  // TBC: Sanctuary check
390  if (unit->IsPvPSanctuary())
391  return false;
392  }
393  else if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC))
394  return false;
395 
396  const bool unitPlayerControlled = unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED);
397  if (unitPlayerControlled)
398  {
399  if (HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER))
400  return false;
401 
402  // TBC: Sanctuary check
403  if (IsPvPSanctuary())
404  return false;
405  }
406  else if (HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC))
407  return false;
408 
409  if (thisPlayerControlled || unitPlayerControlled)
410  {
411  if (thisPlayerControlled && unitPlayerControlled)
412  {
413  if (IsFriend(unit))
414  return false;
415 
416  const Player* thisPlayer = GetControllingPlayer();
417  if (!thisPlayer)
418  return true;
419 
420  const Player* unitPlayer = unit->GetControllingPlayer();
421  if (!unitPlayer)
422  return true;
423 
424  if (thisPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) && unitPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) && thisPlayer->GetGuidValue(PLAYER_DUEL_ARBITER) == unitPlayer->GetGuidValue(PLAYER_DUEL_ARBITER))
425  return true;
426 
427  if (unitPlayer->IsPvP())
428  return true;
429 
430  if (thisPlayer->IsPvPFreeForAll() && unitPlayer->IsPvPFreeForAll())
431  return true;
432 
433  return false;
434  }
435  return (!IsFriend(unit));
436  }
437  return (IsEnemy(unit) || unit->IsEnemy(this));
438 }

482 bool Unit::CanAssist(const Unit* unit, bool ignoreFlags) const
483 {
484  MANGOS_ASSERT(unit)
485 
486  // Original logic
487 
488  // We can't assist unselectable unit
489  if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE))
490  return false;
491 
492  // Check immunity flags
493  if (!ignoreFlags)
494  {
495  if (HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
496  {
497  if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER))
498  return false;
499  }
500  else if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC))
501  return false;
502  }
503 
504  // Exclude non-friendlies at this point
505  if (GetReactionTo(unit) < REP_FRIENDLY)
506  return false;
507 
508  // Pre-WotLK: backbone of lua UnitIsPVP(), a member of unit class client-side
509  auto isPvPUI = [](Unit const * self)
510  {
511  if (Unit const* master = self->GetMaster())
512  {
513  if (self->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER))
514  return false;
515  return master->IsPvP();
516  }
517  return self->IsPvP();
518  };
519 
520  // Detect player controlled unit and exit early
521  if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
522  {
523  const Player* thisPlayer = GetControllingPlayer();
524  const Player* unitPlayer = unit->GetControllingPlayer();
525 
526  if (thisPlayer && unitPlayer)
527  {
528  if (thisPlayer->GetGuidValue(PLAYER_DUEL_ARBITER) != unitPlayer->GetGuidValue(PLAYER_DUEL_ARBITER)
529  || thisPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) != unitPlayer->GetUInt32Value(PLAYER_DUEL_TEAM))
530  return false;
531 
532  if (unitPlayer->IsPvPFreeForAll() && !thisPlayer->IsPvPFreeForAll())
533  return false;
534 
535  // TBC+: Cannot help a player outside of the sanctuary zone from within
536  if (thisPlayer->IsPvPSanctuary() && !unitPlayer->IsPvPSanctuary())
537  {
538  if (isPvPUI(unitPlayer))
539  return false;
540  }
541  }
542  return true;
543  }
544 
545  // If we continue here, unit is an npc. Detect if we are an npc too, so we can exit early
546  if (!HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
547  return true;
548 
549  // We are left with player assisting an npc case here: can assist if we ignore flags during the check, or friendly NPCs with PVP flag or special assist flag
550  if (ignoreFlags)
551  return true;
552 
553  if (isPvPUI(unit))
554  return true;
555 
556  if (unit->GetTypeId() == TYPEID_UNIT)
557  {
558  if (const uint32 flags = static_cast<const Creature*>(unit)->GetCreatureInfo()->CreatureTypeFlags)
559  {
560  if (flags & CREATURE_TYPEFLAGS_CAN_ASSIST)
561  return true;
562  }
563  }
564  return false;
565 }
