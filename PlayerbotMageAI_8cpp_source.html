93 CombatManeuverReturns PlayerbotMageAI::DoFirstCombatManeuver(Unit* pTarget)
94 {
95  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
96  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
97  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
98  {
99  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
100  {
101  return RETURN_NO_ACTION_OK; // wait it out
102  }
103  else
104  {
105  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
106  }
107  }
108 
109  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
110  {
111  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
112  return RETURN_NO_ACTION_OK; // wait it out
113  else
114  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
115  }
116 
117  switch (m_ai.GetScenarioType())
118  {
119  case PlayerbotAI::SCENARIO_PVP_DUEL:
120  case PlayerbotAI::SCENARIO_PVP_BG:
121  case PlayerbotAI::SCENARIO_PVP_ARENA:
122  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
123  return DoFirstCombatManeuverPVP(pTarget);
124  case PlayerbotAI::SCENARIO_PVE:
125  case PlayerbotAI::SCENARIO_PVE_ELITE:
126  case PlayerbotAI::SCENARIO_PVE_RAID:
127  default:
128  return DoFirstCombatManeuverPVE(pTarget);
129  }
130 }

162 CombatManeuverReturns PlayerbotMageAI::DoNextCombatManeuverPVE(Unit* pTarget)
163 {
164  Unit* pVictim = pTarget->GetVictim();
165  bool meleeReach = m_bot.CanReachWithMeleeAttack(pTarget);
166 
167  uint32 spec = m_bot.GetSpec();
168 
169  if (m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_RANGED && !meleeReach)
170  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_RANGED);
171  // switch to melee if in melee range AND can't shoot OR have no ranged (wand) equipped
172  else if (m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_MELEE
173  && meleeReach
174  && (SHOOT == 0 || !m_bot.GetWeaponForAttack(RANGED_ATTACK, true, true)))
175  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_MELEE);
176 
177  //Used to determine if this bot is highest on threat
178  Unit* newTarget = m_ai.FindAttacker((PlayerbotAI::ATTACKERINFOTYPE)(PlayerbotAI::AIT_VICTIMSELF | PlayerbotAI::AIT_HIGHESTTHREAT), &m_bot);
179 
180  // Remove curse on group members
181  if (m_ai.HasDispelOrder() && DispelPlayer(GetDispelTarget(DISPEL_CURSE)) & RETURN_CONTINUE)
182  return RETURN_CONTINUE;
183 
184  if (newTarget && !PlayerbotAI::IsNeutralized(newTarget)) // Bot has aggro and the mob is not already crowd controled
185  {
186  if (newTarget->GetHealthPercent() > 25)
187  {
188  // If elite
189  if (m_ai.IsElite(newTarget))
190  {
191  // If the attacker is a beast or humanoid, let's the bot give it a form more suited to the low intellect of something fool enough to attack a mage
192  // No need to check if target is a player as IsElite() returns false for players
193  Creature* creature = (Creature*) newTarget;
194  if (creature->GetCreatureInfo()->CreatureType == CREATURE_TYPE_HUMANOID || creature->GetCreatureInfo()->CreatureType == CREATURE_TYPE_BEAST)
195  {
196  if (POLYMORPH > 0 && !PlayerbotAI::IsImmuneToSchool(newTarget, SPELL_SCHOOL_MASK_ARCANE) && CastSpell(POLYMORPH, newTarget))
197  return RETURN_CONTINUE;
198  }
199 
200  // Things are getting dire: cast Ice block
201  if (ICE_BLOCK > 0 && m_bot.IsSpellReady(ICE_BLOCK) && m_ai.GetHealthPercent() < 30 && !m_bot.HasAura(ICE_BLOCK, EFFECT_INDEX_0) && m_ai.CastSpell(ICE_BLOCK) == SPELL_CAST_OK)
202  return RETURN_CONTINUE;
203 
204  // Cast Ice Barrier if health starts to goes low
205  if (ICE_BARRIER > 0 && m_bot.IsSpellReady(ICE_BARRIER) && m_ai.GetHealthPercent() < 50 && !m_bot.HasAura(ICE_BARRIER) && m_ai.SelfBuff(ICE_BARRIER) == SPELL_CAST_OK)
206  return RETURN_CONTINUE;
207 
208  // Have threat, can't quickly lower it. 3 options remain: Stop attacking, lowlevel damage (wand), keep on keeping on.
209  return CastSpell(SHOOT, pTarget);
210  }
211  else // not elite
212  {
213  // Cast mana shield if no shield is already up
214  if (MANA_SHIELD > 0 && m_ai.GetHealthPercent() < 70 && !m_bot.HasAura(MANA_SHIELD) && !m_bot.HasAura(ICE_BARRIER) && m_ai.SelfBuff(MANA_SHIELD) == SPELL_CAST_OK)
215  return RETURN_CONTINUE;
216  }
217  }
218  }
219 
220  // Mana check and replenishment
221  if (EVOCATION && m_ai.GetManaPercent() <= 10 && m_bot.IsSpellReady(EVOCATION) && !newTarget && m_ai.SelfBuff(EVOCATION) == SPELL_CAST_OK)
222  return RETURN_CONTINUE;
223  if (m_ai.GetManaPercent() <= 20)
224  {
225  Item* gem = FindManaGem();
226  if (gem)
227  m_ai.UseItem(gem);
228  }
229 
230  // If bot has frost/fire resist order use Frost/Fire Ward when available
231  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_RESIST_FROST && FROST_WARD && m_bot.IsSpellReady(FROST_WARD) && m_ai.SelfBuff(FROST_WARD) == SPELL_CAST_OK)
232  return RETURN_CONTINUE;
233  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_RESIST_FIRE && FIRE_WARD && m_bot.IsSpellReady(FIRE_WARD) && m_ai.SelfBuff(FIRE_WARD) == SPELL_CAST_OK)
234  return RETURN_CONTINUE;
235 
236  if (COUNTERSPELL > 0 && m_bot.IsSpellReady(COUNTERSPELL) && pTarget->IsNonMeleeSpellCasted(true) && CastSpell(COUNTERSPELL, pTarget))
237  return RETURN_CONTINUE;
238 
239  // If Clearcasting is active, cast arcane missiles
240  // Bot could also cast flamestrike or blizzard for free, but the AoE could break some crowd control
241  // or add threat on mobs ignoring the bot currently, so only focus on the bot's current target
242  if (m_bot.HasAura(CLEARCASTING_1) && ARCANE_MISSILES > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && CastSpell(ARCANE_MISSILES, pTarget))
243  {
244  m_ai.SetIgnoreUpdateTime(3);
245  return RETURN_CONTINUE;
246  }
247 
248  switch (spec)
249  {
250  case MAGE_SPEC_FROST:
251  if (ICY_VEINS > 0 && m_ai.In_Reach(&m_bot, ICY_VEINS) && !m_bot.HasAura(ICY_VEINS, EFFECT_INDEX_0) && CastSpell(ICY_VEINS, &m_bot))
252  return RETURN_CONTINUE;
253  if (COLD_SNAP && m_bot.IsSpellReady(COLD_SNAP) && CheckFrostCooldowns() > 2 && m_ai.SelfBuff(COLD_SNAP) == SPELL_CAST_OK) // Clear frost spell cooldowns if bot has more than 2 active
254  return RETURN_CONTINUE;
255  if (CONE_OF_COLD > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_bot.IsSpellReady(CONE_OF_COLD) && meleeReach)
256  {
257  // Cone of Cold does not require a target, so ensure that the bot faces the current one before casting
258  m_ai.FaceTarget(pTarget);
259  if (m_ai.CastSpell(CONE_OF_COLD) == SPELL_CAST_OK)
260  return RETURN_CONTINUE;
261  }
262  if (FROSTBOLT > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget, FROSTBOLT) && !pTarget->HasAura(FROSTBOLT, EFFECT_INDEX_0) && CastSpell(FROSTBOLT, pTarget))
263  return RETURN_CONTINUE;
264  if (FROST_NOVA > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_bot.IsSpellReady(FROST_NOVA) && meleeReach && !pTarget->HasAura(FROST_NOVA, EFFECT_INDEX_0) && CastSpell(FROST_NOVA, pTarget))
265  return RETURN_CONTINUE;
266  if (ICE_LANCE > 0 && !m_ai.IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget, ICE_LANCE) && CastSpell(ICE_LANCE, pTarget))
267  return RETURN_CONTINUE;
268  if (SUMMON_WATER_ELEMENTAL > 0 && CastSpell(SUMMON_WATER_ELEMENTAL))
269  return RETURN_CONTINUE;
270 
271  // Default frost spec action
272  if (FROSTBOLT > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget, FROSTBOLT))
273  return CastSpell(FROSTBOLT, pTarget);
274  /*
275  if (BLIZZARD > 0 && !m_ai.IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget,BLIZZARD) && m_ai.GetAttackerCount() >= 5 && CastSpell(BLIZZARD, pTarget))
276  {
277  m_ai.SetIgnoreUpdateTime(8);
278  return RETURN_CONTINUE;
279  }
280  */
281  break;
282 
283  case MAGE_SPEC_FIRE:
284  if (COMBUSTION > 0 && m_ai.SelfBuff(COMBUSTION) == SPELL_CAST_OK)
285  return RETURN_CONTINUE;
286  if (BLAST_WAVE > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.GetAttackerCount() >= 3 && meleeReach && CastSpell(BLAST_WAVE, pTarget))
287  return RETURN_CONTINUE;
288  // Try to have 3 scorch stacks to let tank build aggro while getting a nice crit% bonus
289  if (IMPROVED_SCORCH > 0 && SCORCH > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE))
290  {
291  if (!pTarget->HasAura(FIRE_VULNERABILITY, EFFECT_INDEX_0) && CastSpell(SCORCH, pTarget)) // no stacks: cast it
292  return RETURN_CONTINUE;
293  else
294  {
295  SpellAuraHolder* holder = pTarget->GetSpellAuraHolder(FIRE_VULNERABILITY);
296  if (holder && (holder->GetStackAmount() < 3) && CastSpell(SCORCH, pTarget))
297  return RETURN_CONTINUE;
298  }
299  }
300  // At least 3 stacks of Scorch: cast an opening fireball
301  if (FIREBALL > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && !pTarget->HasAura(FIREBALL, EFFECT_INDEX_1) && CastSpell(FIREBALL, pTarget))
302  return RETURN_CONTINUE;
303  // 3 stacks of Scorch and fireball DoT: use fire blast if available
304  if (FIRE_BLAST > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_bot.IsSpellReady(FIRE_BLAST) && CastSpell(FIRE_BLAST, pTarget))
305  return RETURN_CONTINUE;
306  if (DRAGONS_BREATH > 0 && !m_ai.IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && meleeReach && CastSpell(DRAGONS_BREATH, pTarget))
307  return RETURN_CONTINUE;
308 
309  // All DoTs, cooldowns used, try to maximise scorch stacks (5) to get a even nicer crit% bonus
310  if (IMPROVED_SCORCH > 0 && SCORCH > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE))
311  {
312  SpellAuraHolder* holder = pTarget->GetSpellAuraHolder(FIRE_VULNERABILITY);
313  if (holder && (holder->GetStackAmount() < 5) && CastSpell(SCORCH, pTarget))
314  return RETURN_CONTINUE;
315  }
316  // Default fire spec action
317  if (FIREBALL > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIREBALL))
318  return CastSpell(FIREBALL, pTarget);
319  /*
320  if (FLAMESTRIKE > 0 && !m_ai.IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget,FLAMESTRIKE) && CastSpell(FLAMESTRIKE, pTarget))
321  return RETURN_CONTINUE;
322  */
323  break;
324 
325  case MAGE_SPEC_ARCANE:
326  if (ARCANE_POWER > 0 && m_bot.IsSpellReady(ARCANE_POWER) && m_ai.IsElite(pTarget) && m_ai.CastSpell(ARCANE_POWER) == SPELL_CAST_OK) // Do not waste Arcane Power on normal NPCs as the bot is likely in a group
327  return RETURN_CONTINUE;
328  if (PRESENCE_OF_MIND > 0 && !m_bot.HasAura(PRESENCE_OF_MIND) && m_bot.IsSpellReady(PRESENCE_OF_MIND) && m_ai.IsElite(pTarget) && m_ai.SelfBuff(PRESENCE_OF_MIND) == SPELL_CAST_OK)
329  return RETURN_CONTINUE;
330  // If bot has presence of mind active, cast long casting time spells
331  if (PRESENCE_OF_MIND && m_bot.HasAura(PRESENCE_OF_MIND))
332  {
333  // Instant Pyroblast, yeah! Tanks will probably hate this, but what do they know about power? Nothing...
334  if (PYROBLAST > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && CastSpell(PYROBLAST, pTarget))
335  return RETURN_CONTINUE;
336  if (FIREBALL > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && CastSpell(FIREBALL, pTarget))
337  return RETURN_CONTINUE;
338  }
339  if (ARCANE_EXPLOSION > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && m_ai.GetAttackerCount() >= 3 && meleeReach && CastSpell(ARCANE_EXPLOSION, pTarget))
340  return RETURN_CONTINUE;
341  // Default arcane spec actions (yes, two fire spells)
342  if (FIRE_BLAST > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_bot.IsSpellReady(FIRE_BLAST) && CastSpell(FIRE_BLAST, pTarget))
343  return RETURN_CONTINUE;
344  if (SLOW > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && m_ai.In_Reach(pTarget, SLOW) && !pTarget->HasAura(SLOW, EFFECT_INDEX_0) && CastSpell(SLOW, pTarget))
345  return RETURN_CONTINUE;
346  if (ARCANE_BLAST > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && m_ai.In_Reach(pTarget, ARCANE_BLAST) && CastSpell(ARCANE_BLAST, pTarget))
347  return RETURN_CONTINUE;
348 
349  if (FIREBALL > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIREBALL))
350  return CastSpell(FIREBALL, pTarget);
351  // If no fireball, arcane missiles
352  if (ARCANE_MISSILES > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && CastSpell(ARCANE_MISSILES, pTarget))
353  {
354  m_ai.SetIgnoreUpdateTime(3);
355  return RETURN_CONTINUE;
356  }
357  break;
358  }
359 
360  // No spec due to low level OR no spell found yet
361  if (FROSTBOLT > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget, FROSTBOLT) && !pTarget->HasAura(FROSTBOLT, EFFECT_INDEX_0) && CastSpell(FROSTBOLT, pTarget))
362  return RETURN_CONTINUE;
363  if (FIREBALL > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIREBALL) && CastSpell(FIREBALL, pTarget)) // Very low levels

378 // Function to keep track of active frost cooldowns to clear with Cold Snap
379 uint8 PlayerbotMageAI::CheckFrostCooldowns()
380 {
381  uint8 uiFrostActiveCooldown = 0;
382  if (FROST_NOVA && !m_bot.IsSpellReady(FROST_NOVA))
383  uiFrostActiveCooldown++;
384  if (ICE_BARRIER && !m_bot.IsSpellReady(ICE_BARRIER))
385  uiFrostActiveCooldown++;
386  if (CONE_OF_COLD && !m_bot.IsSpellReady(CONE_OF_COLD))

426 void PlayerbotMageAI::DoNonCombatActions()
427 {
428  // Remove curse on group members
429  if (m_ai.HasDispelOrder() && DispelPlayer(GetDispelTarget(DISPEL_CURSE)) & RETURN_CONTINUE)
430  return;
431 
432  // Buff armor
433  if (MOLTEN_ARMOR)
434  {
435  if (m_ai.SelfBuff(MOLTEN_ARMOR) == SPELL_CAST_OK)
436  return;
437  }
438  else if (MAGE_ARMOR)
439  {
440  if (m_ai.SelfBuff(MAGE_ARMOR) == SPELL_CAST_OK)
441  return;
442  }
443  else if (ICE_ARMOR)
444  {
445  if (m_ai.SelfBuff(ICE_ARMOR) == SPELL_CAST_OK)
446  return;
447  }
448  else if (FROST_ARMOR)
449  {
450  if (m_ai.SelfBuff(FROST_ARMOR) == SPELL_CAST_OK)
451  return;
452  }
453 
454  if (COMBUSTION && m_bot.IsSpellReady(COMBUSTION) && m_ai.SelfBuff(COMBUSTION) == SPELL_CAST_OK)
455  return;
456 
457  // buff group
458  // the check for group targets is performed by NeedGroupBuff (if group is found for bots by the function)
459  if (NeedGroupBuff(ARCANE_BRILLIANCE, ARCANE_INTELLECT) && m_ai.HasSpellReagents(ARCANE_BRILLIANCE))
460  {
461  if (Buff(&PlayerbotMageAI::BuffHelper, ARCANE_BRILLIANCE) & RETURN_CONTINUE)
462  return;
463  }
464  else if (Buff(&PlayerbotMageAI::BuffHelper, ARCANE_INTELLECT, JOB_MANAONLY) & RETURN_CONTINUE)
465  return;
466 
467  // if there is space on bag try to conjure some consumables
468  if (m_ai.CanStore())
469  {
470  Item* gem = FindManaGem();
471  if (!gem && CONJURE_MANA_GEM && m_ai.CastSpell(CONJURE_MANA_GEM, m_bot) == SPELL_CAST_OK)
472  {
473  m_ai.SetIgnoreUpdateTime(3);
474  return;
475  }
476 
477  // TODO: The beauty of a mage is not only its ability to supply itself with water, but to share its water
478  // So, conjure at *least* 1.25 stacks, ready to trade a stack and still have some left for self
479  if (m_ai.FindDrink() == nullptr && CONJURE_WATER && m_ai.CastSpell(CONJURE_WATER, m_bot) == SPELL_CAST_OK)
480  {
481  m_ai.TellMaster("I'm conjuring some water.");
482  m_ai.SetIgnoreUpdateTime(3);
483  return;
484  }
485 
486  if (m_ai.FindFood() == nullptr && CONJURE_FOOD && m_ai.CastSpell(CONJURE_FOOD, m_bot) == SPELL_CAST_OK)
487  {
488  m_ai.TellMaster("I'm conjuring some food.");
489  m_ai.SetIgnoreUpdateTime(3);
490  return;
491  }
492  }
493 
494  if (EatDrinkBandage())
495  return;
496 } // end DoNonCombatActions
