344  {

346  if (effIdx != EFFECT_INDEX_0 || !caster->AI())
347  return;
348 
349  Unit* target = spell->GetUnitTarget();
350  auto itr = std::find_if(spell->GetTargetList().begin(), spell->GetTargetList().end(), [target](Spell::TargetInfo const& info) { return info.targetGUID == target->GetObjectGuid(); });
351  if (itr == spell->GetTargetList().end())
352  return;
353 
354  caster->getThreatManager().modifyAllThreatPercent(100);
355  caster->AI()->AttackStart(target);
