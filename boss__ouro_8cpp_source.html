216  void ExecuteAction(uint32 action) override
217  {
218  switch (action)
219  {
220  case OURO_SPAWN:
221  {
222  DoCastSpellIfCan(nullptr, SPELL_BIRTH_SPAWN);
223  if (m_creature->GetUInt32Value(UNIT_CREATED_BY_SPELL))
224  {
225  if (Creature* trigger = GetClosestCreatureWithEntry(m_creature, NPC_OURO_TRIGGER, 100.f)) // if not sufficient in the future, do using SPELL_SET_OURO_HEALTH
226  m_creature->SetHealthPercent(static_cast<npc_ouro_triggerAI*>(trigger->AI())->m_ouroHealth);
227  }
228  else
229  {
230  if (m_instance)
231  m_instance->SetData(TYPE_OURO, IN_PROGRESS);
232  }
233  SetCombatScriptStatus(true);
234  if (GameObject* go = m_creature->GetMap()->GetGameObject(m_base))
235  go->SendGameObjectCustomAnim(m_base);
236  ResetTimer(OURO_SPAWN_DELAY_ATTACK, 4000);
237  DisableCombatAction(action);
238  break;
239  }
240  case OURO_ENRAGE:
241  {
242  if (m_creature->GetHealthPercent() < 20.0f)
243  {
244  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
245  {
246  m_firstMound = false; // in order to avoid spawning more ouros
247  ResetCombatAction(OURO_BOULDER, 500);
248  SetActionReadyStatus(action, false);
249  DisableCombatAction(OURO_SUBMERGE);
250  DisableCombatAction(OURO_NO_MELEE_BURROW);
251  return;
252  }
253  }
254  break;
255  }
256  case OURO_SUBMERGE:
257  {
258  Burrow();
259  break;
260  }
261  case OURO_SAND_BLAST:
262  {
263  if (DoCastSpellIfCan(nullptr, SPELL_SANDBLAST) == CAST_OK)
264  ResetCombatAction(action, 25000);
265  break;
266  }
267  case OURO_SWEEP:
268  {
269  if (DoCastSpellIfCan(nullptr, SPELL_SWEEP) == CAST_OK)
270  ResetCombatAction(action, 20000);
271  break;
272  }
273  case OURO_BOULDER:
274  {
275  uint32 timer = 500;
276  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
277  m_rangeCheckState = -1;
278  else
279  {
280  ++m_rangeCheckState;
281  if (m_rangeCheckState > 1)
282  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_BOULDER, SELECT_FLAG_PLAYER))
283  if (DoCastSpellIfCan(target, SPELL_BOULDER) == CAST_OK)
284  timer = 2500;
285  }
286  ResetCombatAction(action, timer);
287  break;
288  }
289  case OURO_NO_MELEE_BURROW:
290  {
291  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
292  m_burrowCounter = 0;
293  else
294  {
295  ++m_burrowCounter;
296  if (m_burrowCounter >= 10)
297  Burrow();
298  }
299  ResetCombatAction(action, 1000);
300  break;
301  }
302  }
303  }
