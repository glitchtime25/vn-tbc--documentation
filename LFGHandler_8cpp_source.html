83 static inline bool LookingForGroupUpdateQueueStatus(Player* _this, WorldSession* _session)
84 {
85  Group* _group = _this->GetGroup();
86 
87  const bool autojoin = (_this->m_lookingForGroup.isAutoJoin() && _session->LookingForGroup_auto_join);
88  const bool autofill = (_this->m_lookingForGroup.isAutoFill() && _session->LookingForGroup_auto_add);
89  const bool notraid = (!_group || (!_group->IsRaidGroup() && _group->IsLeader(_this->GetObjectGuid())));
90 
91  const bool status = ((autojoin && !_group) || (autofill && notraid));
92 
93  if (_session->LookingForGroup_queue == status)
94  return false;
95 
96  _session->LookingForGroup_queue = status;
97 
98  // Sync client's UI queue status
99 
100  const Player* pov = LookingForGroupGetCurrentLeader(_this);
101  const bool queued = pov->GetSession()->LookingForGroup_queue;
102  auto const& info = pov->m_lookingForGroup;
103 
104  WorldPacket data(SMSG_LFG_UPDATE_QUEUED, 1);
105  data << uint8(queued);
106 
107  WorldPacket message;
108  if (queued)
109  LookingForGroupMakeMeetingStoneQueueJoinedFor(message, info.more.entry);
110  else
111  LookingForGroupMakeMeetingStoneQueueLeftFor(message, info.more.entry);
112 
113  if (!_group || !notraid)
114  {
115  _session->SendPacket(data);
116 
117  if (info.isLFM())
118  _session->SendPacket(message);
119  }
120  else
121  {
122  _group->BroadcastPacket(data, true);
123 
124  if (info.isLFM())
125  _group->BroadcastPacket(message, true);
126  }
127  return true;
128 }

195 static inline bool LookingForGroupAddMemberFor(Player* _this, WorldSession* _session, Player* member, uint32 entry, Group*& _group)
196 {
197  const bool create = !_group;
198 
199  if (create)
200  {
201  _group = new Group;
202 
203  if (!_group->Create(_this->GetObjectGuid(), _this->GetName()))
204  {
205  delete _group;
206  _group = nullptr;
207  return false;
208  }
209 
210  sObjectMgr.AddGroup(_group);
211  }
212 
213  const ObjectGuid _guid = member->GetObjectGuid();
214 
215  if (!_group->AddMember(_guid, member->GetName()))
216  return false;
217 
218  WorldPacket data;
219 
220  if (create)
221  {
222  LookingForGroupMakeMeetingStoneQueueMatchedFor(data, entry, true);
223  _session->SendPacket(data);
224  }
225 
226  LookingForGroupMakeMeetingStoneMemberAdded(data, _guid);
227  _group->BroadcastPacket(data, true, -1, _guid);
228 
229  WorldSession* session = member->GetSession();
230 
231  LookingForGroupMakeMeetingStoneQueueMatchedFor(data, entry);
232  session->SendPacket(data);
233 
234  LookingForGroupClear(member, session);
235 
236  if (_group->IsFull())
237  {
238  LookingForGroupClear(_this, _this->GetSession(), true);
239  return false;
240  }
241 
242  return true;
243 }

245 static void LookingForGroupTryJoin(Player* _player, bool initial = false)
246 {
247  WorldSession* _session = _player->GetSession();
248 
249  // skip not can autojoin cases and player group case
250  if (Group* _group = _player->GetGroup())
251  {
252  if (_group->IsBattleGroup() || _group->IsFull() || !_group->IsLeader(_player->GetObjectGuid()))
253  LookingForGroupClear(_player, _session);
254  else
255  LookingForGroupClearLFG(_player, _session);
256  }
257 
258  LookingForGroupUpdateQueueStatus(_player, _session);
259 
260  if (!_session->LookingForGroup_queue || !_player->m_lookingForGroup.isLFG())
261  return;
262 
263  bool attempted = false;
264 
265  // TODO: Guard Player Map
266  HashMapHolder<Player>::MapType const& players = sObjectAccessor.GetPlayers();
267  for (HashMapHolder<Player>::MapType::const_iterator iter = players.begin(); iter != players.end(); ++iter)
268  {
269  Player* plr = iter->second;
270 
271  // skip enemies and self
272  if (!plr || plr == _player || plr->GetTeam() != _player->GetTeam())
273  continue;
274 
275  WorldSession* session = plr->GetSession();
276 
277  // skip players not in world or reconnecting
278  if (!plr->IsInWorld() || session->IsOffline())
279  continue;
280 
281  Group* grp = plr->GetGroup();
282 
283  // skip players in not compatinle groups and dequeue them if discovered
284  if (grp && (grp->IsBattleGroup() || grp->IsFull() || !grp->IsLeader(_player->GetObjectGuid())))
285  {
286  LookingForGroupClear(plr, session);
287  continue;
288  }
289 
290  LookingForGroupUpdateQueueStatus(plr, session);
291 
292  // skip not queued
293  if (!session->LookingForGroup_queue || plr->m_lookingForGroup.more.empty())
294  continue;
295 
296  // skip not fitting slots
297  if (!_player->m_lookingForGroup.isLFG(plr->m_lookingForGroup, true))
298  continue;
299 
300  Group* result = grp;
301 
302  // stop at join success
303  if (LookingForGroupAddMemberFor(plr, session, _player, plr->m_lookingForGroup.more.entry, result))
304  break;
305  else if (!grp)
306  attempted = true;
307  // group is full or cant join for other reasons, continue
308  }
309 
310  if (!initial && attempted && _session->LookingForGroup_queue)
311  _session->SendMeetingStoneInProgress();
312 }

314 static void LookingForGroupTryFill(Player* _player, bool initial = false)
315 {
316  WorldSession* _session = _player->GetSession();
317 
318  Group* _group = _player->GetGroup();
319 
320  // skip non auto-join slot or player in a battleground, not group leader, group is full cases
321  if (_group && (_group->IsBattleGroup() || _group->IsFull() || !_group->IsLeader(_player->GetObjectGuid())))
322  {
323  LookingForGroupClear(_player, _session);
324  return;
325  }
326 
327  LookingForGroupUpdateQueueStatus(_player, _session);
328 
329  if (!_session->LookingForGroup_queue || _player->m_lookingForGroup.more.empty())
330  return;
331 
332  bool attempted = false;
333 
334  // TODO: Guard Player map
335  HashMapHolder<Player>::MapType const& players = sObjectAccessor.GetPlayers();
336  for (HashMapHolder<Player>::MapType::const_iterator iter = players.begin(); iter != players.end(); ++iter)
337  {
338  Player* plr = iter->second;
339 
340  // skip enemies and self
341  if (!plr || plr == _player || plr->GetTeam() != _player->GetTeam())
342  continue;
343 
344  WorldSession* session = plr->GetSession();
345 
346  // skip players not in world or reconnecting
347  if (!plr->IsInWorld() || session->IsOffline())
348  continue;
349 
350  // skip players in groups, dequeue and remove them from LFG if discovered
351  if (Group* grp = plr->GetGroup())
352  {
353  if (grp->IsBattleGroup() || grp->IsFull() || !grp->IsLeader(plr->GetObjectGuid()))
354  LookingForGroupClear(plr, session);
355  else
356  LookingForGroupClearLFG(plr, session);
357  continue;
358  }
359 
360  LookingForGroupUpdateQueueStatus(plr, session);
361 
362  // skip not queued
363  if (!session->LookingForGroup_queue || !plr->m_lookingForGroup.isLFG())
364  continue;
365 
366  // skip not fitting slots
367  if (!plr->m_lookingForGroup.isLFG(_player->m_lookingForGroup, true))
368  continue;
369 
370  Group* result = _group;
371 
372  // stop at false result (full?)
373  if (!LookingForGroupAddMemberFor(_player, _session, plr, _player->m_lookingForGroup.more.entry, result))
374  {
375  attempted = true;
376  break;
377  }
378  // group is not full yet, continue
379  }
380 
381  if (!initial && attempted && _session->LookingForGroup_queue)
382  _session->SendMeetingStoneInProgress();
383 }

482 void WorldSession::HandleSetLfgOpcode(WorldPacket& recv_data)
483 {
484  DEBUG_LOG("CMSG_SET_LOOKING_FOR_GROUP");
485 
486  uint32 slot, data;
487 
488  recv_data >> slot >> data;
489 
490  if (slot >= MAX_LOOKING_FOR_GROUP_SLOT)
491  return;
492 
493  if (Group* _group = _player->GetGroup())
494  {
495  const Player* pov = LookingForGroupGetCurrentLeader(_player);
496 
497  if (pov != _player && pov->m_lookingForGroup.isLFM())
498  {
499  WorldPacket data(SMSG_LFG_LEADER_IS_LFM);
500  SendPacket(data);
501  return;
502  }
503 
504  SendLFGUpdate();
505  return;
506  }
507 
508  uint16 entry = (data & 0xFFFF);
509  uint16 type = ((data >> 24) & 0xFFFF);
510 
511  DEBUG_LOG("LFG set: looknumber %u, temp %X, type %u, entry %u", slot, data, type, entry);
512  _player->m_lookingForGroup.group[slot].set(entry, type);
513  LookingForGroupClearLFM(_player, this);
514 
515  LookingForGroupUpdateQueueStatus(_player, this);
516 
517  if (LookingForGroup_auto_join)
518  LookingForGroupTryJoin(_player);
519 }

605 void WorldSession::SendLFGListQueryResponse(LfgType type, uint32 entry)
606 {
607  // start prepare packet
608  WorldPacket data(MSG_LOOKING_FOR_GROUP);
609  data << uint32(type); // type
610  data << uint32(entry); // entry from LFGDungeons.dbc
611  data << uint32(0); // displayed players count, placeholder
612  data << uint32(0); // found players count, placeholder
613 
614  uint32 displayed = 0;
615  uint32 found = 0;
616 
617  // TODO: Guard Player map
618  HashMapHolder<Player>::MapType const& players = sObjectAccessor.GetPlayers();
619  for (const auto& i : players)
620  {
621  Player* plr = i.second;
622 
623  if (!plr || plr->GetTeam() != _player->GetTeam())
624  continue;
625 
626  if (!plr->IsInWorld() || plr->GetSession()->IsOffline())
627  continue;
628 
629  if (!plr->m_lookingForGroup.isLFG(entry, type, false) && !plr->m_lookingForGroup.isLFM(entry, type))
630  continue;
631 
632  const Group* grp = plr->GetGroup();
633 
634  if (grp && (grp->IsBattleGroup() || grp->IsFull() || !grp->IsLeader(plr->GetObjectGuid())))
635  continue;
636 
637  ++found;
638 
639  // Client hardcoded limitation on amount of players sent in the packet handler and UI:
640  if (found > 50)
641  continue;
642 
643  ++displayed;
644 
645  const bool isLFM = plr->m_lookingForGroup.isLFM(entry, type);
646 
647  data << plr->GetPackGUID(); // packed guid
648  data << uint32(plr->GetLevel()); // level
649  data << uint32(plr->GetZoneId()); // current zone
650  data << uint8(isLFM); // 0x00 - LFG, 0x01 - LFM
651 
652  if (isLFM)
653  {
654  data << uint32(plr->m_lookingForGroup.more.entry | (plr->m_lookingForGroup.more.type << 24));
655  data << uint32(0x1000000);
656  data << uint32(0x1000000);
657  }
658  else
659  {
660  for (uint8 j = 0; j < MAX_LOOKING_FOR_GROUP_SLOT; ++j)
661  data << uint32(plr->m_lookingForGroup.group[j].entry | (plr->m_lookingForGroup.group[j].type << 24));
662  }
663 
664  data << plr->m_lookingForGroup.comment;
665 
666  data << uint32(0); // other group members count, placeholder
667 
668  if (grp)
669  {
670  const size_t offset = (data.wpos() - 4); // other group members count, offset
671  uint32 count = 0; // other group members count
672 
673  for (const GroupReference* itr = grp->GetFirstMember(); itr != nullptr; itr = itr->next())
674  {
675  if (const Player* member = itr->getSource())
676  {
677  if (member->GetObjectGuid() != plr->GetObjectGuid())
678  {
679  data << member->GetPackGUID(); // packed guid
680  data << uint32(member->GetLevel()); // player level
681  ++count;
682  }
683  }
684  }
685 
686  data.put<uint32>(offset, count); // other group members count, fill the placeholder
687  }
688  }
689 
690  // fill count placeholders
691  data.put<uint32>(4 + 4, displayed);
692  data.put<uint32>(4 + 4 + 4, found);
693 
694  SendPacket(data);
695 }

713 void WorldSession::SendLFGUpdateLFG()
714 {
715  // Syncs player's own LFG UI selection with what we send here
716  auto const& selection = _player->m_lookingForGroup.group;
717
