49 bool ProcessEventTransports(uint32 uiEventId, Object* pSource, Object* /*pTarget*/, bool /*bIsStart*/)
50 {
51  sWorldState.HandleConditionStateChange(pSource->GetEntry(), uiEventId);
52 
53  WorldObject* transport = (WorldObject*)pSource;
54  uint32 entry = 0;
55  switch (uiEventId)
56  {
57  case EVENT_UC_FROM_GROMGOL_ARRIVAL: // UC arrival from gromgol
58  entry = NPC_HINDENBURG;
59  break;
60  case EVENT_GROMGOL_FROM_UC_ARRIVAL: // gromgol arrival from UC
61  entry = NPC_SQUIBBY_OVERSPECK;
62  break;
63  case EVENT_OG_FROM_UC_ARRIVAL: // OG arrival from UC
64  entry = NPC_FREZZA;
65  break;
66  case EVENT_UC_FROM_OG_ARRIVAL: // UC arrival from OG
67  entry = NPC_ZAPETTA;
68  break;
69  case EVENT_OG_FROM_GROMGOL_ARRIVAL: // OG arrival from gromgol
70  entry = NPC_SNURK_BUCKSQUICK;
71  break;
72  case EVENT_GROMGOL_FROM_OG_ARRIVAL: // gromgol arrival from OG
73  entry = NPC_NEZRAZ;
74  break;
75  }
76  if (entry)
77  if (Creature* zeppelinMaster = static_cast<ScriptedInstance*>(transport->GetMap()->GetInstanceData())->GetSingleCreatureFromStorage(entry, true))
78  zeppelinMaster->PlayDistanceSound(SOUND_ZEPPELIN_HORN);
79 
80  return true;
81 }

119  checkAreaId = AREAID_FALCONWING_SQUARE;

121  }
122

125

137  return (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_GOLDSHIRE] == SHADE_PHASE_VICTORY && areaId == AREAID_GOLDSHIRE) ||

180 bool ShadeOfTheHorsemanData::Update(const uint32 diff)
181 {
182  if (!sGameEventMgr.IsActiveHoliday(HOLIDAY_HALLOWS_END))
183  return false;
184 
185  if (m_horsemanTimer <= diff)
186  {
187  std::time_t now = time(nullptr);
188  m_horsemanTime = *std::gmtime(&now);
189  m_horsemanTimer = 60 * IN_MILLISECONDS;
190  uint32 timerValue = CalculateWorldstateTimerValue();
191  for (uint32 areaId : m_areaIds)
192  {
193  sWorldState.ExecuteOnAreaPlayers(areaId, [=](Player* player)->void
194  {
195  player->SendUpdateWorldState(WORLD_STATE_SHADE_OF_THE_HORSEMAN_TIMER, timerValue);
196  });
197  }
198  if (timerValue == 0)
199  return true;
200  }
201  else m_horsemanTimer -= diff;
202 
203  return false;
204 }
