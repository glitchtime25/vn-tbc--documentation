222 void MailDraft::SendReturnToSender(uint32 sender_acc, ObjectGuid sender_guid, ObjectGuid receiver_guid)
223 {
224  Player* receiver = sObjectMgr.GetPlayer(receiver_guid);
225 
226  uint32 rc_account = 0;
227  if (!receiver)
228  rc_account = sObjectMgr.GetPlayerAccountIdByGUID(receiver_guid);
229 
230  if (!receiver && !rc_account) // sender not exist
231  {
232  deleteIncludedItems(true);
233  return;
234  }
235 
236  // prepare mail and send in other case
237  bool needItemDelay = false;
238 
239  if (!m_items.empty())
240  {
241  // if item send to character at another account, then apply item delivery delay
242  needItemDelay = sender_acc != rc_account;
243 
244  // set owner to new receiver (to prevent delete item with sender char deleting)
245  CharacterDatabase.BeginTransaction();
246  for (auto& m_item : m_items)
247  {
248  Item* item = m_item.second;
249  item->SaveToDB(); // item not in inventory and can be save standalone
250  // owner in data will set at mail receive and item extracting
251  CharacterDatabase.PExecute("UPDATE item_instance SET owner_guid = '%u' WHERE guid='%u'", receiver_guid.GetCounter(), item->GetGUIDLow());
252  }
253  CharacterDatabase.CommitTransaction();
254  }
255 
256  // If theres is an item, there is a one hour delivery delay.
257  uint32 deliver_delay = needItemDelay ? sWorld.getConfig(CONFIG_UINT32_MAIL_DELIVERY_DELAY) : 0;
258 
259  // will delete item or place to receiver mail list
260  SendMailTo(MailReceiver(receiver, receiver_guid), MailSender(MAIL_NORMAL, sender_guid.GetCounter()), MAIL_CHECK_MASK_RETURNED, deliver_delay);
261 }
270 void MailDraft::SendMailTo(MailReceiver const& receiver, MailSender const& sender, MailCheckMask checked, uint32 deliver_delay)
271 {
272  Player* pReceiver = receiver.GetPlayer(); // can be nullptr
273 
274  uint32 pReceiverAccount = 0;
275  if (!pReceiver)
276  pReceiverAccount = sObjectMgr.GetPlayerAccountIdByGUID(receiver.GetPlayerGuid());
277 
278  if (!pReceiver && !pReceiverAccount) // receiver not exist
279  {
280  deleteIncludedItems(true);
281  return;
282  }
283 
284  bool has_items = !m_items.empty();
285 
286  // generate mail template items for online player, for offline player items will generated at open
287  if (pReceiver)
288  {
289  if (prepareItems(pReceiver))
290  has_items = true;
291  }
292 
293  uint32 mailId = sObjectMgr.GenerateMailID();
294 
295  time_t deliver_time = time(nullptr) + deliver_delay;
296 
297  // expire time if COD 3 days, if no COD 30 days, if auction sale pending 1 hour
298  uint32 expire_delay;
299  // auction mail without any items and money (auction sale note) pending 1 hour
300  if (sender.GetMailMessageType() == MAIL_AUCTION && m_items.empty() && !m_money)
301  expire_delay = HOUR;
302  // mail from battlemaster (rewardmarks) should last only one day
303  else if (sender.GetMailMessageType() == MAIL_CREATURE && sBattleGroundMgr.GetBattleMasterBG(sender.GetSenderId()) != BATTLEGROUND_TYPE_NONE)
304  expire_delay = DAY;
305  // default case: expire time if COD 3 days, if no COD 30 days
306  else
307  expire_delay = (m_COD > 0) ? 3 * DAY : 30 * DAY;
308 
309  time_t expire_time = deliver_time + expire_delay;
310 
311  // Add to DB
312  std::string safe_subject = GetSubject();
313 
314  CharacterDatabase.BeginTransaction();
315  CharacterDatabase.escape_string(safe_subject);
316  CharacterDatabase.PExecute("INSERT INTO mail (id,messageType,stationery,mailTemplateId,sender,receiver,subject,itemTextId,has_items,expire_time,deliver_time,money,cod,checked) "
317  "VALUES ('%u', '%u', '%u', '%u', '%u', '%u', '%s', '%u', '%u', '" UI64FMTD "','" UI64FMTD "', '%u', '%u', '%u')",
318  mailId, sender.GetMailMessageType(), sender.GetStationery(), GetMailTemplateId(), sender.GetSenderId(), receiver.GetPlayerGuid().GetCounter(), safe_subject.c_str(), GetBodyId(), (has_items ? 1 : 0), (uint64)expire_time, (uint64)deliver_time, m_money, m_COD, checked);
319 
320  for (MailItemMap::const_iterator mailItemIter = m_items.begin(); mailItemIter != m_items.end(); ++mailItemIter)
321  {
322  Item* item = mailItemIter->second;
323  CharacterDatabase.PExecute("INSERT INTO mail_items (mail_id,item_guid,item_template,receiver) VALUES ('%u', '%u', '%u','%u')",
324  mailId, item->GetGUIDLow(), item->GetEntry(), receiver.GetPlayerGuid().GetCounter());
325  }
326  CharacterDatabase.CommitTransaction();
327 
328  // For online receiver update in game mail status and data
329  if (pReceiver)
330  {
331  pReceiver->AddNewMailDeliverTime(deliver_time);
332 
333  Mail* m = new Mail;
334  m->messageID = mailId;
335  m->mailTemplateId = GetMailTemplateId();
336  m->subject = GetSubject();
337  m->itemTextId = GetBodyId();
338  m->money = GetMoney();
339  m->COD = GetCOD();
340 
341  for (MailItemMap::const_iterator mailItemIter = m_items.begin(); mailItemIter != m_items.end(); ++mailItemIter)
342  {
343  Item* item = mailItemIter->second;
344  m->AddItem(item->GetGUIDLow(), item->GetEntry());
345  }
346 
347  m->messageType = sender.GetMailMessageType();
348  m->stationery = sender.GetStationery();
349  m->sender = sender.GetSenderId();
350  m->receiverGuid = receiver.GetPlayerGuid();
351  m->expire_time = expire_time;
352  m->deliver_time = deliver_time;
353  m->checked = checked;
354  m->state = MAIL_STATE_UNCHANGED;
355 
356  pReceiver->AddMail(m); // to insert new mail to beginning of maillist
357 
358  if (!m_items.empty())
359  {
360  for (auto& m_item : m_items)
361  pReceiver->AddMItem(m_item.second);
362  }
363  }
364  else if (!m_items.empty())
365  deleteIncludedItems();
366 }
