286 void BattleGround::Update(uint32 diff)
287 {
288  // remove offline players from bg after 5 minutes
289  if (!m_offlineQueue.empty())
290  {
291  BattleGroundPlayerMap::iterator itr = m_players.find(*(m_offlineQueue.begin()));
292  if (itr != m_players.end())
293  {
294  if (itr->second.offlineRemoveTime <= sWorld.GetGameTime())
295  {
296  // add deserter at next login
297  CharacterDatabase.PExecute("UPDATE characters SET at_login = at_login | '%u' WHERE guid = '%u'", uint32(AT_LOGIN_ADD_BG_DESERTER), itr->first.GetCounter());
298 
299  RemovePlayerAtLeave(itr->first, true, true);// remove player from BG
300  m_offlineQueue.pop_front(); // remove from offline queue
301  // do not use itr for anything, because it is erased in RemovePlayerAtLeave()
302  }
303  }
304  }
305 
306  /*********************************************************/
307  /*** BATTLEGROUND BALLANCE SYSTEM ***/
308  /*********************************************************/
309 
310  // if less then minimum players are in on one side, then start premature finish timer
311  if (GetStatus() == STATUS_IN_PROGRESS && !IsArena() && sBattleGroundMgr.GetPrematureFinishTime() && (GetPlayersCountByTeam(ALLIANCE) < GetMinPlayersPerTeam() || GetPlayersCountByTeam(HORDE) < GetMinPlayersPerTeam()))
312  {
313  if (!m_prematureCountDown)
314  {
315  m_prematureCountDown = true;
316  m_prematureCountDownTimer = sBattleGroundMgr.GetPrematureFinishTime();
317  }
318  else if (m_prematureCountDownTimer < diff)
319  {
320  EndBattleGround(GetPrematureWinner());
321  m_prematureCountDown = false;
322  }
323  else if (!sBattleGroundMgr.IsTesting())
324  {
325  uint32 newtime = m_prematureCountDownTimer - diff;
326  // announce every minute
327  if (newtime > (MINUTE * IN_MILLISECONDS))
328  {
329  if (newtime / (MINUTE * IN_MILLISECONDS) != m_prematureCountDownTimer / (MINUTE * IN_MILLISECONDS))
330  PSendMessageToAll(LANG_BATTLEGROUND_PREMATURE_FINISH_WARNING, CHAT_MSG_SYSTEM, nullptr, (uint32)(m_prematureCountDownTimer / (MINUTE * IN_MILLISECONDS)));
331  }
332  else
333  {
334  // announce every 15 seconds
335  if (newtime / (15 * IN_MILLISECONDS) != m_prematureCountDownTimer / (15 * IN_MILLISECONDS))
336  PSendMessageToAll(LANG_BATTLEGROUND_PREMATURE_FINISH_WARNING_SECS, CHAT_MSG_SYSTEM, nullptr, (uint32)(m_prematureCountDownTimer / IN_MILLISECONDS));
337  }
338  m_prematureCountDownTimer = newtime;
339  }
340  }
341  else if (m_prematureCountDown)
342  m_prematureCountDown = false;
343 
344  /*********************************************************/
345  /*** ARENA BUFF OBJECT SPAWNING ***/
346  /*********************************************************/
347  if (IsArena() && !m_arenaBuffSpawned)
348  {
349  // 60 seconds after start the buffobjects in arena should get spawned
350  if (m_startTime > uint32(m_startDelayTimes[BG_STARTING_EVENT_FIRST] + ARENA_SPAWN_BUFF_OBJECTS))
351  {
352  SpawnEvent(ARENA_BUFF_EVENT, 0, true);
353  m_arenaBuffSpawned = true;
354  }
355  }
356 
357  /*********************************************************/
358  /*** BATTLEGROUND STARTING SYSTEM ***/
359  /*********************************************************/
360 
361  if (GetStatus() == STATUS_WAIT_JOIN && GetPlayersSize())
362  {
363  float maxDist = GetStartMaxDist();
364  if (maxDist > 0.0f)
365  {
366  if (m_validStartPositionTimer < diff)
367  {
368  for (const auto& itr : GetPlayers())
369  {
370  if (Player* player = sObjectMgr.GetPlayer(itr.first))
371  {
372  float x, y, z, o;
373  GetTeamStartLoc(player->GetTeam(), x, y, z, o);
374  if (!player->IsWithinDist3d(x, y, z, maxDist))
375  {
376  player->TeleportTo(GetMapId(), x, y, z, o);
377  }
378  }
379  }
380  m_validStartPositionTimer = CHECK_PLAYER_POSITION_INVERVAL;
381  }
382  else
383  m_validStartPositionTimer -= diff;
384  }
385 
386  ModifyStartDelayTime(diff);
387 
388  if (!(m_events & BG_STARTING_EVENT_1))
389  {
390  m_events |= BG_STARTING_EVENT_1;
391 
392  StartingEventCloseDoors();
393  SetStartDelayTime(m_startDelayTimes[BG_STARTING_EVENT_FIRST]);
394  // first start warning - 2 or 1 minute, only if defined
395  if (m_startMessageIds[BG_STARTING_EVENT_FIRST])
396  SendMessageToAll(m_startMessageIds[BG_STARTING_EVENT_FIRST], CHAT_MSG_BG_SYSTEM_NEUTRAL);
397 
398  // Announce BG Started
399  if (sWorld.getConfig(CONFIG_BOOL_BATTLEGROUND_QUEUE_ANNOUNCER_START))
400  {
401  BattleGroundBracketId bracketId = GetBracketId();
402  BattleGroundTypeId BgTypeId = GetTypeId();
403 
404  if (!IsArena())
405  sWorld.SendWorldText(LANG_BG_STARTED_ANNOUNCE_WORLD, GetName(), Player::GetMinLevelForBattleGroundBracketId(bracketId, BgTypeId), Player::GetMaxLevelForBattleGroundBracketId(bracketId, BgTypeId));
406  else
407  sWorld.SendWorldText(LANG_ARENA_STARTED_ANNOUNCE_WORLD, GetName(), GetArenaType(), GetArenaType());
408  }
409  }
410  // After 1 minute or 30 seconds, warning is signalled
411  else if (GetStartDelayTime() <= m_startDelayTimes[BG_STARTING_EVENT_SECOND] && !(m_events & BG_STARTING_EVENT_2))
412  {
413  m_events |= BG_STARTING_EVENT_2;
414  SendMessageToAll(m_startMessageIds[BG_STARTING_EVENT_SECOND], CHAT_MSG_BG_SYSTEM_NEUTRAL);
415  }
416  // After 30 or 15 seconds, warning is signalled
417  else if (GetStartDelayTime() <= m_startDelayTimes[BG_STARTING_EVENT_THIRD] && !(m_events & BG_STARTING_EVENT_3))
418  {
419  m_events |= BG_STARTING_EVENT_3;
420  SendMessageToAll(m_startMessageIds[BG_STARTING_EVENT_THIRD], CHAT_MSG_BG_SYSTEM_NEUTRAL);
421  }
422  // delay expired (atfer 2 or 1 minute)
423  else if (GetStartDelayTime() <= 0 && !(m_events & BG_STARTING_EVENT_4))
424  {
425  m_events |= BG_STARTING_EVENT_4;
426 
427  StartingEventOpenDoors();
428 
429  if (m_startMessageIds[BG_STARTING_EVENT_FOURTH])
430  SendMessageToAll(m_startMessageIds[BG_STARTING_EVENT_FOURTH], CHAT_MSG_BG_SYSTEM_NEUTRAL);
431 
432  SetStatus(STATUS_IN_PROGRESS);
433  SetStartDelayTime(m_startDelayTimes[BG_STARTING_EVENT_FOURTH]);
434 
435  // remove preparation
436  if (IsArena())
437  {
438  for (BattleGroundPlayerMap::const_iterator itr = m_players.begin(); itr != m_players.end(); ++itr)
439  {
440  if (Player* player = sObjectMgr.GetPlayer(itr->first))
441  player->RemoveAurasDueToSpell(SPELL_ARENA_PREPARATION);
442  }
443 
444  CheckArenaWinConditions();
445  }
446  else
447  {
448  PlaySoundToAll(SOUND_BG_START);
449 
450  for (BattleGroundPlayerMap::const_iterator itr = m_players.begin(); itr != m_players.end(); ++itr)
451  if (Player* plr = sObjectMgr.GetPlayer(itr->first))
452  plr->RemoveAurasDueToSpell(SPELL_PREPARATION);
453  }
454  }
455  }
456 
457  /*********************************************************/
458  /*** BATTLEGROUND ENDING SYSTEM ***/
459  /*********************************************************/
460  if (GetStatus() == STATUS_IN_PROGRESS && IsArena())
461  {
462  // after 45 minutes without one team losing, the arena closes with no winner and -16 rating change for both
463  if (m_startTime > uint32(m_startDelayTimes[BG_STARTING_EVENT_FIRST] + ARENA_FORCED_DRAW))
464  {
465  EndBattleGround(TEAM_NONE);
466  }
467  }
468 
469  if (GetStatus() == STATUS_WAIT_LEAVE)
470  {
471  // remove all players from battleground after 2 minutes
472  m_endTime -= diff;
473  if (m_endTime <= 0)
474  {
475  m_endTime = 0;
476  BattleGroundPlayerMap::iterator next;
477  for (BattleGroundPlayerMap::iterator itr = m_players.begin(); itr != m_players.end(); itr = next)
478  {
479  next = itr;
480  ++next;
481  // itr is erased here!
482  RemovePlayerAtLeave(itr->first, true, true);// remove player from BG
483  // do not change any battleground's private variables
484  }
485  }
486  }
487 
488  // update start time
489  m_startTime += diff;
490 }
