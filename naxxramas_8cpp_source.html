446  {
447  DoUseDoorOrButton(GO_ARAC_EYE_RAMP);

452  }
453  break;

455  m_auiEncounter[type] = data;
456  DoUseDoorOrButton(GO_PLAG_NOTH_ENTRY_DOOR);

459  DoUseDoorOrButton(GO_PLAG_NOTH_EXIT_DOOR);
460  DoUseDoorOrButton(GO_PLAG_HEIG_ENTRY_DOOR);
461  }
462  break;
463  case TYPE_HEIGAN:
464  m_auiEncounter[type] = data;
465  // Open the entrance door on encounter win or failure (we specifically set the GOState to avoid issue in case encounter is reset before gate is closed in Heigan script)
466  if (data == DONE || data == FAIL)
467  {
468  if (GameObject* door = GetSingleGameObjectFromStorage(GO_PLAG_HEIG_ENTRY_DOOR))
469  door->SetGoState(GO_STATE_ACTIVE);
470  }
471  if (data == DONE)
472  DoUseDoorOrButton(GO_PLAG_HEIG_EXIT_HALLWAY);
473  break;
474  case TYPE_LOATHEB:
475  m_auiEncounter[type] = data;
476  DoUseDoorOrButton(GO_PLAG_LOAT_DOOR);
477  if (data == DONE)
478  {
479  DoUseDoorOrButton(GO_PLAG_EYE_RAMP);
480  DoUseDoorOrButton(GO_PLAG_EYE_BOSS);
481  DoRespawnGameObject(GO_PLAG_PORTAL, 30 * MINUTE);
482  DoToggleGameObjectFlags(GO_PLAG_PORTAL, GO_FLAG_NO_INTERACT, false);
483  m_tauntTimer = 5000;
484  }
485  break;
486  case TYPE_RAZUVIOUS:
487  m_auiEncounter[type] = data;
488  break;
489  case TYPE_GOTHIK:
490  m_auiEncounter[type] = data;
491  switch (data)
492  {
493  case IN_PROGRESS:
494  // Encounter begins: close the gate and start timer to summon unrelenting trainees
495  DoUseDoorOrButton(GO_MILI_GOTH_ENTRY_GATE);
496  DoUseDoorOrButton(GO_MILI_GOTH_COMBAT_GATE);
497  InitializeGothikTriggers();
498  break;
499  case SPECIAL:
500  DoUseDoorOrButton(GO_MILI_GOTH_COMBAT_GATE);
501  for (auto& spectralGuid : m_spectralSideList)
502  {
503  if (Creature* spectral = instance->GetCreature(spectralGuid))
504  spectral->CastSpell(spectral, SPELL_SPECTRAL_ASSAULT, TRIGGERED_OLD_TRIGGERED);
505  }
506  for (auto& unrelentingGuid : m_unrelentingSideList)
507  {
508  if (Creature* unrelenting = instance->GetCreature(unrelentingGuid))
509  unrelenting->CastSpell(unrelenting, SPELL_UNRELENTING_ASSAULT, TRIGGERED_OLD_TRIGGERED);
510  }
511  break;
512  case FAIL:
513  if (m_auiEncounter[type] == IN_PROGRESS)
514  DoUseDoorOrButton(GO_MILI_GOTH_COMBAT_GATE);
515  DoUseDoorOrButton(GO_MILI_GOTH_ENTRY_GATE);
516  break;
517  case DONE:
518  DoUseDoorOrButton(GO_MILI_GOTH_ENTRY_GATE);
519  DoUseDoorOrButton(GO_MILI_GOTH_EXIT_GATE);
520  DoUseDoorOrButton(GO_MILI_HORSEMEN_DOOR);
521  // Open the central gate if Gothik is defeated before doing so
522  if (m_auiEncounter[type] == IN_PROGRESS)
523  DoUseDoorOrButton(GO_MILI_GOTH_COMBAT_GATE);
524  StartNextDialogueText(NPC_THANE);
525  break;
526  }
527  break;
528  case TYPE_FOUR_HORSEMEN:
529  {
530  // Skip if already set
531  if (m_auiEncounter[type] == data)
532  return;
533  if (data == SPECIAL)
534  {
535  ++m_horsemenKilled;
536 
537  if (m_horsemenKilled == 4)
538  SetData(TYPE_FOUR_HORSEMEN, DONE);
539 
540  // Don't store special data
541  break;
542  }
543  if (data == FAIL)
544  {
545  m_horsemenKilled = 0;
546  std::vector<uint32> entries = {NPC_BLAUMEUX, NPC_MOGRAINE, NPC_ZELIEK, NPC_THANE};
547  for (uint32 entry : entries)
548  {
549  if (Creature* creature = GetSingleCreatureFromStorage(entry))
550  {
551  if (creature->IsInCombat())
552  creature->AI()->EnterEvadeMode();
553  }
554  }
555  }
556  m_auiEncounter[type] = data;
557  DoUseDoorOrButton(GO_MILI_HORSEMEN_DOOR);
558  // Despawn Horsemen's spirit on FAIL or DONE
559  std::vector<uint32> entries = {NPC_SPIRIT_OF_BLAUMEUX, NPC_SPIRIT_OF_MOGRAINE, NPC_SPIRIT_OF_KORTHAZZ, NPC_SPIRIT_OF_ZELIREK};
560  for (uint32 entry : entries)
561  {
562  if (Creature* spirit = GetSingleCreatureFromStorage(entry))
563  spirit->ForcedDespawn();
564  }
565  if (data == DONE)
566  {
567  DoUseDoorOrButton(GO_MILI_EYE_RAMP);
568  DoUseDoorOrButton(GO_MILI_EYE_BOSS);
