175 uint32 Quest::XPValue(Player* pPlayer) const
176 {
177  if (pPlayer)
178  {
179  if (RewMoneyMaxLevel > 0)
180  {
181  uint32 pLevel = pPlayer->GetLevel();
182  uint32 qLevel = QuestLevel > 0 ? (uint32)QuestLevel : 0;
183  float fullxp = 0;
184  if (qLevel >= 65)
185  fullxp = RewMoneyMaxLevel / 6.0f;
186  else if (qLevel == 64)
187  fullxp = RewMoneyMaxLevel / 4.8f;
188  else if (qLevel == 63)
189  fullxp = RewMoneyMaxLevel / 3.6f;
190  else if (qLevel == 62)
191  fullxp = RewMoneyMaxLevel / 2.4f;
192  else if (qLevel == 61)
193  fullxp = RewMoneyMaxLevel / 1.2f;
194  else if (qLevel > 0 && qLevel <= 60)
195  fullxp = RewMoneyMaxLevel / 0.6f;
196 
197  if (pLevel <= qLevel + 5)
198  return uint32(ceilf(fullxp));
199  else if (pLevel == qLevel + 6)
200  return uint32(ceilf(fullxp * 0.8f));
201  else if (pLevel == qLevel + 7)
202  return uint32(ceilf(fullxp * 0.6f));
203  else if (pLevel == qLevel + 8)
204  return uint32(ceilf(fullxp * 0.4f));
205  else if (pLevel == qLevel + 9)
206  return uint32(ceilf(fullxp * 0.2f));
207  else
208  return uint32(ceilf(fullxp * 0.1f));
209  }
210  }
211 
212  return 0;
213 }
