97  void UpdateAI(const uint32 diff) override
98  {
99  if (isSpawnedForHallowsEnd)
100  {
101  if (relightingFires)
102  {
103  if (relightTimer)
104  {
105  if (relightTimer < diff)
106  {
107  relightTimer = 3000;
108 
109  if (DoCastSpellIfCan(nullptr, SPELL_START_FIRE_TARGET_TEST_GUARD, CAST_TRIGGERED) == CAST_OK)
110  relitCount++;
111 
112  if (relitCount == lPracticeFires.size())
113  {
114  relitCount = 0;
115  relightingFires = false;
116  checkExtinguishedTimer = 3000;
117  }
118  }
119  else
120  relightTimer -= diff;
121  }
122  }
123  else
124  {
125  if (checkExtinguishedTimer)
126  {
127  if (checkExtinguishedTimer < diff)
128  {
129  // go home if necessary
130  if (m_creature->GetPositionX() != spawnPos.x || m_creature->GetPositionY() != spawnPos.y || m_creature->GetPositionZ() != spawnPos.z)
131  {
132  m_creature->GetMotionMaster()->MovePoint(0, Position(spawnPos.x, spawnPos.y, spawnPos.z, spawnPos.o));
133  m_creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
134  }
135 
136  checkExtinguishedTimer = 3000;
137  extinguishedCount = 0;
138  if (lPracticeFires.size() < 3)
139  {
140  std::list<Creature*> lPracticeFireCreatures;
141  GetCreatureListWithEntryInGrid(lPracticeFireCreatures, m_creature, NPC_HEADLESS_HORSEMAN_FIRE, 65.f);
142  for (auto& fire : lPracticeFireCreatures)
143  lPracticeFires.push_back(fire->GetObjectGuid());
144  }
145  for (ObjectGuid fireGuid : lPracticeFires)
146  if (Creature* fire = m_creature->GetMap()->GetCreature(fireGuid))
147  if (!fire->HasAura(SPELL_FIRE))
148  extinguishedCount++;
149 
150  if (extinguishedCount == lPracticeFires.size())
151  {
152  DoScriptText(SAY_GUARD_RELIGHT_FIRE, m_creature);
153  relightingFires = true;
154  m_creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
155 
156  switch (m_creature->GetEntry())
157  {
158  case NPC_DEATHGUARD_FLORENCE:
159  m_creature->GetMotionMaster()->MovePoint(1, Position(2242.474f, 474.683f, 36.0789f, 3.351452f));
160  break;
161  case NPC_RAZOR_HILL_GRUNT:
162  m_creature->GetMotionMaster()->MovePoint(1, Position(277.6265f, -4563.144f, 24.1798f, 5.736432f));
163  break;
164  case NPC_IRONFORGE_MOUNTAINEER:
165  m_creature->GetMotionMaster()->MovePoint(1, Position(-5755.2036f, -504.95849f, 397.72644f, 4.769157f));
166  break;
167  case NPC_STORMWIND_GUARD:
168  m_creature->GetMotionMaster()->MovePoint(1, Position(-9331.7294f, 50.873886f, 60.544579f, 0.261875f));
169  break;
170  default: // no movement required, already stationed close enough (should all have to move in TBC?)
171  relightTimer = 5000;
172  break;
173  }
174  }
175  }
176  else
177  checkExtinguishedTimer -= diff;
178  }
179  }
180  }
181  guardAI::UpdateAI(diff);
182  }
