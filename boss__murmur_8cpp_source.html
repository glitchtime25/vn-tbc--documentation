93  void HandleOocAttack()
94  {
95  if (m_creature->IsInCombat())
96  return;
97 
98  // kill one that's moving
99  if (urand(0, 1))
100  {
101  GuidVector moversVector;
102  for (ObjectGuid& guid : spellbindersVector)
103  {
104  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
105  {
106  if (creature->GetMotionMaster()->GetCurrentMovementGeneratorType() == WAYPOINT_MOTION_TYPE)
107  {
108  moversVector.push_back(guid);
109  }
110  }
111  }
112  for (ObjectGuid& guid : summonersVector)
113  {
114  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
115  {
116  if (creature->GetMotionMaster()->GetCurrentMovementGeneratorType() == WAYPOINT_MOTION_TYPE)
117  {
118  moversVector.push_back(guid);
119  }
120  }
121  }
122  if (moversVector.size() > 0)
123  {
124  if (ObjectGuid& guid = moversVector[urand(0, moversVector.size() - 1)])
125  {
126  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
127  {
128  DoCastSpellIfCan(creature, SPELL_MURMURS_WRATH);
129  }
130  }
131  }
132  }
133  // stun 5 targets
134  else
135  DoCastSpellIfCan(nullptr, SPELL_SUPPRESSION_BLAST);
136 
137  ResetTimer(MURMUR_OOC_RP_ATTACK, 3000);
138  }
139 
140  void HandleOocCasterAttack()
141  {
142  if (m_creature->IsInCombat())
143  return;
144 
145  for (ObjectGuid& guid : spellbindersVector)
146  {
147  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
148  {
149  m_creature->AI()->SendAIEvent(AI_EVENT_CUSTOM_EVENTAI_A, m_creature, creature);
150  }
151  }
152  for (ObjectGuid& guid : summonersVector)
153  {
154  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
155  {
156  m_creature->AI()->SendAIEvent(AI_EVENT_CUSTOM_EVENTAI_A, m_creature, creature);
157  }
158  }
159 
160  ResetTimer(MURMUR_OOC_CASTER_ATTACK, urand(3000, 8000));
161  }
