130  {

135  case 2: DoBroadcastText(SAY_AGGRO_3, m_creature); break;

137 
138  for (uint32 spellId : aPortalSpells)
139  DoCastSpellIfCan(nullptr, spellId);
140 
141  m_creature->CastSpell(nullptr, SPELL_SUMMON_VOID_SUMMONER, TRIGGERED_OLD_TRIGGERED);

144  m_instance->SetData(TYPE_VORPIL, IN_PROGRESS);
