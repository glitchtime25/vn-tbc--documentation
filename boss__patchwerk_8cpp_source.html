98  void ExecuteAction(uint32 action) override
99  {
100  switch (action)
101  {
102  case PATCHWERK_BERSERK_HP_CHECK:
103  {
104  if (m_creature->GetHealthPercent() < 5.0f)
105  {
106  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
107  {
108  DoScriptText(EMOTE_GENERIC_ENRAGED, m_creature);
109  DisableCombatAction(action);
110  }
111  }
112  break;
113  }
114  case PATCHWERK_HATEFUL_STRIKE:
115  {
116  if (DoCastSpellIfCan(m_creature, SPELL_HATEFULSTRIKE_PRIMER) == CAST_OK)
117  ResetCombatAction(action, 1.2 * IN_MILLISECONDS);
118  break;
119  }
120  case PATCHWERK_BERSERK:
121  {
122  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
123  {
124  DoScriptText(EMOTE_GENERIC_BERSERK, m_creature);
125  DisableCombatAction(action);
126  ResetCombatAction(PATCHWERK_BERSERK_SILMEBOLT, 30 * IN_MILLISECONDS);
127  }
128  break;
129  }
130  case PATCHWERK_BERSERK_SILMEBOLT:
131  {
132  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SLIMEBOLT) == CAST_OK)
133  ResetCombatAction(PATCHWERK_BERSERK_SILMEBOLT, 1 * IN_MILLISECONDS);
134  break;
135  }
136  default:
137  break;
138  }
139  }

150  {
151  if (!spell->GetCaster()->CanReachWithMeleeAttack(target))
152  return false;
153 
154  auto const& threatList = spell->GetCaster()->getThreatManager().getThreatList();
155  uint32 i = 0;
156  for (auto itr = threatList.begin(); itr != threatList.end() && i < 4; ++itr, ++i)
157  if ((*itr)->getTarget() == target)
158  return true;
159  return false;
160  }

163  {
