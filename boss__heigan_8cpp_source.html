82  m_creature->GetCombatManager().SetLeashingCheck([&](Unit*, float x, float /*y*/, float /*z*/)
83  {
84  float respawnX, respawnY, respawnZ;
85  m_creature->GetRespawnCoord(respawnX, respawnY, respawnZ);

181  void HandlePlatformPhase()
182  {
183  for (uint32 action : {HEIGAN_FEVER, HEIGAN_TAUNT, HEIGAN_MANA_BURN, HEIGAN_TELEPORT_PLAYERS})
184  DisableCombatAction(action);
185  m_creature->GetMotionMaster()->MoveIdle();
186  if (DoCastSpellIfCan(m_creature, SPELL_TELEPORT_SELF) == CAST_OK)
187  {
188  StopEruptions();
189  m_phase = PHASE_PLATFORM;
190  ResetTimer(HEIGAN_ERUPTION, 3000u); // Small delay to let enough time for players to reach the first safe zone
191  ResetTimer(HEIGAN_CHANNELING, 100u);
192  }
193  ResetTimer(HEIGAN_GROUND_PHASE, 45u * IN_MILLISECONDS);
194  }

229  void ExecuteAction(uint32 action) override
230  {
231  if (!m_instance)
232  return;
233 
234  switch (action)
235  {
236  case HEIGAN_TAUNT:
237  {
238  switch (urand(0, 3))
239  {
240  case 0: DoScriptText(SAY_TAUNT1, m_creature); break;
241  case 1: DoScriptText(SAY_TAUNT2, m_creature); break;
242  case 2: DoScriptText(SAY_TAUNT3, m_creature); break;
243  case 3: DoScriptText(SAY_TAUNT4, m_creature); break;
244  }
245  ResetCombatAction(action, GetSubsequentActionTimer(action));
246  return;
247  }
248  case HEIGAN_FEVER:
249  {
250  if (DoCastSpellIfCan(m_creature, SPELL_DECREPIT_FEVER) == CAST_OK)
251  ResetCombatAction(action, GetSubsequentActionTimer(action));
252  return;
253  }
254  case HEIGAN_MANA_BURN:
255  {
256  if (DoCastSpellIfCan(m_creature, SPELL_MANA_BURN) == CAST_OK)
257  ResetCombatAction(action, GetSubsequentActionTimer(action));
258  return;
259  }
260  case HEIGAN_TELEPORT_PLAYERS:
261  {
262  std::vector<Unit*> targets;
263  m_creature->SelectAttackingTargets(targets, ATTACKING_TARGET_ALL_SUITABLE, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_SKIP_TANK, m_teleportParams);
264 
265  if (targets.size() > MAX_PLAYERS_TELEPORT)
266  {
267  std::shuffle(targets.begin(), targets.end(), *GetRandomGenerator());
268  targets.resize(MAX_PLAYERS_TELEPORT);
269  }
270 
271  if (!targets.empty())
272  {
273  for (auto& target : targets)
274  target->CastSpell(target, SPELL_TELEPORT_PLAYERS, TRIGGERED_OLD_TRIGGERED);
275  }
276  DisableCombatAction(action);
277  return;
278  }
279  default:
280  return;
281  }
282  }

306  void UpdateAI(const uint32 diff) override
307  {
308  // Do nothing if no target
309  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
310  return;
311 
312  if (m_resetCheckTimer <= diff)
313  {
314  // Check if we are in range of the trigger NPC in the middle of Heigan room, if so: force evade
315  if (Creature* trigger = GetClosestCreatureWithEntry(m_creature, NPC_WORLD_TRIGGER, 45.0f))
316  m_creature->AI()->EnterEvadeMode();
317  m_resetCheckTimer = 3 * IN_MILLISECONDS;
318  }
319  else
320  m_resetCheckTimer -= diff;
321 
322  DoMeleeAttackIfReady();
323  }
