249  void ExecuteAction(uint32 i) override
250  {
251  switch (i)
252  {
253  case ANZU_ACTION_SPAWN_BROODS:
254  if (m_creature->GetHealthPercent() < m_healthBanishCheck && CanExecuteCombatAction())
255  {
256  DoSummonBroodsOfAnzu();
257  m_healthBroodCheck -= 40.0f;
258  }
259  return;
260  case ANZU_ACTION_BANISH:
261  if (m_creature->GetHealthPercent() < m_healthBanishCheck && CanExecuteCombatAction())
262  {
263  if (DoCastSpellIfCan(nullptr, SPELL_BANISH_SELF) == CAST_OK)
264  {
265  DoBroadcastText(SAY_BANISH, m_creature);
266  SetMeleeEnabled(false);
267  m_healthBanishCheck -= 40.0f;
268  ResetTimer(ANZU_BROOD_ATTACK, 15000);
269  return;
270  }
271  }
272  return;
273  }
