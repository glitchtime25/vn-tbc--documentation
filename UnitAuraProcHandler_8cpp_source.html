348 {

350  if (procFlags & MELEE_BASED_TRIGGER_MASK)

363  ((Player*)this)->UpdateDefense();
364  }
365  // If exist crit/parry/dodge/block need update aura state (for victim and attacker)

401  }
402  }

492 void Unit::ProcDamageAndSpellFor(ProcSystemArguments& argData, bool isVictim)
493 {
494  ProcExecutionData execData(argData, isVictim);
495 
496  ProcTriggeredList procTriggered;
497  std::vector<SpellAuraHolder*> holdersForDeletion;
498  // Fill procTriggered list
499  for (SpellAuraHolderMap::const_iterator itr = GetSpellAuraHolderMap().begin(); itr != GetSpellAuraHolderMap().end(); ++itr)
500  {
501  SpellAuraHolder* holder = itr->second;
502  // skip deleted auras (possible at recursive triggered call
503  if (holder->GetState() != SPELLAURAHOLDER_STATE_READY || holder->IsDeleted())
504  continue;
505 
506  SpellProcEventEntry const* spellProcEvent = nullptr;
507  SpellProcEventTriggerCheck result = IsTriggeredAtSpellProcEvent(execData, holder, spellProcEvent);
508  if (holder->GetSpellProto()->HasAttribute(SPELL_ATTR_PROC_FAILURE_BURNS_CHARGE) &&
509  result == SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_ROLL_FAILED && holder->GetAuraCharges() > 0)
510  holdersForDeletion.push_back(holder);
511 
512  if (holder->GetSpellProto()->HasAttribute(SPELL_ATTR_EX2_PROC_COOLDOWN_ON_FAILURE) && result == SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_ROLL_FAILED)
513  {
514  uint32 cooldown = 0;
515  if (spellProcEvent && spellProcEvent->cooldown)
516  cooldown = spellProcEvent->cooldown;
517  if (cooldown)
518  holder->SetProcCooldown(std::chrono::seconds(cooldown), GetMap()->GetCurrentClockTime());
519  }
520 
521  if (result != SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_OK)
522  continue;
523 
524  procTriggered.push_back(ProcTriggeredData(spellProcEvent, itr->second));
525  }
526 
527  for (SpellAuraHolder* holder : holdersForDeletion)
528  if (holder->DropAuraCharge())
529  RemoveSpellAuraHolder(holder);
530 
531  // Nothing found
532  if (procTriggered.empty())
533  return;
534 
535  // Handle effects proceed this time
536  for (ProcTriggeredList::const_iterator itr = procTriggered.begin(); itr != procTriggered.end(); ++itr)
537  {
538  // Some auras can be deleted in function called in this loop (except first, ofc)
539  SpellAuraHolder* triggeredByHolder = itr->triggeredByHolder;
540  if (triggeredByHolder->IsDeleted())
541  continue;
542 
543  SpellProcEventEntry const* spellProcEvent = itr->spellProcEvent;
544  bool useCharges = triggeredByHolder->GetAuraCharges() > 0;
545  bool procSuccess = true;
546  bool anyAuraProc = false;
547 
548  execData.cooldown = 0;
549  if (spellProcEvent && spellProcEvent->cooldown)
550  execData.cooldown = spellProcEvent->cooldown;
551 
552  for (int32 i = 0; i < MAX_EFFECT_INDEX; ++i)
553  {
554  execData.triggeredByAura = triggeredByHolder->GetAuraByEffectIndex(SpellEffectIndex(i));
555  if (!execData.triggeredByAura)
556  continue;
557 
558  Modifier* auraModifier = execData.triggeredByAura->GetModifier();
559 
560  if (execData.spellInfo)
561  {
562  if (spellProcEvent)
563  {
564  if (spellProcEvent->spellFamilyMask[i])
565  {
566  if (!execData.spellInfo->IsFitToFamilyMask(spellProcEvent->spellFamilyMask[i]))
567  continue;
568 
569  // don't allow proc from cast end for non modifier spells
570  // unless they have proc ex defined for that
571  if (IsCastEndProcModifierAura(triggeredByHolder->GetSpellProto(), SpellEffectIndex(i), execData.spellInfo))
572  {
573  if (useCharges && execData.procExtra != PROC_EX_CAST_END && spellProcEvent->procEx == PROC_EX_NONE)
574  continue;
575  }
576  else if (spellProcEvent->procEx == PROC_EX_NONE && execData.procExtra == PROC_EX_CAST_END)
577  continue;
578  }
579  // don't check dbc FamilyFlags if schoolMask exists
580  else if (!execData.triggeredByAura->CanProcFrom(execData.spellInfo, spellProcEvent->procEx, execData.procExtra, execData.damage != 0, !spellProcEvent->schoolMask))
581  continue;
582  }
583  else if (!execData.triggeredByAura->CanProcFrom(execData.spellInfo, PROC_EX_NONE, execData.procExtra, execData.damage != 0, true))
584  continue;
585  }
586 
587  execData.triggeredSpellId = 0;
588  execData.basepoints = { 0, 0, 0 };
589  SpellAuraProcResult procResult = execData.triggeredByAura->OnProc(execData);
590  if (procResult == SPELL_AURA_PROC_OK)
591  procResult = (*this.*AuraProcHandler[auraModifier->m_auraname])(execData);
592  switch (procResult)
593  {
594  case SPELL_AURA_PROC_CANT_TRIGGER:
595  continue;
596  case SPELL_AURA_PROC_FAILED:
597  // example - drain soul vanilla - third effect fails when not have talent but charge should drop
598  if (!triggeredByHolder->GetSpellProto()->HasAttribute(SPELL_ATTR_PROC_FAILURE_BURNS_CHARGE))
599  procSuccess = false;
600  break;
601  case SPELL_AURA_PROC_OK:
602  if (execData.procOnce && execData.spell)
603  execData.spell->RegisterAuraProc(execData.triggeredByAura);
604  break;
605  }
606 
607  anyAuraProc = true;
608  }
609 
610  if (procSuccess && execData.cooldown)
611  triggeredByHolder->SetProcCooldown(std::chrono::seconds(execData.cooldown), GetMap()->GetCurrentClockTime());
612 
613  // Remove charge (aura can be removed by triggers)
614  if (useCharges && procSuccess && anyAuraProc && !triggeredByHolder->IsDeleted())
615  {
616  // If last charge dropped add spell to remove list
617  if (triggeredByHolder->DropAuraCharge())
618  RemoveSpellAuraHolder(triggeredByHolder);
619  }
620  }

623 Unit::SpellProcEventTriggerCheck Unit::IsTriggeredAtSpellProcEvent(ProcExecutionData& data, SpellAuraHolder* holder, SpellProcEventEntry const*& spellProcEvent)
624 {
625  SpellEntry const* spellProto = holder->GetSpellProto();
626 
627  // Get proc Event Entry
628  spellProcEvent = sSpellMgr.GetSpellProcEvent(spellProto->Id);
629 
630  // Get EventProcFlag
631  uint32 EventProcFlag;
632  if (spellProcEvent && spellProcEvent->procFlags) // if exist get custom spellProcEvent->procFlags
633  EventProcFlag = spellProcEvent->procFlags;
634  else
635  EventProcFlag = spellProto->procFlags; // else get from spell proto
636  // Continue if no trigger exist
637  if (!EventProcFlag)
638  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
639 
640  // Check spellProcEvent data requirements
641  if (!SpellMgr::IsSpellProcEventCanTriggeredBy(spellProcEvent, EventProcFlag, data.spellInfo, data.procFlags, data.procExtra))
642  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
643 
644  // In most cases req get honor or XP from kill
645  if (EventProcFlag & PROC_FLAG_KILL && GetTypeId() == TYPEID_PLAYER)
646  {
647  bool allow = ((Player*)this)->isHonorOrXPTarget(data.target);
648  // Shadow Word: Death - can trigger from every kill
649  if (holder->GetId() == 32409)
650  allow = true;
651  if (!allow)
652  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
653  }
654  // Aura added by spell can`t trigger from self (prevent drop charges/do triggers)
655  // But except periodic triggers (can triggered from self)
656  if (data.spellInfo && data.spellInfo->Id == spellProto->Id && !(EventProcFlag & PROC_FLAG_TAKE_HARMFUL_PERIODIC))
657  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
658 
659  // Check if current equipment allows aura to proc
660  if (!data.isVictim && GetTypeId() == TYPEID_PLAYER && !spellProto->HasAttribute(SPELL_ATTR_EX3_NO_PROC_EQUIP_REQUIREMENT))
661  {
662  if (spellProto->EquippedItemClass == ITEM_CLASS_WEAPON)
663  {
664  Item* item = nullptr;
665  switch (data.attType)
666  {
667  default:
668  case BASE_ATTACK:
669  item = ((Player*)this)->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_MAINHAND);
670  break;
671  case OFF_ATTACK:
672  item = ((Player*)this)->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_OFFHAND);
673  break;
674  case RANGED_ATTACK:
675  item = ((Player*)this)->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_RANGED);
676  break;
677  }
678 
679  if (!CanUseEquippedWeapon(data.attType))
680  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
681 
682  if (!item || item->IsBroken() || item->GetProto()->Class != ITEM_CLASS_WEAPON || !((1 << item->GetProto()->SubClass) & spellProto->EquippedItemSubClassMask))
683  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
684  }
685  else if (spellProto->EquippedItemClass == ITEM_CLASS_ARMOR)
686  {
687  // Check if player is wearing shield
688  Item* item = ((Player*)this)->GetItemByPos(INVENTORY_SLOT_BAG_0, EQUIPMENT_SLOT_OFFHAND);
689  if (!item || item->IsBroken() || item->GetProto()->Class != ITEM_CLASS_ARMOR || !((1 << item->GetProto()->SubClass) & spellProto->EquippedItemSubClassMask))
690  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
691  }
692  }
693 
694  if (spellProto->HasAttribute(SPELL_ATTR_EX3_ONLY_PROC_OUTDOORS) && !GetTerrain()->IsOutdoors(GetPositionX(), GetPositionY(), GetPositionZ()))
695  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
696 
697  if (spellProto->HasAttribute(SPELL_ATTR_EX3_ONLY_PROC_ON_CASTER) && holder->GetTarget()->GetObjectGuid() != holder->GetCasterGuid())
698  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
699 
700  if (IsSitState() && !spellProto->HasAttribute(SPELL_ATTR_ALLOW_WHILE_SITTING) && !spellProto->HasAttribute(SPELL_ATTR_EX4_ALLOW_PROC_WHILE_SITTING))
701  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
702 
703  // Get chance from spell
704  float chance = (float)spellProto->procChance;
705  // If in spellProcEvent exist custom chance, chance = spellProcEvent->customChance;
706  if (spellProcEvent && spellProcEvent->customChance)
707  chance = spellProcEvent->customChance;
708  // If PPM exist calculate chance from PPM
709  if (!data.isVictim && spellProcEvent && spellProcEvent->ppmRate != 0)
710  {
711  uint32 WeaponSpeed = GetAttackTime(data.attType);
712  chance = GetPPMProcChance(WeaponSpeed, spellProcEvent->ppmRate);
713  }
714  // Apply chance modifier aura
715  if (Player* modOwner = GetSpellModOwner())
716  modOwner->ApplySpellMod(spellProto->Id, SPELLMOD_CHANCE_OF_SUCCESS, chance);
717 
718  // proc chance is reduced by an additional 3.333% per level past 60
719  if (holder->IsReducedProcChancePast60() && GetLevel() > 60)
720  chance = std::max(0.f, (1.f - ((GetLevel() - 60) * 1.f / 30.f)) * chance);
721 
722  if (data.spell)
723  {
724  if (spellProto->HasAttribute(SPELL_ATTR_EX6_AURA_IS_WEAPON_PROC) && data.spell->m_spellInfo->HasAttribute(SPELL_ATTR_EX4_SUPPRESS_WEAPON_PROCS))
725  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
726 
727  if (data.spell->m_IsTriggeredSpell)
728  {
729  if (!data.spell->m_spellInfo->HasAttribute(SPELL_ATTR_EX3_NOT_A_PROC) && !spellProto->HasAttribute(SPELL_ATTR_EX3_CAN_PROC_FROM_PROCS))
730  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
731  }
732 
733  for (uint8 i = 0; i < MAX_EFFECT_INDEX; ++i)
734  if (Aura* aura = holder->m_auras[i])
735  if (data.spell->IsAuraProcced(aura))
736  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
737  }
738 
739  for (uint8 i = 0; i < MAX_EFFECT_INDEX; ++i)
740  if (Aura* aura = holder->m_auras[i])
741  if (!aura->OnCheckProc(data))
742  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_FAILED;
743 
744  if (roll_chance_f(chance))
745  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_OK;
746  else
747  return SpellProcEventTriggerCheck::SPELL_PROC_TRIGGER_ROLL_FAILED;
748 }
