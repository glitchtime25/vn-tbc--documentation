71 CombatManeuverReturns PlayerbotRogueAI::DoFirstCombatManeuver(Unit* pTarget)
72 {
73  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
74  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
75  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
76  {
77  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
78  {
79  return RETURN_NO_ACTION_OK; // wait it out
80  }
81  else
82  {
83  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
84  }
85  }
86 
87  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
88  {
89  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
90  return RETURN_NO_ACTION_OK; // wait it out
91  else
92  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
93  }
94 
95  switch (m_ai.GetScenarioType())
96  {
97  case PlayerbotAI::SCENARIO_PVP_DUEL:
98  case PlayerbotAI::SCENARIO_PVP_BG:
99  case PlayerbotAI::SCENARIO_PVP_ARENA:
100  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
101  return DoFirstCombatManeuverPVP(pTarget);
102  case PlayerbotAI::SCENARIO_PVE:
103  case PlayerbotAI::SCENARIO_PVE_ELITE:
104  case PlayerbotAI::SCENARIO_PVE_RAID:
105  default:
106  return DoFirstCombatManeuverPVE(pTarget);
107  break;
108  }
109 
110  return RETURN_NO_ACTION_ERROR;
111 }

196  return RETURN_CONTINUE;

204  return RETURN_CONTINUE;

208  return RETURN_CONTINUE;
209  }

211  //Used to determine if this bot has highest threat
212  Unit* newTarget = m_ai.FindAttacker((PlayerbotAI::ATTACKERINFOTYPE)(PlayerbotAI::AIT_VICTIMSELF | PlayerbotAI::AIT_HIGHESTTHREAT), &m_bot);

217  {

222  }

224  return RETURN_CONTINUE;
225  if (EVASION > 0 && m_ai.GetHealthPercent() <= 35 && m_bot.IsSpellReady(EVASION) && !m_bot.HasAura(EVASION, EFFECT_INDEX_0) && m_ai.CastSpell(EVASION) == SPELL_CAST_OK)
226  return RETURN_CONTINUE;
227  if (FEINT > 0 && m_bot.IsSpellReady(FEINT) && m_ai.CastSpell(FEINT, *newTarget) == SPELL_CAST_OK)
228  return RETURN_CONTINUE;
229  if (PREPARATION > 0 && m_bot.IsSpellReady(PREPARATION) && (!m_bot.IsSpellReady(EVASION) || !m_bot.IsSpellReady(VANISH)) && m_ai.CastSpell(PREPARATION) == SPELL_CAST_OK)
230  return RETURN_CONTINUE;
231  }
232 
233  // Default: Gouge the target

254  if (!meleeReach)
255  return RETURN_CONTINUE;

257  // If target is elite and wounded: use adrenaline rush to finish it quicker
258  if (ADRENALINE_RUSH > 0 && m_ai.IsElite(pTarget) && pTarget->GetHealthPercent() < 50 && !m_bot.HasAura(ADRENALINE_RUSH, EFFECT_INDEX_0) && m_bot.IsSpellReady(ADRENALINE_RUSH) && m_ai.CastSpell(ADRENALINE_RUSH, m_bot) == SPELL_CAST_OK)

290  }
291 
292  if (RUPTURE > 0 && !pTarget->HasAura(RUPTURE, EFFECT_INDEX_0) && m_ai.CastSpell(RUPTURE, *pTarget) == SPELL_CAST_OK) // 25 energy (checked above)

318 CombatManeuverReturns PlayerbotRogueAI::DoNextCombatManeuverPVP(Unit* pTarget)
319 {
320  if (!pTarget) return RETURN_NO_ACTION_ERROR;
321 
322  Unit* pVictim = pTarget->GetVictim();
323  bool meleeReach = m_bot.CanReachWithMeleeAttack(pTarget);
324 
325  // decide what to do:
326  if (pVictim == &m_bot && CLOAK_OF_SHADOWS > 0 && m_bot.HasAura(SPELL_AURA_PERIODIC_DAMAGE) && !m_bot.HasAura(CLOAK_OF_SHADOWS, EFFECT_INDEX_0) && m_ai.CastSpell(CLOAK_OF_SHADOWS) == SPELL_CAST_OK)
327  {
328  m_ai.TellMaster("CoS!");
329  return RETURN_CONTINUE;
330  }
331  else if (m_bot.HasAura(STEALTH, EFFECT_INDEX_0))
332  SpellSequence = RogueStealth;
333  else if (pTarget->IsNonMeleeSpellCasted(true))
334  SpellSequence = RogueSpellPreventing;
335  else if (pVictim == &m_bot && m_ai.GetHealthPercent() < 40)
336  SpellSequence = RogueThreat;
337  else
338  SpellSequence = RogueCombat;
339 
340  // we fight in melee, target is not in range, skip the next part!
341  if (!meleeReach)
342  return RETURN_CONTINUE;
343 
344  std::ostringstream out;
345  switch (SpellSequence)
346  {
347  case RogueStealth:
348  if (PREMEDITATION > 0 && m_ai.CastSpell(PREMEDITATION, *pTarget) == SPELL_CAST_OK)
349  return RETURN_CONTINUE;
350  if (AMBUSH > 0 && m_ai.CastSpell(AMBUSH, *pTarget) == SPELL_CAST_OK)
351  return RETURN_CONTINUE;
352  if (CHEAP_SHOT > 0 && !pTarget->HasAura(CHEAP_SHOT, EFFECT_INDEX_0) && m_ai.CastSpell(CHEAP_SHOT, *pTarget) == SPELL_CAST_OK)
353  return RETURN_CONTINUE;
354  if (GARROTE > 0 && m_ai.CastSpell(GARROTE, *pTarget) == SPELL_CAST_OK)
355  return RETURN_CONTINUE;
356 
357  // No appropriate action found, remove stealth
358  m_bot.RemoveSpellsCausingAura(SPELL_AURA_MOD_STEALTH);
359  return RETURN_CONTINUE;
360 
361  case RogueThreat:
362  if (GOUGE > 0 && !pTarget->HasAura(GOUGE, EFFECT_INDEX_0) && m_ai.CastSpell(GOUGE, *pTarget) == SPELL_CAST_OK)
363  return RETURN_CONTINUE;
364  if (EVASION > 0 && m_ai.GetHealthPercent() <= 35 && !m_bot.HasAura(EVASION, EFFECT_INDEX_0) && m_ai.CastSpell(EVASION) == SPELL_CAST_OK)
365  return RETURN_CONTINUE;
366  if (BLIND > 0 && m_ai.GetHealthPercent() <= 30 && !pTarget->HasAura(BLIND, EFFECT_INDEX_0) && m_ai.CastSpell(BLIND, *pTarget) == SPELL_CAST_OK)
367  return RETURN_CONTINUE;
368  if (FEINT > 0 && m_ai.GetHealthPercent() <= 25 && m_ai.CastSpell(FEINT) == SPELL_CAST_OK)
369  return RETURN_CONTINUE;
370  if (VANISH > 0 && m_ai.GetHealthPercent() <= 20 && !m_bot.HasAura(FEINT, EFFECT_INDEX_0) && m_ai.CastSpell(VANISH) == SPELL_CAST_OK)
371  return RETURN_CONTINUE;
372  if (PREPARATION > 0 && m_ai.CastSpell(PREPARATION) == SPELL_CAST_OK)
373  return RETURN_CONTINUE;
374  break;
375 
376  case RogueSpellPreventing:
377  if ((KIDNEY_SHOT > 0 && m_bot.GetComboPoints() >= 2 && m_ai.CastSpell(KIDNEY_SHOT, *pTarget) == SPELL_CAST_OK) ||
378  (KICK > 0 && m_ai.CastSpell(KICK, *pTarget) == SPELL_CAST_OK))
379  return RETURN_CONTINUE;
380  // break; // No action? Go combat!
381 
382  case RogueCombat:
383  default:
384  if (m_bot.GetComboPoints() >= 5)
385  {
386  // wait for energy
387  if (m_ai.GetEnergyAmount() < 25 && (KIDNEY_SHOT || SLICE_DICE || EXPOSE_ARMOR))
388  return RETURN_NO_ACTION_OK;
389 
390  switch (pTarget->getClass())
391  {
392  case CLASS_SHAMAN:
393  if (KIDNEY_SHOT > 0 && m_ai.CastSpell(KIDNEY_SHOT, *pTarget) == SPELL_CAST_OK) // 25 energy (checked above)
394  return RETURN_CONTINUE;
395  break;
396 
397  case CLASS_WARLOCK:
398  case CLASS_HUNTER:
399  if (SLICE_DICE > 0 && m_ai.CastSpell(SLICE_DICE, *pTarget) == SPELL_CAST_OK) // 25 energy (checked above)
400  return RETURN_CONTINUE;
401  break;
402 
403  case CLASS_WARRIOR:
404  case CLASS_PALADIN:
405  if (EXPOSE_ARMOR > 0 && !pTarget->HasAura(EXPOSE_ARMOR, EFFECT_INDEX_0) && m_ai.CastSpell(EXPOSE_ARMOR, *pTarget) == SPELL_CAST_OK) // 25 energy (checked above)
406  return RETURN_CONTINUE;
407  break;
408 
409 
410  case CLASS_MAGE:
411  case CLASS_PRIEST:
412  if (RUPTURE > 0 && m_ai.CastSpell(RUPTURE, *pTarget) == SPELL_CAST_OK) // 25 energy (checked above)
413  return RETURN_CONTINUE;
414  break;
415 
416  case CLASS_ROGUE:
417  case CLASS_DRUID:
418  default:
419  break; // fall through to below
420  }
421 
422  // default combo action for rogue/druid or if other combo action is unavailable/failed
423  // wait for energy
424  if (m_ai.GetEnergyAmount() < 35 && EVISCERATE)
425  return RETURN_NO_ACTION_OK;
426  if (EVISCERATE > 0 && m_ai.CastSpell(EVISCERATE, *pTarget) == SPELL_CAST_OK)
427  return RETURN_CONTINUE;
428 
429  // failed for some (non-energy related) reason, fall through to normal attacks to maximize DPS
