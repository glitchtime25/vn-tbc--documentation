221  void SpellHit(Unit* /*caster*/, const SpellEntry* spell) override
222  {
223  // Phase 1 start
224  if (spell->Id == SPELL_CHANNEL_VISUAL)
225  {
226  DoScriptText(SAY_SUMMON_MINIONS, m_creature);
227 
228  // Get all summoning trigger NPCs
229  m_summoningTriggers.clear();
230  GetCreatureListWithEntryInGrid(m_summoningTriggers, m_creature, NPC_WORLD_TRIGGER, 100.0f);
231  }
232  // Phase 1 periodic update every 1 second (everything in phase 1 is handled here rather than in UpdateAI())
233  else if (spell->Id == SPELL_CHANNEL_VISUAL_EFFECT)
234  {
235  ++m_summonTicks;
236  switch (m_summonTicks)
237  {
238  case 4:
239  SummonIntroCreatures(NPC_SOLDIER_FROZEN, 9);
240  break;
241  case 9:
242  SummonIntroCreatures(NPC_UNSTOPPABLE_ABOM, 3);
243  break;
244  case 12:
245  SummonIntroCreatures(NPC_SOUL_WEAVER, 1);
246  break;
247  case MAX_SUMMON_TICKS:
248  DespawnIntroCreatures(); // Will leave those that for some reason are in combat
249  switch (urand(0, 2))
250  {
251  case 0:
252  DoScriptText(SAY_AGGRO1, m_creature);
253  break;
254  case 1:
255  DoScriptText(SAY_AGGRO2, m_creature);
256  break;
257  case 2:
258  DoScriptText(SAY_AGGRO3, m_creature);
259  break;
260  }
261  break;
262  case MAX_CHANNEL_TICKS:
263  StartPhase2();
264  break;
265  default:
266  UpdateSummoning();
267  // Kel'Thuzad does not enter combat until phase 2, so we check every second if there are still players alive and force him to evade otherwise
268  if (!m_instance->GetPlayerInMap(true, false))
269  EnterEvadeMode();
270  break;
271  }
272  }

275  // Remove all periodic auras on Kel'Thuzad used to summon a given NPC type
276  void StopAllSummoningForNPC(uint32 entry)
277  {
278  auto iter = summoningSpells.begin();
279  while (iter != summoningSpells.end())
280  {
281  if (iter->first == entry)
282  for (auto aura : iter->second)
283  m_creature->RemoveAurasDueToSpell(aura);
284  ++iter;
285  }

288  // Called every tick (1 second) to check if the periodic summoning auras on Kel'Thuzad need to be updated or removed
289  void UpdateSummoning()
290  {
291  for (auto add : phaseOneAdds)
292  {
293  uint32 summoningAura = 0;
294  std::vector<std::vector<uint32>> spellTimers;
295  switch (add)
296  {
297  case NPC_SOUL_WEAVER:
298  spellTimers = soulWeaverSpellsTimers;
299  break;
300  case NPC_UNSTOPPABLE_ABOM:
301  spellTimers = unstoppableAbominationSpellsTimers;
302  break;
303  case NPC_SOLDIER_FROZEN:
304  spellTimers = soldierFrozenWasteSpellsTimers;
305  break;
306  default:
307  return;
308  }
309 
310  // Iterate the summoning auras timeline for the current add as long as the tick is in range
311  // then exit with the last valid summoning aura
312  for (const auto& spellTimer : spellTimers)
313  {
314  if (spellTimer[0] <= m_summonTicks)
315  summoningAura = spellTimer[1];
316  else
317  break;
318  }
319 
320  // Do not update aura for current add type if summoning aura is already present
321  if (m_creature->HasAura(summoningAura))
322  continue;
323 
324  // Clear all summoning auras for current add before applying the new one (especially needed in case new one is NULL)
325  StopAllSummoningForNPC(add);
326  DoCastSpellIfCan(m_creature, summoningAura, CAST_TRIGGERED);
327  }
328  }

347  // Summon three type of adds in each of the surrounding alcoves
348  bool SummonIntroCreatures(uint32 entry, uint8 count)
349  {
350  if (!m_instance)
351  return false;
352 
353  float newX, newY, newZ;
354 
355  // Spawn all the adds for phase 1 from each of the trigger NPCs
356  for (auto& trigger : m_summoningTriggers)
357  {
358  // "How many NPCs per type" is stored in a vector: {npc_entry:number_of_npcs}
359  for (uint8 i = 0; i < count; ++i)
360  {
361  m_creature->GetRandomPoint(trigger->GetPositionX(), trigger->GetPositionY(), trigger->GetPositionZ(), 12.0f, newX, newY, newZ);
362  if (Creature* summoned = m_creature->SummonCreature(entry, newX, newY, newZ, 0.0f, TEMPSPAWN_CORPSE_DESPAWN, 5 * MINUTE * IN_MILLISECONDS))
363  {
364  if (summoned->AI())
365  summoned->AI()->SetReactState(REACT_PASSIVE); // Intro mobs only attack if engaged or hostile target in range
366  }
367  }
368  }
369  return true;
370  }

414  void UpdateAI(const uint32 diff) override
415  {
416  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
417  return;
418 
419  // Frostbolt: main target
420  if (m_frostBoltTimer < diff)
421  {
422  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FROST_BOLT) == CAST_OK)
423  m_frostBoltTimer = urand(5, 25) * IN_MILLISECONDS;
424  }
425  else
426  m_frostBoltTimer -= diff;
427 
428  // Frostbolt volley: all targets (about 30% chance every 20 seconds)
429  if (m_frostBoltNovaTimer < diff)
430  {
431  if (urand(0, 2) > 1)
432  DoCastSpellIfCan(m_creature, SPELL_FROST_BOLT_NOVA);
433  m_frostBoltNovaTimer = 20 * IN_MILLISECONDS;
434  }
435  else
436  m_frostBoltNovaTimer -= diff;
437 
438  // Mana Detonation: only target players with mana
439  if (m_manaDetonationTimer < diff)
440  {
441  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MANA_DETONATION, SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA))
442  {
443  if (DoCastSpellIfCan(target, SPELL_MANA_DETONATION) == CAST_OK)
444  m_manaDetonationTimer = urand(27, 40) * IN_MILLISECONDS;
445  }
446  }
447  else
448  m_manaDetonationTimer -= diff;
449 
450  // Shadow Fissure: random target
451  if (m_shadowFissureTimer < diff)
452  {
453  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
454  {
455  if (DoCastSpellIfCan(target, SPELL_SHADOW_FISSURE) == CAST_OK)
456  m_shadowFissureTimer = urand(15, 25) * IN_MILLISECONDS;
457  }
458  }
459  else
460  m_shadowFissureTimer -= diff;
461 
462  // Frostblast: random target
463  if (m_frostBlastTimer < diff)
464  {
465  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
466  {
467  if (DoCastSpellIfCan(target, SPELL_FROST_BLAST) == CAST_OK)
468  m_frostBlastTimer = urand(40, 60) * IN_MILLISECONDS;
469  }
470  }
471  else
472  m_frostBlastTimer -= diff;
473 
474  // Chains of Kel'Thuzad: main targets and four random ones
475  if (m_chainsTimer < diff)
476  {
477  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CHAINS_OF_KELTHUZAD) == CAST_OK)
478  m_chainsTimer = urand(60, 220) * IN_MILLISECONDS;
479  }
480  else
481  m_chainsTimer -= diff;
482 
483  if (m_phase == PHASE_NORMAL)
484  {
485  if (m_creature->GetHealthPercent() < 40.0f)
486  {
487  m_phase = PHASE_GUARDIANS;
488  DoScriptText(SAY_REQUEST_AID, m_creature);
489  }
490  }
491  if (m_phase == PHASE_GUARDIANS)
492  {
493  if (m_lichKingAnswerTimer && m_instance)
494  {
495  if (m_lichKingAnswerTimer <= diff)
496  {
497  // Start summoning Guardians of Icecrown
498  DoCastSpellIfCan(m_creature, SPELL_SUMMON_PERIODIC_D, CAST_TRIGGERED);
499  // Start checking if Guardians are shackled or not
500  DoCastSpellIfCan(m_creature, SPELL_GUARDIAN_INIT, CAST_TRIGGERED);
501 
502  if (Creature* lichKing = m_instance->GetSingleCreatureFromStorage(NPC_THE_LICHKING))
503  DoScriptText(SAY_ANSWER_REQUEST, lichKing);
504 
505  m_instance->DoUseDoorOrButton(GO_KELTHUZAD_WINDOW_1);
506  m_instance->DoUseDoorOrButton(GO_KELTHUZAD_WINDOW_2);
507  m_instance->DoUseDoorOrButton(GO_KELTHUZAD_WINDOW_3);
508  m_instance->DoUseDoorOrButton(GO_KELTHUZAD_WINDOW_4);
509 
510  m_lichKingAnswerTimer = 0;
511  }
512  else
513  m_lichKingAnswerTimer -= diff;
514  }
515 
516  // Stop summoning Guardians of Icecrown once five of them are spawned
517  if (m_guardiansSummonTimer)
518  {
519  if (m_guardiansSummonTimer <= diff)
520  {
521  m_creature->RemoveAurasDueToSpell(SPELL_SUMMON_PERIODIC_D);
522  m_guardiansSummonTimer = 0;
523  }
524  else
525  m_guardiansSummonTimer -= diff;
526  }
527  }
528 
529  DoMeleeAttackIfReady();
530  }
