97  void UpdateAI(const uint32 diff) override
98  {
99  // The nearest player is the one that summoned the NPC
100  if (!m_player)
101  {
102  PlayerList closestPlayersList;
103  GetPlayerListWithEntryInWorld(closestPlayersList, m_creature, 2.0f);
104  if (!closestPlayersList.empty())
105  m_player = closestPlayersList.front();
106  }
107 
108  if (m_player)
109  {
110  // Check if the web wrapped player is still alive, if not, clear ourselves
111  if (m_checkVictimAliveTimer <= diff)
112  {
113  if (!m_player->IsAlive())
114  DoCastSpellIfCan(m_creature, SPELL_CLEAR_WEB_WRAP_SELF, CAST_TRIGGERED);
115  m_checkVictimAliveTimer = 1 * IN_MILLISECONDS;
116  }
117  else
118  m_checkVictimAliveTimer -= diff;
119  }
120  }

201  void ExecuteAction(uint32 action) override
202  {
203  switch (action)
204  {
205  case MAEXXNA_ENRAGE_HP_CHECK:
206  {
207  if (m_creature->GetHealthPercent() < 30.0f)
208  {
209  if (DoCastSpellIfCan(m_creature, SPELL_FRENZY) == CAST_OK)
210  {
211  DoScriptText(EMOTE_BOSS_GENERIC_FRENZY, m_creature);
212  DisableCombatAction(action);
213  }
214  }
215  break;
216  }
217  case MAEXXNA_WEBWRAP:
218  {
219  // Web Wrap
220  // ToDo: the targets and triggers selection is probably done through spells 29280, 29281, 29282, 29283, 29285 & 29287
221  // But because we are walking in Guessland until more reliable data are available, let's do the selection manually here
222  // Randomly pick 3 targets, excluding current victim (main tank)
223  std::vector<Unit*> targets;
224  m_creature->SelectAttackingTargets(targets, ATTACKING_TARGET_ALL_SUITABLE, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_SKIP_TANK, m_webWrapParams);
225 
226  if (targets.size() > MAX_PLAYERS_WEB_WRAP)
227  {
228  std::shuffle(targets.begin(), targets.end(), *GetRandomGenerator());
229  targets.resize(MAX_PLAYERS_WEB_WRAP);
230  }
231 
232  if (!targets.empty())
233  {
234  // Check we have enough summoning NPCs spawned in regards of player targets
235  if (m_summoningTriggers.size() < targets.size())
236  {
237  script_error_log("Error in script Naxxramas::boss_maexxna: less summoning NPCs (entry %u) than expected targets (%u) for Web Wrap ability. Check your DB", NPC_INVISIBLE_MAN, MAX_PLAYERS_WEB_WRAP);
238  break;
239  }
240 
241  // Randomly pick up to three trigger NPCs
242  std::vector<Unit*> invisibleMen(m_summoningTriggers.begin(), m_summoningTriggers.end());
243  std::shuffle(invisibleMen.begin(), invisibleMen.end(), *GetRandomGenerator());
244  invisibleMen.resize(targets.size());
245 
246  for (uint8 i = 0; i < targets.size(); i++)
247  targets[i]->CastSpell(invisibleMen[i], SPELL_WEB_WRAP_INIT, TRIGGERED_IGNORE_UNSELECTABLE_FLAG, nullptr, nullptr, m_creature->GetObjectGuid());
248 
249  ResetCombatAction(action, TIMER_40_SEC);
250  break;
251  }
252  }
253  case MAEXXNA_WEBSPRAY:
254  {
255  if (DoCastSpellIfCan(m_creature, SPELL_WEBSPRAY) == CAST_OK)
256  ResetCombatAction(action, TIMER_40_SEC);
257  break;
258  }
259  case MAEXXNA_POISON_SHOCK:
260  {
261  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_POISONSHOCK) == CAST_OK)
262  ResetCombatAction(action, urand(10, 20) * IN_MILLISECONDS);
263  break;
264  }
265  case MAEXXNA_NECROTIC_POISON:
266  {
267  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_NECROTICPOISON) == CAST_OK)
268  ResetCombatAction(action, urand(20, 30) * IN_MILLISECONDS);
269  break;
270  }
271  // Summon 10 Spiderlings
272  case MAEXXNA_SUMMON_SPIDERLING:
273  {
274  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_SPIDERLING_1) == CAST_OK)
275  ResetCombatAction(action, TIMER_40_SEC);
276  break;
277  }
278  default:
279  break;
280  }
281  }

341 // Web Wrap (Maexxna: pull spell initialiser)
342 struct WebWrap : public SpellScript
343 {
344  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
345  {
346  if (effIdx == EFFECT_INDEX_0)
347  {
348  if (!spell->GetAffectiveCasterObject())
349  return;
350 
351  Unit* unitTarget = spell->GetUnitTarget();
352  if (!unitTarget || unitTarget->GetTypeId() != TYPEID_UNIT || spell->GetCaster()->GetTypeId() != TYPEID_PLAYER)
353  return;
354 
355  float dist = spell->GetCaster()->GetDistance(unitTarget, false);
356  // Switch the pull target spell based on the distance from the web wrap position
357  uint32 pullSpellId = SPELL_WEB_WRAP_500;
358  if (dist < 25.0f)
359  pullSpellId = SPELL_WEB_WRAP_200;
360  else if (dist < 50.0f)
361  pullSpellId = SPELL_WEB_WRAP_300;
362  else if (dist < 75.0f)
363  pullSpellId = SPELL_WEB_WRAP_400;
364 
365  unitTarget->CastSpell(spell->GetCaster(), pullSpellId, TRIGGERED_INSTANT_CAST, nullptr, nullptr, spell->GetAffectiveCasterObject()->GetObjectGuid());
366  }
367  }
368 };

370 // Clear Web Wrap
371 struct ClearWebWrap : public SpellScript
372 {
373  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
374  {
375  if (effIdx == EFFECT_INDEX_0)
376  {
377  Unit* unitTarget = spell->GetUnitTarget();
378  switch(spell->m_spellInfo->Id)
379  {

381  {

383  {
384  unitTarget->RemoveAurasDueToSpell(SPELL_WEB_WRAP_SUMMON); // Web Wrap polymorph
385  unitTarget->RemoveAurasDueToSpell(SPELL_WEBWRAP_STUN); // Web Wrap stun and DoT
386  }

390  {

392  unitTarget->CastSpell(nullptr, SPELL_KILL_WEBWRAP, TRIGGERED_OLD_TRIGGERED); // Kill Web Wrap
