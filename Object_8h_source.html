213  void Update(TimePoint const& now)
214  {
215  auto spellCDItr = m_spellIdMap.begin();
216  while (spellCDItr != m_spellIdMap.end())
217  {
218  auto& cd = spellCDItr->second;
219  if (cd->IsSpellCDExpired(now) && cd->IsCatCDExpired(now)) // will not remove permanent CD
220  spellCDItr = erase(spellCDItr);
221  else
222  {
223  if (cd->m_category && cd->IsCatCDExpired(now))
224  {
225  m_categoryMap.erase(cd->m_category);
226  cd->m_category = 0;
227  }
228  ++spellCDItr;
229  }
230  }
231  }
232 
233  bool AddCooldown(TimePoint clockNow, uint32 spellId, uint32 duration, uint32 spellCategory = 0, uint32 categoryDuration = 0, uint32 itemId = 0, bool onHold = false)
234  {
235  RemoveBySpellId(spellId);
236  auto resultItr = m_spellIdMap.emplace(spellId, std::make_unique<CooldownData>(clockNow, spellId, duration, spellCategory, categoryDuration, itemId, onHold));
237  // do not overwrite one permanent category cooldown with another permanent category cooldown
238  if (resultItr.second && spellCategory && categoryDuration)
239  {
240  auto catItr = FindByCategory(spellCategory);
241  if (!onHold || catItr == m_spellIdMap.end() || !catItr->second->IsPermanent())
242  {
243  // we must keep original category cd owner for sake of client sync
244  if (catItr != m_spellIdMap.end())
245  {
246  catItr->second->SetCatCDExpireTime(std::chrono::milliseconds(categoryDuration) + clockNow);
247  catItr->second->m_typePermanent = false;
248  resultItr.first->second->m_category = 0;
249  }
250  else
251  m_categoryMap.emplace(spellCategory, resultItr.first);
252  }
253  else
254  resultItr.first->second->m_category = 0;
255  }
256 
257  return resultItr.second;
258  }
259 
260  void RemoveBySpellId(uint32 spellId)
261  {
262  auto spellCDItr = m_spellIdMap.find(spellId);
263  if (spellCDItr != m_spellIdMap.end())
264  {
265  auto& cdData = spellCDItr->second;
266  if (cdData->m_category)
267  {
268  auto catCDItr = m_categoryMap.find(cdData->m_category);
269  if (catCDItr != m_categoryMap.end())
270  m_categoryMap.erase(catCDItr);
271  }
272  m_spellIdMap.erase(spellCDItr);
273  }
274  }
275 
276  void RemoveByCategory(uint32 category)
277  {
278  auto spellCDItr = m_categoryMap.find(category);
279  if (spellCDItr != m_categoryMap.end())
280  {
281  spellCDItr->second->second->m_category = 0;
282  m_categoryMap.erase(spellCDItr);
283  }
284  }
285 
286  Iterator erase(ConstIterator spellCDItr)
287  {
288  auto& cdData = spellCDItr->second;
289  if (cdData->m_category)
290  {
291  auto catCDItr = m_categoryMap.find(cdData->m_category);
292  if (catCDItr != m_categoryMap.end())
293  m_categoryMap.erase(catCDItr);
294  }
295  return m_spellIdMap.erase(spellCDItr);
296  }
297 
298  ConstIterator FindBySpellId(uint32 id) const { return m_spellIdMap.find(id); }
299 
300  ConstIterator FindByCategory(uint32 category) const
301  {
302  auto itr = m_categoryMap.find(category);
303  return itr != m_categoryMap.end() ? itr->second : end();
304  }
