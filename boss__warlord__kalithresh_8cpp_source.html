128  void MovementInform(uint32 type, uint32 pointId)
129  {
130  if (type != POINT_MOTION_TYPE || !pointId)
131  return;
132 
133  if (pointId == POINT_MOVE_DISTILLER)
134  {
135  SetCombatScriptStatus(false);
136  SetCombatMovement(true);
137  SetMeleeEnabled(true);
138  DoStartMovement(m_creature->GetVictim());
139  DoCastSpellIfCan(nullptr, SPELL_WARLORDS_RAGE);
140 
141  // Make distiller cast on arrival
142  if (Creature* distiller = m_creature->GetMap()->GetCreature(m_distillerGuid))
143  {
144  distiller->CastSpell(nullptr, SPELL_WARLORDS_RAGE_NAGA, TRIGGERED_OLD_TRIGGERED);
145  distiller->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
146  }
147  }
148  }
149 
150  void ExecuteAction(uint32 action) override
151  {
152  if (action == WARLORD_KALITHRESH_ACTION_WARLORDS_RAGE)
153  {
154  DoBroadcastText(SAY_REGEN, m_creature);
155  SetCombatScriptStatus(true);
156  SetCombatMovement(false);
157  SetMeleeEnabled(false);
158  // Move to closest distiller
159  if (Creature* distiller = GetClosestCreatureWithEntry(m_creature, NPC_NAGA_DISTILLER, 100.0f))
160  {
161  float fX, fY, fZ;
162  distiller->GetContactPoint(m_creature, fX, fY, fZ, INTERACTION_DISTANCE);
163  m_creature->SetWalk(false, true); // Prevent him from slowing down while meleehit/casted upon while starting to move
164  m_creature->GetMotionMaster()->MovePoint(POINT_MOVE_DISTILLER, fX, fY, fZ);
165  m_distillerGuid = distiller->GetObjectGuid();
166 
167  ResetCombatAction(action, GetSubsequentActionTimer(action));
168  return;
169  }
170  }
171  }
