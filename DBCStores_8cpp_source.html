210 template<class T>
211 inline void LoadDBC(uint32& availableDbcLocales, BarGoLink& bar, StoreProblemList& errlist, DBCStorage<T>& storage, const std::string& dbc_path, const std::string& filename)
212 {
213  // compatibility format and C++ structure sizes
214  MANGOS_ASSERT(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()) == sizeof(T) || LoadDBC_assert_print(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()), sizeof(T), filename));
215 
216  std::string dbc_filename = dbc_path + filename;
217  if (storage.Load(dbc_filename.c_str()))
218  {
219  bar.step();
220  for (uint8 i = 0; fullLocaleNameList[i].name; ++i)
221  {
222  if (!(availableDbcLocales & (1 << i)))
223  continue;
224 
225  std::string dbc_filename_loc = dbc_path + fullLocaleNameList[i].name + "/" + filename;
226  if (!storage.LoadStringsFrom(dbc_filename_loc.c_str()))
227  availableDbcLocales &= ~(1 << i); // mark as not available for speedup next checks
228  }
229  }
230  else
231  {
232  // sort problematic dbc to (1) non compatible and (2) nonexistent
233  FILE* f = fopen(dbc_filename.c_str(), "rb");
234  if (f)
235  {
236  char buf[100];
237  snprintf(buf, 100, " (exist, but have %u fields instead " SIZEFMTD ") Wrong client version DBC file?", storage.GetFieldCount(), strlen(storage.GetFormat()));
238  errlist.push_back(dbc_filename + buf);
239  fclose(f);
240  }
241  else
242  errlist.push_back(dbc_filename);
243  }
244 }

904 bool IsPointInAreaTriggerZone(AreaTriggerEntry const* atEntry, uint32 mapid, float x, float y, float z, float delta)
905 {
906  if (mapid != atEntry->mapid)
907  return false;
908 
909  if (atEntry->radius > 0)
910  {
911  // if we have radius check it
912  float dist2 = (x - atEntry->x) * (x - atEntry->x) + (y - atEntry->y) * (y - atEntry->y) + (z - atEntry->z) * (z - atEntry->z);
913  if (dist2 > (atEntry->radius + delta) * (atEntry->radius + delta))
914  return false;
915  }
916  else
917  {
918  // we have only extent
919 
920  // rotate the players position instead of rotating the whole cube, that way we can make a simplified
921  // is-in-cube check and we have to calculate only one point instead of 4
922 
923  // 2PI = 360, keep in mind that ingame orientation is counter-clockwise
924  double rotation = 2 * M_PI - atEntry->box_orientation;
925  double sinVal = sin(rotation);
926  double cosVal = cos(rotation);
927 
928  float playerBoxDistX = x - atEntry->x;
929  float playerBoxDistY = y - atEntry->y;
930 
931  float rotPlayerX = float(atEntry->x + playerBoxDistX * cosVal - playerBoxDistY * sinVal);
932  float rotPlayerY = float(atEntry->y + playerBoxDistY * cosVal + playerBoxDistX * sinVal);
933 
934  // box edges are parallel to coordiante axis, so we can treat every dimension independently :D
935  float dz = z - atEntry->z;
936  float dx = rotPlayerX - atEntry->x;
937  float dy = rotPlayerY - atEntry->y;
938  if ((fabs(dx) > atEntry->box_x / 2 + delta) ||
939  (fabs(dy) > atEntry->box_y / 2 + delta) ||
940  (fabs(dz) > atEntry->box_z / 2 + delta))
941  {
942  return false;
943  }
944  }
945 
946  return true;
947 }
