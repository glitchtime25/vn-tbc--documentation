119  uint32 GetSubsequentActionTimer(uint32 action)
120  {
121  switch (action)
122  {
123  case SAPPHIRON_CLEAVE: return urand(5, 10) * IN_MILLISECONDS;
124  case SAPPHIRON_TAIL_SWEEP: return urand(7, 10) * IN_MILLISECONDS;
125  case SAPPHIRON_LIFE_DRAIN: return 24u * IN_MILLISECONDS;
126  case SAPPHIRON_BLIZZARD: return urand(10, 30) * IN_MILLISECONDS;
127  case SAPPHIRON_ICEBOLT: return 3u * IN_MILLISECONDS;
128  case SAPPHIRON_BERSERK: return 300u * IN_MILLISECONDS;
129  case SAPPHIRON_AIR_PHASE: return 46u * IN_MILLISECONDS;
130  default: return 0;

165  void MovementInform(uint32 type, uint32 /*pointId*/) override
166  {
167  if (type == POINT_MOTION_TYPE && m_phase == PHASE_LIFT_OFF)
168  {
169  // Summon the Wing Buffet NPC and cast the triggered aura to despawn it
170  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_WING_BUFFET) == CAST_OK)
171  DoCastSpellIfCan(m_creature, SPELL_DESPAWN_WING_BUFFET);
172 
173  // Actual take off
174  m_creature->HandleEmote(EMOTE_ONESHOT_LIFTOFF);
175  m_creature->SetHover(true);
176  m_creature->CastSpell(nullptr, SPELL_DRAGON_HOVER, TRIGGERED_OLD_TRIGGERED);
177  ResetCombatAction(SAPPHIRON_ICEBOLT, 8u * IN_MILLISECONDS);
178  }

181  void HandleAirPhase()
182  {
183  if (m_creature->GetHealthPercent() > 10.f)
184  {
185  m_phase = PHASE_LIFT_OFF;
186  for (uint32 action : groundPhaseActions)
187  DisableCombatAction(action);
188  SetDeathPrevention(true);
189  m_creature->InterruptNonMeleeSpells(false);
190  SetCombatMovement(false);
191  SetMeleeEnabled(false);
192  m_creature->SetTarget(nullptr);
193  m_creature->GetMotionMaster()->MovePoint(1, aLiftOffPosition[0], aLiftOffPosition[1], aLiftOffPosition[2]);
194  }
195  else
196  DisableTimer(SAPPHIRON_AIR_PHASE);
197  }
198 
199  void HandleLandingPhase()
200  {
201  m_creature->HandleEmote(EMOTE_ONESHOT_LAND);
202  ResetTimer(SAPPHIRON_GROUND_PHASE, 2u * IN_MILLISECONDS);
203  }
204 
205  void HandleGroundPhase()
206  {
207  m_phase = PHASE_GROUND;
208  for (uint32 action : groundPhaseActions)
209  ResetCombatAction(action, GetSubsequentActionTimer(action));
210  SetDeathPrevention(false);
211  SetCombatMovement(true);
212  SetMeleeEnabled(true);
213  m_creature->RemoveAurasDueToSpell(SPELL_DRAGON_HOVER);
214  m_creature->SetHover(false);
215  m_creature->GetMotionMaster()->Clear(false);
216  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
217 
218  ResetTimer(SAPPHIRON_AIR_PHASE, GetSubsequentActionTimer(SAPPHIRON_AIR_PHASE));
219  }

221  void ExecuteAction(uint32 action) override
222  {
223  switch(action)
224  {
225  case SAPPHIRON_CLEAVE:
226  {
227  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
228  ResetCombatAction(action, GetSubsequentActionTimer(action));
229  return;
230  }
231  case SAPPHIRON_TAIL_SWEEP:
232  {
233  if (DoCastSpellIfCan(m_creature, SPELL_TAIL_SWEEP) == CAST_OK)
234  ResetCombatAction(action, GetSubsequentActionTimer(action));
235  return;
236  }
237  case SAPPHIRON_LIFE_DRAIN:
238  {
239  if (DoCastSpellIfCan(m_creature, SPELL_LIFE_DRAIN) == CAST_OK)
240  ResetCombatAction(action, GetSubsequentActionTimer(action));
241  return;
242  }
243  case SAPPHIRON_BLIZZARD:
244  {
245  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_BLIZZARD_INIT) == CAST_OK)
246  ResetCombatAction(action, GetSubsequentActionTimer(action));
247  return;
248  }
249  case SAPPHIRON_ICEBOLT:
250  {
251  if (m_iceboltCount < 5)
252  {
253  if (DoCastSpellIfCan(m_creature, SPELL_ICEBOLT_INIT) == CAST_OK)
254  {
255  ++m_iceboltCount;
256  ResetCombatAction(action, GetSubsequentActionTimer(action));
257  }
258  }
259  else
260  {
261  m_iceboltCount = 0;
262  DisableCombatAction(action);
263  ResetCombatAction(SAPPHIRON_FROST_BREATH, 100u); // Five Icebolt were cast, switch to Frost Breath (Ice Bomb) after that
264  }
265  return;
266  }
267  case SAPPHIRON_FROST_BREATH:
268  {
269  if (DoCastSpellIfCan(m_creature, SPELL_FROST_BREATH) == CAST_OK)
270  {
271  DoCastSpellIfCan(m_creature, SPELL_FROST_BREATH_DUMMY, CAST_TRIGGERED);
272  DoScriptText(EMOTE_BREATH, m_creature);
273  DisableCombatAction(action);
274  ResetTimer(SAPPHIRON_LANDING_PHASE, 10u * IN_MILLISECONDS);
275  }
276  return;
277  }
278  case SAPPHIRON_BERSERK:
279  {
280  if (DoCastSpellIfCan(m_creature, SPELL_BESERK) == CAST_OK)
281  {
282  DoScriptText(EMOTE_GENERIC_ENRAGED, m_creature);
283  DisableCombatAction(action);
284  }
285  return;
286  }
287  }
288  }
