76  void CorpseRemoved(uint32&) override
77  {
78  // TODO: would be better to cast a dummy spell
79  Map* pMap = m_creature->GetMap();
80 
81  if (!pMap || !pMap->IsBattleGround())
82  return;
83 
84  Map::PlayerList const& PlayerList = pMap->GetPlayers();
85 
86  for (const auto& itr : PlayerList)
87  {
88  Player* pPlayer = itr.getSource();
89  if (!pPlayer || !pPlayer->IsWithinDistInMap(m_creature, 20.0f) || !pPlayer->HasAura(SPELL_WAITING_TO_RESURRECT))

91 
92  // repop player again - now this node won't be counted and another node is searched
93  pPlayer->RepopAtGraveyard();

177 struct FlagClickBg : public SpellScript
178 {
179  SpellCastResult OnCheckCast(Spell* spell, bool /*strict*/) const override
180  {
181  switch (spell->m_spellInfo->Id)
182  {
183  case 23333: // Warsong Flag
184  case 23335: // Silverwing Flag
185  return spell->GetTrueCaster()->GetMapId() == 489 && spell->GetTrueCaster()->GetMap()->IsBattleGround() ? SPELL_CAST_OK : SPELL_FAILED_REQUIRES_AREA;
186  case 34976: // Netherstorm Flag
187  return spell->GetTrueCaster()->GetMapId() == 566 && spell->GetTrueCaster()->GetMap()->IsBattleGround() ? SPELL_CAST_OK : SPELL_FAILED_REQUIRES_AREA;
188  }
189  return SPELL_CAST_OK;
190  }
191 
192  void OnEffectExecute(Spell* spell, SpellEffectIndex /*effIdx*/) const override
193  {
194  Unit* target = spell->GetUnitTarget();
195  uint32 spellId = 0;
196  switch (spell->m_spellInfo->Id)
197  {
198  case 23383: spellId = 23335; break; // Alliance Flag Pickup
199  case 23384: spellId = 23333; break; // Horde Flag Pickup
200  }
201 
202  // flagstand and flagdrop share spells
203  if (spell->GetTrueCaster()->IsGameObject() && static_cast<GameObject*>(spell->GetTrueCaster())->GetGoType() == GAMEOBJECT_TYPE_FLAGDROP)
204  {
205  if (WorldObject* spawner = static_cast<GameObject*>(spell->GetTrueCaster())->GetSpawner())
206  if (spawner->IsPlayer() && target->IsPlayer())
207  if (static_cast<Player*>(spawner)->GetTeam() != static_cast<Player*>(target)->GetTeam())
208  return; // return of own flag already handled
209  }
210 
211  // misusing original caster to pass along original flag GO - if in future conflicts, substitute it for something else
212  target->CastSpell(target, spellId, TRIGGERED_IGNORE_GCD | TRIGGERED_HIDE_CAST_IN_COMBAT_LOG | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL, nullptr, nullptr, spell->GetTrueCaster()->GetObjectGuid());
213  }
