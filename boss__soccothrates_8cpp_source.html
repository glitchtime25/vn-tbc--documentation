104  void Aggro(Unit* /*who*/) override
105  {
106  DoBroadcastText(SAY_AGGRO, m_creature);
107 
108  GuidVector felfireVector; // at aggro felfire mobs always teleport to respawn location
109  m_instance->GetCreatureGuidVectorFromStorage(NPC_WRATH_SCRYER_FELFIRE, felfireVector);
110  for (ObjectGuid& guid : felfireVector)
111  {
112  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
113  {
114  float x, y, z, ori;
115  creature->GetRespawnCoord(x, y, z, &ori);

180  void SpellHitTarget(Unit* target, const SpellEntry* spellInfo) override
181  {
182  if (spellInfo->Id == SPELL_FELFIRE_LINE_UP)
183  {
184  // need to get even points between caster and target to reposition felfire evenly
185  float sX = m_creature->GetPositionX(), sY = m_creature->GetPositionY(); // source coords
186  float tX = m_x, tY = m_y; // target coords
187  float felfireDistX = (tX - sX) / 7, felfireDistY = (tY - sY) / 7;
188  float fX = sX + (felfireDistX * m_lineUpCounter), fY = sY + (felfireDistY * m_lineUpCounter);
189  target->NearTeleportTo(fX, fY, m_creature->GetPositionZ(), m_creature->GetOrientation());
190  m_lineUpCounter++;
191  }
192  else if (spellInfo->Id == SPELL_CHARGE && target->GetEntry() == NPC_WRATH_SCRYER_CHARGE_TARGET)
193  SetCombatMovement(true);
194  }

207  }
208  }
209  }
