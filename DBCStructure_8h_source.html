429 struct FactionTemplateEntry
430 {
431  uint32 ID; // 0
432  uint32 faction; // 1
433  uint32 factionFlags; // 2 specific flags for that faction
434  uint32 factionGroupMask; // 3 if mask set (see FactionGroupMask) then faction included in masked group
435  uint32 friendGroupMask; // 4 if mask set (see FactionGroupMask) then faction friendly to masked group
436  uint32 enemyGroupMask; // 5 if mask set (see FactionGroupMask) then faction hostile to masked group
437  uint32 enemyFaction[4]; // 6-9
438  uint32 friendFaction[4]; // 10-13
439  //------------------------------------------------------- end structure
440 
441  // helpers
442  bool IsFriendlyTo(FactionTemplateEntry const& entry) const
443  {
444  if (entry.faction)
445  {
446  for (unsigned int i : enemyFaction)
447  if (i == entry.faction)
448  return false;
449  for (unsigned int i : friendFaction)
450  if (i == entry.faction)
451  return true;
452  }
453  return (friendGroupMask & entry.factionGroupMask) || (factionGroupMask & entry.friendGroupMask);
454  }
455  bool IsHostileTo(FactionTemplateEntry const& entry) const
456  {
457  if (entry.faction)
458  {
459  for (unsigned int i : enemyFaction)
460  if (i == entry.faction)
461  return true;
462  for (unsigned int i : friendFaction)
463  if (i == entry.faction)
464  return false;
465  }
466  return (enemyGroupMask & entry.factionGroupMask) != 0;
467  }
468  bool IsHostileToPlayers() const { return (enemyGroupMask & FACTION_GROUP_MASK_PLAYER) != 0; }
469  bool IsNeutralToAll() const
470  {
471  for (unsigned int i : enemyFaction)
472  if (i != 0)
473  return false;
474  return enemyGroupMask == 0 && friendGroupMask == 0;
475  }
476  bool IsContestedGuardFaction() const { return (factionFlags & FACTION_TEMPLATE_FLAG_ATTACK_PVP_ACTIVE_PLAYERS) != 0; }
477 };
