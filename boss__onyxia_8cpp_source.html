261  void MovementInform(uint32 motionType, uint32 pointId) override
262  {
263  if (motionType != POINT_MOTION_TYPE || !m_instance)
264  return;
265 
266  switch (pointId)
267  {
268  case POINT_ID_LAND:
269  // undo flying
270  m_creature->HandleEmote(EMOTE_ONESHOT_LAND);
271  m_creature->SetHover(false);
272  m_creature->RemoveAurasDueToSpell(SPELL_PACIFY_SELF);
273  ResetTimer(ONYXIA_PHASE_TRANSITIONS, 2000); // Start PHASE_END shortly delayed
274  return;
275  case POINT_ID_LIFTOFF:
276  ResetTimer(ONYXIA_PHASE_TRANSITIONS, 3500); // Start Flying shortly delayed
277  m_creature->SetHover(true);
278  m_creature->HandleEmote(EMOTE_ONESHOT_LIFTOFF);
279  break;
280  case POINT_ID_INIT_NORTH: // Start PHASE_BREATH
281  m_uiPhase = PHASE_BREATH;
282  m_creature->CastSpell(nullptr, SPELL_DRAGON_HOVER, TRIGGERED_OLD_TRIGGERED);
283  m_creature->CastSpell(nullptr, SPELL_PACIFY_SELF, TRIGGERED_OLD_TRIGGERED);
284  SetCombatScriptStatus(false);
285  HandlePhaseTransition();
286  break;
287  default:
288  m_creature->CastSpell(nullptr, SPELL_DRAGON_HOVER, TRIGGERED_OLD_TRIGGERED);
289  break;
290  }
291 
292  if (Creature* pTrigger = m_instance->GetSingleCreatureFromStorage(NPC_ONYXIA_TRIGGER))
293  m_creature->SetFacingToObject(pTrigger);
294  }

302  // Onyxian Whelps summoning during phase 2
303  void SummonWhelps()
304  {
305  if (m_uiSummonCount >= m_uiWhelpsPerWave)
306  {
307  ResetTimer(ONYXIA_SUMMON_WHELPS, 90 * IN_MILLISECONDS - m_uiWhelpsPerWave * (m_HasSummonedFirstWave ? 3000 : 1750));
308  m_HasSummonedFirstWave = true;
309  m_uiWhelpsPerWave = urand(4, 10);
310  m_uiSummonCount = 0;
311  return;
312  }
313 
314  for (uint8 i = 0; i < 2; i++)
315  {
316  // Should probably make use of SPELL_SUMMON_ONYXIAN_WHELPS instead. Correct caster and removal of the spell is unkown, so make Onyxia do the summoning
317  if (Creature* pWhelp = m_creature->SummonCreature(NPC_ONYXIA_WHELP, afSpawnLocations[i][0], afSpawnLocations[i][1], afSpawnLocations[i][2], 0.0f, TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN, 3 * IN_MILLISECONDS))
318  {
319  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_TOPAGGRO, 0))
320  pWhelp->AI()->AttackStart(pTarget);
321  }
322  }
323  ResetTimer(ONYXIA_SUMMON_WHELPS, m_HasSummonedFirstWave ? urand(2000, 4000) : urand(1000, 2500));
324  }

326  void PhaseTransition()
327  {
328  switch (m_uiPhase)
329  {
330  case PHASE_TO_LIFTOFF:
331  m_uiPhase = PHASE_BREATH_PRE;
332  // Initial Onyxian Whelps spawn
333  ResetTimer(ONYXIA_SUMMON_WHELPS, 3000);
334  if (m_instance)
335  m_instance->SetData(TYPE_ONYXIA, DATA_LIFTOFF);
336  m_creature->GetMotionMaster()->MovePoint(POINT_ID_INIT_NORTH, aMoveData[POINT_ID_NORTH].fX, aMoveData[POINT_ID_NORTH].fY, aMoveData[POINT_ID_NORTH].fZ, FORCED_MOVEMENT_FLIGHT);
337  break;
338  case PHASE_BREATH_POST:
339  m_uiPhase = PHASE_END;
340  m_creature->SetTarget(m_creature->GetVictim());
341  SetCombatMovement(true, true);
342  SetMeleeEnabled(true);
343  SetCombatScriptStatus(false);
344  HandlePhaseTransition();
345  break;
346  }
347  }

349  void HandlePhaseTransition()
350  {
351  switch (m_uiPhase)
352  {
353  case PHASE_BREATH:
354  {
355  ResetCombatAction(ONYXIA_FIREBALL, 0);
356  ResetCombatAction(ONYXIA_MOVEMENT, 25000);
357  DisableCombatAction(ONYXIA_FLAME_BREATH);
358  DisableCombatAction(ONYXIA_CLEAVE);
359  DisableCombatAction(ONYXIA_TAIL_SWEEP);
360  DisableCombatAction(ONYXIA_WING_BUFFET);
361  DisableCombatAction(ONYXIA_KNOCK_AWAY);
362  DisableCombatAction(ONYXIA_CHECK_IN_LAIR);
363  SetActionReadyStatus(ONYXIA_PHASE_3_TRANSITION, true);
364  break;
365  }
366  case PHASE_END:
367  {
368  DisableCombatAction(ONYXIA_FIREBALL);
369  DisableCombatAction(ONYXIA_MOVEMENT);
370  ResetCombatAction(ONYXIA_BELLOWING_ROAR, 0);
371  ResetCombatAction(ONYXIA_FLAME_BREATH, urand(10000, 20000));
372  ResetCombatAction(ONYXIA_CLEAVE, urand(2000, 5000));
373  ResetCombatAction(ONYXIA_TAIL_SWEEP, urand(15000, 20000));
374  ResetCombatAction(ONYXIA_KNOCK_AWAY, urand(10000, 20000));
375  ResetCombatAction(ONYXIA_CHECK_IN_LAIR, 3000);
376  break;
377  }
378  }
379  }

389  void ExecuteAction(uint32 action) override
390  {
391  switch (action)
392  {
393  case ONYXIA_PHASE_2_TRANSITION:
394  {
395  if (m_creature->GetHealthPercent() > 65.0f)
396  return;
397 
398  m_uiPhase = PHASE_TO_LIFTOFF;
399  DoScriptText(SAY_PHASE_2_TRANS, m_creature);
400  SetCombatMovement(false);
401  SetMeleeEnabled(false);
402  m_creature->SetTarget(nullptr);
403  SetCombatScriptStatus(true);
404 
405  m_creature->GetMotionMaster()->MovePoint(POINT_ID_LIFTOFF, Position(aMoveData[POINT_ID_SOUTH].fX, aMoveData[POINT_ID_SOUTH].fY, -84.25523f, 6.248279f));
406  SetActionReadyStatus(action, false);
407  break;
408  }
409  case ONYXIA_PHASE_3_TRANSITION:
410  {
411  if (m_creature->GetHealthPercent() > 40.0f)
412  return;
413 
414  m_uiPhase = PHASE_BREATH_POST;
415  DoScriptText(SAY_PHASE_3_TRANS, m_creature);
416  DisableTimer(ONYXIA_SUMMON_WHELPS);
417 
418  SetCombatScriptStatus(true);
419  m_creature->SetTarget(nullptr);
420  m_creature->RemoveAurasDueToSpell(SPELL_DRAGON_HOVER);
421  m_creature->GetMotionMaster()->MovePoint(POINT_ID_LAND, landPoints[0][0], landPoints[0][1], landPoints[0][2], FORCED_MOVEMENT_FLIGHT);
422  SetActionReadyStatus(action, false);
423  break;
424  }
425  case ONYXIA_CHECK_IN_LAIR:
426  {
427  if (m_creature->IsNonMeleeSpellCasted(false))
428  return;
429  if (m_instance)
430  {
431  Creature* onyxiaTrigger = m_instance->GetSingleCreatureFromStorage(NPC_ONYXIA_TRIGGER);
432  if (onyxiaTrigger && !m_creature->IsWithinDistInMap(onyxiaTrigger, 90.0f, false))
433  {
434  if (!m_bHasYelledLured)
435  {
436  m_bHasYelledLured = true;
437  DoScriptText(SAY_KITE, m_creature);
438  }
439  DoCastSpellIfCan(nullptr, SPELL_BREATH_ENTRANCE);
440  }
441  else
442  m_bHasYelledLured = false;
443  }
444  ResetCombatAction(action, 3000);
445  break;
446  }
447  case ONYXIA_SUMMON_WHELPS:
448  {
449  SummonWhelps();
450  break;
451  }
452  case ONYXIA_BELLOWING_ROAR:
453  {
454  if (DoCastSpellIfCan(nullptr, SPELL_BELLOWINGROAR) == CAST_OK)
455  ResetCombatAction(action, urand(15000, 45000));
456  break;
457  }
458  case ONYXIA_FLAME_BREATH:
459  {
460  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FLAMEBREATH) == CAST_OK)
461  ResetCombatAction(action, urand(10000, 20000));
462  break;
463  }
464  case ONYXIA_CLEAVE:
465  {
466  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
467  ResetCombatAction(action, urand(5000, 10000));
468  break;
469  }
470  case ONYXIA_TAIL_SWEEP:
471  {
472  if (DoCastSpellIfCan(nullptr, SPELL_TAILSWEEP) == CAST_OK)
473  ResetCombatAction(action, urand(15000, 20000));
474  break;
475  }
476  case ONYXIA_WING_BUFFET:
477  {
478  if (DoCastSpellIfCan(nullptr, SPELL_WINGBUFFET) == CAST_OK)
479  ResetCombatAction(action, urand(15000, 30000));
480  break;
481  }
482  case ONYXIA_KNOCK_AWAY:
483  {
484  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_KNOCK_AWAY) == CAST_OK)
485  ResetCombatAction(action, urand(25000, 40000));
486  break;
487  }
488  case ONYXIA_FIREBALL:
489  {
490  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_FIREBALL, SELECT_FLAG_PLAYER))
491  {
492  if (DoCastSpellIfCan(target, SPELL_FIREBALL) == CAST_OK)
493  {
494  ResetCombatAction(action, urand(3000, 5000));
495  m_fireballVictim = target->GetObjectGuid();
496  }
497  }
498  break;
499  }
500  case ONYXIA_MOVEMENT:
501  {
502  if (m_creature->IsNonMeleeSpellCasted(false))
503  return;
504  // 3 possible actions
505  switch (urand(0, 2))
506  {
507  case 0: // breath
508  DoScriptText(EMOTE_BREATH, m_creature);
509  DoCastSpellIfCan(m_creature, aMoveData[m_uiMovePoint].uiSpellId, CAST_INTERRUPT_PREVIOUS);
510  m_uiMovePoint += NUM_MOVE_POINT / 2;
511  m_uiMovePoint %= NUM_MOVE_POINT;
512  ResetCombatAction(action, 25000);
513  return;
514  case 1: // a point on the left side
515  {
516  // C++ is stupid, so add -1 with +7
517  m_uiMovePoint += NUM_MOVE_POINT - 1;
518  m_uiMovePoint %= NUM_MOVE_POINT;
519  break;
520  }
521  case 2: // a point on the right side
522  m_uiMovePoint = (m_uiMovePoint + 1) % NUM_MOVE_POINT;
523  break;
524  }
525 
526  ResetCombatAction(action, urand(15000, 25000));
527  m_creature->RemoveAurasDueToSpell(SPELL_DRAGON_HOVER);
528  m_creature->GetMotionMaster()->MovePoint(m_uiMovePoint, aMoveData[m_uiMovePoint].fX, aMoveData[m_uiMovePoint].fY, aMoveData[m_uiMovePoint].fZ, FORCED_MOVEMENT_FLIGHT);
529  break;
530  }
531  }
532  }
