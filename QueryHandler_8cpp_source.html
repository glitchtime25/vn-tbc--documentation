35 void WorldSession::SendNameQueryResponse(CharacterNameQueryResponse& response) const
36 {
37  // guess size
38  WorldPacket data(SMSG_NAME_QUERY_RESPONSE, (8 + 1 + 4 + 4 + 4 + 10));
39  data << response.guid;
40  data << (!response.name.empty() ? response.name : GetMangosString(LANG_NON_EXIST_CHARACTER));
41 
42  if (response.realm.empty())
43  data << uint8(0);
44  else
45  data << response.realm;
46 
47  data << response.race;
48  data << response.gender;
49  data << response.classid;
50 
51  // if the first declined name field is empty, the rest must be too
52  if (response.declined.name[0].empty())
53  data << uint8(0);
54  else
55  {
56  data << uint8(1);
57 
58  for (const auto& i : response.declined.name)
59  data << i;
60  }
61 
62  SendPacket(data, true);
63 }

83 void WorldSession::SendNameQueryResponseFromDBCallBack(QueryResult* result, uint32 accountId)
84 {
85  if (!result)
86  return;
87 
88  WorldSession* session = sWorld.FindSession(accountId);
89  if (!session)
90  {
91  delete result;
92  return;
93  }
94 
95  Field* fields = result->Fetch();
96 
97  CharacterNameQueryResponse response;
98 
99  response.guid = ObjectGuid(HIGHGUID_PLAYER, fields[0].GetUInt32());
100  response.name = fields[1].GetCppString();
101  response.realm = "";
102 
103  if (!response.name.empty())
104  {
105  response.race = fields[2].GetUInt8();
106  response.gender = fields[3].GetUInt8();
107  response.classid = fields[4].GetUInt8();
108  }
109 
110  // if the first declined name field (5) is empty, the rest must be too
111  if (sWorld.getConfig(CONFIG_BOOL_DECLINED_NAMES_USED) && !fields[5].GetCppString().empty())
112  {
113  for (int i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
114  response.declined.name[i] = fields[(5 + i)].GetCppString();
115  }
116 
117  if (session->m_sessionState != WORLD_SESSION_STATE_READY)
118  session->m_offlineNameResponses.push_back(response);
119  else
120  session->SendNameQueryResponse(response);
121 
122  delete result;
123 }

126 {
127  ObjectGuid guid;
128 
129  recv_data >> guid;
130 
131  // When not logged in: check if name was already queried
132  if (m_sessionState != WORLD_SESSION_STATE_READY)

135

141 
142  if (pChar)

145

147  response.name = pChar->GetName();
148  response.realm = "";
149  response.race = uint32(pChar->getRace());
150  response.gender = uint32(pChar->getGender());
151  response.classid = uint32(pChar->getClass());
152 
153  if (DeclinedName const* declined = pChar->GetDeclinedNames())
154  {
155  for (int i = 0; i < MAX_DECLINED_NAME_CASES; ++i)
156  response.declined.name[i] = declined->name[i];
157  }
158 
159  if (m_sessionState != WORLD_SESSION_STATE_READY)
160  m_offlineNameResponses.push_back(response);
161  else
162  SendNameQueryResponse(response);
163  }
164  else
165  SendNameQueryResponseFromDB(guid);
166 }

180 
181  CreatureInfo const* ci = ObjectMgr::GetCreatureTemplate(entry);

184  int loc_idx = GetSessionDbLocaleIndex();
185 
186  char const* name = ci->Name;

188  sObjectMgr.GetCreatureLocaleStrings(entry, loc_idx, &name, &subName);
189 
190  DETAIL_LOG("WORLD: CMSG_CREATURE_QUERY '%s' - Entry: %u.", ci->Name, entry);
191  // guess size
192  WorldPacket data(SMSG_CREATURE_QUERY_RESPONSE, 100);
193  data << uint32(entry); // creature entry
194  data << name;
195  data << uint8(0) << uint8(0) << uint8(0); // name2, name3, name4, always empty
196  data << subName;
197  data << ci->IconName; // "Directions" for guard, string for Icons 2.3.0
198  data << uint32(ci->CreatureTypeFlags); // flags
199  data << uint32(ci->CreatureType); // CreatureType.dbc
200  data << uint32(ci->Family); // CreatureFamily.dbc
201  data << uint32(ci->Rank); // Creature Rank (elite, boss, etc)
202  data << uint32(0); // unknown wdbFeild11
203  data << uint32(ci->PetSpellDataId); // Id from CreatureSpellData.dbc wdbField12
204 
205  for (uint32 i : ci->ModelId)
206  data << i;
207 
208  data << float(ci->HealthMultiplier); // health multiplier
209  data << float(ci->PowerMultiplier); // mana multiplier
210  data << uint8(ci->RacialLeader);
211  SendPacket(data);
212  DEBUG_LOG("WORLD: Sent SMSG_CREATURE_QUERY_RESPONSE");
213  }
214  else
215  {
216  DEBUG_LOG("WORLD: CMSG_CREATURE_QUERY - Guid: %s Entry: %u NO CREATURE INFO!",
217  guid.GetString().c_str(), entry);
218  WorldPacket data(SMSG_CREATURE_QUERY_RESPONSE, 4);
219  data << uint32(entry | 0x80000000);

234  if (info)

236  std::string Name = info->name;
237  std::string IconName = info->IconName;
238  std::string CastBarCaption = info->castBarCaption;
239 
240  int loc_idx = GetSessionDbLocaleIndex();

243  GameObjectLocale const* gl = sObjectMgr.GetGameObjectLocale(entryID);
244  if (gl)

248  if (gl->CastBarCaption.size() > size_t(loc_idx) && !gl->CastBarCaption[loc_idx].empty())
249  CastBarCaption = gl->CastBarCaption[loc_idx];
250  }
251  }
252  DETAIL_LOG("WORLD: CMSG_GAMEOBJECT_QUERY '%s' - Entry: %u. ", info->name, entryID);
253  WorldPacket data(SMSG_GAMEOBJECT_QUERY_RESPONSE, 150);
254  data << uint32(entryID);
255  data << uint32(info->type);
256  data << uint32(info->displayId);
257  data << Name;
258  data << uint8(0) << uint8(0) << uint8(0); // name2, name3, name4
259  data << IconName; // 2.0.3, string. Icon name to use instead of default icon for go's (ex: "Attack" makes sword)
260  data << CastBarCaption; // 2.0.3, string. Text will appear in Cast Bar when using GO (ex: "Collecting")
261  data << uint8(0); // 2.0.3, string
262  data.append(info->raw.data, 24);
263  data << float(info->size); // go size
264  SendPacket(data);
265  DEBUG_LOG("WORLD: Sent SMSG_GAMEOBJECT_QUERY_RESPONSE");
266  }
267  else
268  {
269  DEBUG_LOG("WORLD: CMSG_GAMEOBJECT_QUERY - Guid: %s Entry: %u Missing gameobject info!",
270  guid.GetString().c_str(), entryID);
271  WorldPacket data(SMSG_GAMEOBJECT_QUERY_RESPONSE, 4);
272  data << uint32(entryID | 0x80000000);
273  SendPacket(data);
274  DEBUG_LOG("WORLD: Sent SMSG_GAMEOBJECT_QUERY_RESPONSE");

278 void WorldSession::HandleCorpseQueryOpcode(WorldPacket& /*recv_data*/)
279 {
280  DETAIL_LOG("WORLD: Received opcode MSG_CORPSE_QUERY");
281 
282  Corpse* corpse = GetPlayer()->GetCorpse();
283 
284  if (!corpse)
285  {
286  WorldPacket data(MSG_CORPSE_QUERY, 1);
287  data << uint8(0); // corpse not found
288  SendPacket(data);
289  return;
290  }
291 
292  uint32 corpsemapid = corpse->GetMapId();
293  float x = corpse->GetPositionX();
294  float y = corpse->GetPositionY();
295  float z = corpse->GetPositionZ();
296  int32 mapid = corpsemapid;
297 
298  // if corpse at different map
299  if (corpsemapid != _player->GetMapId())
300  {
301  // search entrance map for proper show entrance
302  if (MapEntry const* corpseMapEntry = sMapStore.LookupEntry(corpsemapid))
303  {
304  if (corpseMapEntry->IsDungeon() && corpseMapEntry->ghost_entrance_map >= 0)
305  {
306  // if corpse map have entrance
307  if (TerrainInfo const* entranceMap = sTerrainMgr.LoadTerrain(corpseMapEntry->ghost_entrance_map))
308  {
309  mapid = corpseMapEntry->ghost_entrance_map;
310  x = corpseMapEntry->ghost_entrance_x;
311  y = corpseMapEntry->ghost_entrance_y;
312  z = entranceMap->GetHeightStatic(x, y, MAX_HEIGHT);
313  }
314  }
315  }
316  }
317 
318  WorldPacket data(MSG_CORPSE_QUERY, 1 + (5 * 4));
319  data << uint8(1); // corpse found
320  data << int32(mapid);
321  data << float(x);
322  data << float(y);
323  data << float(z);
324  data << uint32(corpsemapid);
325  SendPacket(data);
326 }

339 
340  WorldPacket data(SMSG_NPC_TEXT_UPDATE, 100); // guess size
341  data << textID;
342 
343  if (!gossip)

345  for (uint32 i = 0; i < MAX_GOSSIP_TEXT_OPTIONS; ++i)
346  {
347  data << float(0);
348  data << "Greetings $N";
349  data << "Greetings $N";

352  data << uint32(0);
353  data << uint32(0);

355  data << uint32(0);
356  data << uint32(0);
357  }
358  }
359  else
360  {
361  std::string Text_0[MAX_GOSSIP_TEXT_OPTIONS], Text_1[MAX_GOSSIP_TEXT_OPTIONS];
362  bool locales = true;
363  int loc_idx = GetSessionDbLocaleIndex();
364  for (int i = 0; i < MAX_GOSSIP_TEXT_OPTIONS; ++i)
365  {
366  if (gossip->Options[i].broadcastTextId)

368  locales = false;

371  }

374  Text_0[i] = gossip->Options[i].Text_0;
375  Text_1[i] = gossip->Options[i].Text_1;
376  }
377  }
378 
379  if (locales)
380  sObjectMgr.GetNpcTextLocaleStringsAll(textID, loc_idx, &Text_0, &Text_1);
381 
382  for (int i = 0; i < MAX_GOSSIP_TEXT_OPTIONS; ++i)
383  {
384  data << gossip->Options[i].Probability;
385 
386  if (Text_0[i].empty())
387  data << Text_1[i];
388  else
389  data << Text_0[i];
390 
391  if (Text_1[i].empty())
392  data << Text_0[i];
393  else
394  data << Text_1[i];
395 
396  data << gossip->Options[i].Language;
397 
398  for (auto Emote : gossip->Options[i].Emotes)
399  {
400  data << Emote._Delay;
401  data << Emote._Emote;
402  }
403  }
404  }
405 
406  SendPacket(data);
407 
408  DEBUG_LOG("WORLD: Sent SMSG_NPC_TEXT_UPDATE");
409 }
