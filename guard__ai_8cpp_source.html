62 void guardAI::JustRespawned()
63 {
64  CombatAI::JustRespawned();
65  const auto& spellList = GetSpellList();
66  if (spellList.ChanceRangedAttack)
67  {
68  for (auto& spell : spellList.Spells)
69  {
70  if (spell.second.Flags & SPELL_LIST_FLAG_RANGED_ACTION)
71  {
72  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spell.second.SpellId);
73  if (!spellInfo)
74  continue;
75  SetRangedMode(true, CalculateSpellRange(spellInfo) * 0.8f, TYPE_PROXIMITY);
76  break;
77  }
78  }
79  }
80 }
