146  void UpdateAI(const uint32 diff) override
147  {
148  m_introDialogue.DialogueUpdate(diff);
149 
150  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
151  return;
152 
153  // Impale
154  if (m_impaleTimer < diff)
155  {
156  // Cast Impale on a random target
157  // Do NOT cast it when we are afflicted by locust swarm
158  if (!m_creature->HasAura(SPELL_LOCUSTSWARM))
159  {
160  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
161  DoCastSpellIfCan(target, SPELL_IMPALE);
162  }
163 
164  m_impaleTimer = urand(12, 18) * IN_MILLISECONDS;
165  }
166  else
167  m_impaleTimer -= diff;
168 
169  // Locust Swarm
170  if (m_locustSwarmTimer < diff)
171  {
172  if (DoCastSpellIfCan(m_creature, SPELL_LOCUSTSWARM) == CAST_OK)
173  {
174  // Summon a crypt guard
175  m_summonTimer = 3 * IN_MILLISECONDS;
176  m_locustSwarmTimer = 90 * IN_MILLISECONDS;
177  m_impaleTimer += 23 * IN_MILLISECONDS; // Delay next Impale by Locust Swarm duration (20 sec) + casting time (3 sec), this prevent Impale to be always cast right after Locust Swarm ends
178  }
179  }
180  else
181  m_locustSwarmTimer -= diff;
182 
183  // Summon Crypt Guard after Locust Swarm
184  if (m_summonTimer)
185  {
186  if (m_summonTimer <= diff)
187  {
188  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_GUARD) == CAST_OK)
189  m_summonTimer = 0;
190  }
191  else
192  m_summonTimer -= diff;
193  }
194 
195  // Summon Corpse Scarabs from dead Crypt Guard
196  if (m_corpseScarabsTimer)
197  {
198  if (m_corpseScarabsTimer <= diff)
199  {
200  if (DoCastSpellIfCan(nullptr, SPELL_SPAWN_CORPSE_SCARABS) == CAST_OK)
201  m_corpseScarabsTimer = urand(65, 105) * IN_MILLISECONDS;
202  }
203  else
204  m_corpseScarabsTimer -= diff;
205  }
206 
207  DoMeleeAttackIfReady();
208  }
