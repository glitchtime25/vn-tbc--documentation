37 void BattleGroundAB::Update(uint32 diff)
38 {
39  BattleGround::Update(diff);
40 
41  if (GetStatus() != STATUS_IN_PROGRESS)
42  return;
43 
44  // start counting the team points on every update tick;
45  // every time we start from 0 for each team
46  int team_points[PVP_TEAM_COUNT] = { 0, 0 };
47 
48  for (uint8 node = 0; node < BG_AB_MAX_NODES; ++node)
49  {
50  // 3 sec delay to spawn new banner instead previous despawned one
51  if (m_bannerTimers[node].timer)
52  {
53  if (m_bannerTimers[node].timer <= diff)
54  {
55  DoUpdateBanner(node, m_bannerTimers[node].status, false);
56  m_bannerTimers[node].timer = 0;
57  }
58  else
59  m_bannerTimers[node].timer -= diff;
60  }
61 
62  // 1-minute to occupy a node from contested state
63  if (m_nodeTimers[node])
64  {
65  if (m_nodeTimers[node] <= diff)
66  {
67  m_nodeTimers[node] = 0;
68 
69  // Process the changes from contested to occupied
70  PvpTeamIndex teamIndex = PvpTeamIndex(m_nodeStatus[node] - 1);
71  Team team = GetTeamIdByTeamIndex(PvpTeamIndex(teamIndex));
72 
73  m_prevNodeStatus[node] = m_nodeStatus[node];
74  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_OCCUPIED);
75 
76  // create new occupied banner
77  DoUpdateBanner(node, m_nodeStatus[node], true);
78  ProcessNodeCapture(node, teamIndex);
79  DoUpdateNodeWorldstate(node, abOccupiedStates[teamIndex][node]);
80 
81  // Message to chatlog
82  ChatMsg chatSystem = teamIndex == TEAM_INDEX_ALLIANCE ? CHAT_MSG_BG_SYSTEM_ALLIANCE : CHAT_MSG_BG_SYSTEM_HORDE;
83  uint32 factionStrig = teamIndex == TEAM_INDEX_ALLIANCE ? LANG_BG_ALLY : LANG_BG_HORDE;
84 
85  SendMessage2ToAll(LANG_BG_AB_NODE_TAKEN, chatSystem, nullptr, factionStrig, GetNodeMessageId(node));
86  PlaySoundToAll(teamIndex == TEAM_INDEX_ALLIANCE ? BG_AB_SOUND_NODE_CAPTURED_ALLIANCE : BG_AB_SOUND_NODE_CAPTURED_HORDE);
87  }
88  else
89  m_nodeTimers[node] -= diff;
90  }
91 
92  // for each occupied node we add team points;
93  // this influences the abTickPoints and the abTickIntervals values
94  for (uint8 teamIndex = 0; teamIndex < PVP_TEAM_COUNT; ++teamIndex)
95  if (m_nodeStatus[node] == ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_OCCUPIED))
96  ++team_points[teamIndex];
97  }
98 
99  // Accumulate points
100  for (uint8 teamIndex = 0; teamIndex < PVP_TEAM_COUNT; ++teamIndex)
101  {
102  int points = team_points[teamIndex];
103  if (!points)
104  continue;
105 
106  m_lastTick[teamIndex] += diff;
107 
108  if (m_lastTick[teamIndex] > abTickIntervals[points])
109  {
110  m_lastTick[teamIndex] -= abTickIntervals[points];
111 
112  int32 worldstateId = teamIndex == TEAM_INDEX_ALLIANCE ? BG_AB_OP_RESOURCES_ALLY : BG_AB_OP_RESOURCES_HORDE;
113  int32 newValue = GetBgMap()->GetVariableManager().GetVariable(worldstateId) + abTickPoints[points];
114 
115  if (newValue > BG_AB_MAX_TEAM_SCORE)
116  newValue = BG_AB_MAX_TEAM_SCORE;
117 
118  m_honorScoreTicks[teamIndex] += abTickPoints[points];
119  m_reputationScoreTics[teamIndex] += abTickPoints[points];
120 
121  // update reputation
122  if (m_reputationScoreTics[teamIndex] >= m_reputationTics)
123  {
124  (teamIndex == TEAM_INDEX_ALLIANCE) ? RewardReputationToTeam(509, 10, ALLIANCE) : RewardReputationToTeam(510, 10, HORDE);
125  m_reputationScoreTics[teamIndex] -= m_reputationTics;
126  }
127 
128  // update honor
129  if (m_honorScoreTicks[teamIndex] >= m_honorTicks)
130  {
131  RewardHonorToTeam(GetBonusHonorFromKill(1), (teamIndex == TEAM_INDEX_ALLIANCE) ? ALLIANCE : HORDE);
132  m_honorScoreTicks[teamIndex] -= m_honorTicks;
133  }
134 
135  // send info for near victory
136  if (!m_isInformedNearVictory && newValue > BG_AB_WARNING_NEAR_VICTORY_SCORE)
137  {
138  SendMessageToAll(teamIndex == TEAM_INDEX_ALLIANCE ? LANG_BG_AB_A_NEAR_VICTORY : LANG_BG_AB_H_NEAR_VICTORY, CHAT_MSG_BG_SYSTEM_NEUTRAL);
139  PlaySoundToAll(teamIndex == TEAM_INDEX_ALLIANCE ? BG_AB_SOUND_NEAR_VICTORY_ALLIANCE : BG_AB_SOUND_NEAR_VICTORY_HORDE);
140  m_isInformedNearVictory = true;
141  }
142 
143  // update resource world state
144  GetBgMap()->GetVariableManager().SetVariable(worldstateId, newValue);
145  }
146  }
147 
148  // Test win condition
149  if (GetBgMap()->GetVariableManager().GetVariable(BG_AB_OP_RESOURCES_ALLY) >= BG_AB_MAX_TEAM_SCORE)
150  EndBattleGround(ALLIANCE);
151  if (GetBgMap()->GetVariableManager().GetVariable(BG_AB_OP_RESOURCES_HORDE) >= BG_AB_MAX_TEAM_SCORE)
152  EndBattleGround(HORDE);
153 }

255 // Method that handles the banner click
256 void BattleGroundAB::HandlePlayerClickedOnFlag(Player* player, GameObject* go)
257 {
258  if (GetStatus() != STATUS_IN_PROGRESS)
259  return;
260 
261  DEBUG_LOG("BattleGroundAB: Handle flag clicked by team %u for gameobject entry %u.", player->GetTeam(), go->GetEntry());
262 
263  PvpTeamIndex teamIndex = GetTeamIndexByTeamId(player->GetTeam());
264  uint32 scoreType = 0;
265  uint32 soundId = 0;
266  uint32 worldState = 0;
267  int32 textEntry = 0;
268  uint32 factionStrig = 0;
269 
270  // process battleground event
271  uint8 event = (sBattleGroundMgr.GetGameObjectEventIndex(go->GetDbGuid())).event1;
272  if (event >= BG_AB_MAX_NODES) // not a node
273  return;
274 
275  ABNodes node = ABNodes(event);
276 
277  player->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_PVP_ACTIVE_CANCELS);
278 
279  // Node is NEUTRAL -> change to CONTESTED
280  if (m_nodeStatus[node] == ABNodeStatus(BG_AB_NODE_TYPE_NEUTRAL))
281  {
282  scoreType = SCORE_BASES_ASSAULTED;
283  worldState = abContestedStates[teamIndex][node];
284 
285  m_prevNodeStatus[node] = m_nodeStatus[node];
286  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_CONTESTED);
287 
288  m_nodeTimers[node] = BG_AB_FLAG_CAPTURING_TIME;
289 
290  textEntry = LANG_BG_AB_NODE_CLAIMED;
291  factionStrig = teamIndex == TEAM_INDEX_ALLIANCE ? LANG_BG_ALLY : LANG_BG_HORDE;
292  soundId = BG_AB_SOUND_NODE_CLAIMED;
293  }
294  // Node is CONTESTED -> change depends on previous status
295  else if (m_nodeStatus[node] == ABNodeStatus(BG_AB_NODE_STATUS_ALLY_CONTESTED) || m_nodeStatus[node] == ABNodeStatus(BG_AB_NODE_STATUS_HORDE_CONTESTED))
296  {
297  // Last state is CONTESTED -> change to CONTESTED (change owner team)
298  if (m_prevNodeStatus[node] < ABNodeStatus(BG_AB_NODE_TYPE_OCCUPIED))
299  {
300  scoreType = SCORE_BASES_ASSAULTED;
301  worldState = abContestedStates[teamIndex][node];
302 
303  m_prevNodeStatus[node] = m_nodeStatus[node];
304  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_CONTESTED);
305 
306  m_nodeTimers[node] = BG_AB_FLAG_CAPTURING_TIME;
307 
308  textEntry = LANG_BG_AB_NODE_ASSAULTED;
309  }
310  // Last state is OCCUPIED -> change back to OCCUPIED (defend the node)
311  else
312  {
313  scoreType = SCORE_BASES_DEFENDED;
314  worldState = abOccupiedStates[teamIndex][node];
315 
316  m_prevNodeStatus[node] = m_nodeStatus[node];
317  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_OCCUPIED);
318 
319  m_nodeTimers[node] = 0;
320 
321  textEntry = LANG_BG_AB_NODE_DEFENDED;
322 
323  // apply the node capture benefits
324  ProcessNodeCapture(node, teamIndex);
325  }
326 
327  soundId = (teamIndex == TEAM_INDEX_ALLIANCE) ? BG_AB_SOUND_NODE_ASSAULTED_ALLIANCE : BG_AB_SOUND_NODE_ASSAULTED_HORDE;
328  }
329  // Node is OCCUPIED -> change to CONTESTED
330  else
331  {
332  scoreType = SCORE_BASES_ASSAULTED;
333  worldState = abContestedStates[teamIndex][node];
334 
335  m_prevNodeStatus[node] = m_nodeStatus[node];
336  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_CONTESTED);
337 
338  m_nodeTimers[node] = BG_AB_FLAG_CAPTURING_TIME;
339 
340  textEntry = LANG_BG_AB_NODE_ASSAULTED;
341  soundId = (teamIndex == TEAM_INDEX_ALLIANCE) ? BG_AB_SOUND_NODE_ASSAULTED_ALLIANCE : BG_AB_SOUND_NODE_ASSAULTED_HORDE;
342 
343  PvpTeamIndex otherTeamIndex = GetOtherTeamIndex(teamIndex);
344  int32 worldstateId = otherTeamIndex == TEAM_INDEX_ALLIANCE ? BG_AB_OP_OCCUPIED_BASES_ALLY : BG_AB_OP_OCCUPIED_BASES_HORDE;
345  int32 newValue = GetBgMap()->GetVariableManager().GetVariable(worldstateId) - 1;
346  GetBgMap()->GetVariableManager().SetVariable(worldstateId, newValue);
347 
348  GetBgMap()->GetGraveyardManager().SetGraveYardLinkTeam(abGraveyardIds[node], BG_AB_ZONE_MAIN, TEAM_INVALID);
349  }
350 
351  // update score
352  UpdatePlayerScore(player, scoreType, 1);
353 
354  // update banner object
355  DoUpdateBanner(node, m_nodeStatus[node], true);
356 
357  // update world state
358  DoUpdateNodeWorldstate(node, worldState);
359 
360  // send message; this depends on the node status
361  ChatMsg chatSystem = teamIndex == TEAM_INDEX_ALLIANCE ? CHAT_MSG_BG_SYSTEM_ALLIANCE : CHAT_MSG_BG_SYSTEM_HORDE;
362 
363  if (factionStrig)
364  SendMessage2ToAll(textEntry, chatSystem, player, GetNodeMessageId(node), factionStrig);
365  else
366  SendMessage2ToAll(textEntry, chatSystem, player, GetNodeMessageId(node));
367 
368  // play sound to all players
369  PlaySoundToAll(soundId);
370 }
