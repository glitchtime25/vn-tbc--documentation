190 void MapManager::Update(uint32 diff)
191 {
192  i_timer.Update(diff);
193  if (!i_timer.Passed())
194  return;
195 
196  for (auto& map : i_maps)
197  {
198  if (m_updater.activated())
199  m_updater.schedule_update(new MapUpdateWorker(*map.second, (uint32)i_timer.GetCurrent(), m_updater));
200  else
201  map.second->Update((uint32)i_timer.GetCurrent());
202  }
203 
204  if (m_updater.activated())
205  m_updater.wait();
206 
207  // remove all maps which can be unloaded
208  MapMapType::iterator iter = i_maps.begin();
209  while (iter != i_maps.end())
210  {
211  Map* pMap = iter->second;
212  // check if map can be unloaded
213  if (pMap->CanUnload((uint32)i_timer.GetCurrent()))
214  {
215  pMap->UnloadAll(true);
216  delete pMap;
217 
218  i_maps.erase(iter++);
219  }
220  else
221  ++iter;
222  }
223 
224  i_timer.SetCurrent(0);
225 }

308 Map* MapManager::CreateInstance(uint32 id, Player* player)
309 {
310  Map* map = nullptr;
311  Map* pNewMap = nullptr;
312  uint32 NewInstanceId = 0; // instanceId of the resulting map
313  const MapEntry* entry = sMapStore.LookupEntry(id);
314 
315  if (entry->IsBattleGroundOrArena())
316  {
317  // find existing bg map for player
318  NewInstanceId = player->GetBattleGroundId();
319  MANGOS_ASSERT(NewInstanceId);
320  map = FindMap(id, NewInstanceId);
321  MANGOS_ASSERT(map);
322  }
323  else if (DungeonPersistentState* pSave = player->GetBoundInstanceSaveForSelfOrGroup(id))
324  {
325  // solo/perm/group
326  NewInstanceId = pSave->GetInstanceId();
327  map = FindMap(id, NewInstanceId);
328  // it is possible that the save exists but the map doesn't
329  if (!map)
330  pNewMap = CreateDungeonMap(id, NewInstanceId, pSave->GetDifficulty(), pSave);
331  }
332  else
333  {
334  // if no instanceId via group members or instance saves is found
335  // the instance will be created for the first time
336  NewInstanceId = GenerateInstanceId();
337 
338  Difficulty diff = player->GetGroup() ? player->GetGroup()->GetDifficulty() : player->GetDifficulty();
339  pNewMap = CreateDungeonMap(id, NewInstanceId, diff);
340  }
341 
342  // add a new map object into the registry
343  if (pNewMap)
344  {
345  i_maps[MapID(id, NewInstanceId)] = pNewMap;
346  map = pNewMap;
347  }
348 
349  return map;
350 }

352 DungeonMap* MapManager::CreateDungeonMap(uint32 id, uint32 InstanceId, Difficulty difficulty, DungeonPersistentState* save)
353 {
354  // make sure we have a valid map id
355  const MapEntry* entry = sMapStore.LookupEntry(id);
356  if (!entry)

359  MANGOS_ASSERT(false);

363  sLog.outError("CreateDungeonMap: no instance template for map %d", id);
364  MANGOS_ASSERT(false);
365  }
366 
367  // some instances only have one difficulty
368  if (entry && !entry->SupportsHeroicMode())
369  difficulty = DUNGEON_DIFFICULTY_NORMAL;
370 
371  DEBUG_LOG("MapInstanced::CreateDungeonMap: %s map instance %d for %d created with difficulty %d", save ? "" : "new ", InstanceId, id, difficulty);
372 
373  DungeonMap* map = new DungeonMap(id, i_gridCleanUpDelay, InstanceId, difficulty);
374 
375  // Dungeons can have saved instance data
376  bool load_data = save != nullptr;
377  map->Initialize(load_data);
378 
379  return map;
