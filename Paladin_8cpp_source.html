61

64

67

71  {
72  if ((*i)->GetSpellProto()->SpellIconID == 205 && (*i)->GetSpellProto()->Attributes == uint64(0x01D0))
73  {
74  int32 chance = (*i)->GetModifier()->m_amount;
75  if (roll_chance_i(chance))

146  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
147  {
148  if (effIdx != EFFECT_INDEX_0)
149  return;
150 
151  Unit* unitTarget = spell->GetUnitTarget();
152  if (!unitTarget)
153  return;
154  Unit* caster = spell->GetCaster();
155 
156  // non-standard cast requirement check
157  if (unitTarget->getAttackers().empty())
158  {
159  caster->RemoveSpellCooldown(*spell->m_spellInfo, true);
160  spell->SendCastResult(SPELL_FAILED_TARGET_AFFECTING_COMBAT);
161  return;
162  }
163 
164  // not empty (checked), copy
165  Unit::AttackerSet attackers = unitTarget->getAttackers();
166 
167  // selected from list 3
168  size_t size = std::min(size_t(3), attackers.size());
169  for (uint32 i = 0; i < size; ++i)
170  {
171  Unit::AttackerSet::iterator aItr = attackers.begin();
172  std::advance(aItr, urand() % attackers.size());
173  caster->CastSpell((*aItr), 31790, TRIGGERED_NONE); // step 2
174  attackers.erase(aItr);
175  }
176  }
