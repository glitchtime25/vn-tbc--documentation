97  void ExecuteAction(uint32 action) override
98  {
99  switch (action)
100  {
101  case LOATHEB_INEVITABLE_DOOM:
102  {
103  if (DoCastSpellIfCan(m_creature, SPELL_INEVITABLE_DOOM, CAST_TRIGGERED) == CAST_OK)
104  ResetCombatAction(action, ((m_corruptedMindCount <= 5) ? 30 : 15) * IN_MILLISECONDS);
105  break;
106  }
107  case LOATHEB_CORRUPTED_MIND:
108  {
109  if (DoCastSpellIfCan(m_creature, SPELL_CORRUPTED_MIND, CAST_TRIGGERED) == CAST_OK)
110  {
111  ++m_corruptedMindCount;
112  ResetCombatAction(action, 60 * IN_MILLISECONDS);
113  }
114  break;
115  }
116  case LOATHEB_SUMMON:
117  {
118  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_SPORE, CAST_TRIGGERED) == CAST_OK)
119  ResetCombatAction(action, 12 * IN_MILLISECONDS);
120  break;
121  }
122  case LOATHEB_POISON_AURA:
123  {
124  if (DoCastSpellIfCan(m_creature, SPELL_POISON_AURA) == CAST_OK)
125  ResetCombatAction(action, 12 * IN_MILLISECONDS);
126  break;
127  }
128  case LOATHEB_REMOVE_CURSE:
129  {
130  SpellCastResult decurseResult = m_creature->CastSpell(m_creature, SPELL_REMOVE_CURSE, TRIGGERED_OLD_TRIGGERED);
131  if (decurseResult == SPELL_CAST_OK || decurseResult == SPELL_FAILED_NOTHING_TO_DISPEL) // Don't throw an error if there is nothing to dispel
132  ResetCombatAction(action, 30 * IN_MILLISECONDS);
133  break;
134  }
135  default:
136  break;
137  }
138  }

141 struct CorruptedMind : public SpellScript
142 {
143  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
144  {
145  if (effIdx == EFFECT_INDEX_0)
146  {
147  if (Unit* target = spell->GetUnitTarget())
148  {
149  // This spell only works on players as it triggers spells that override spell class scripts
150  if (target->GetTypeId() != TYPEID_PLAYER)
151  return;
152 
153  // Determine which sub-spell to trigger for each healing class
154  uint32 spellId = 0;
155  switch (target->getClass())
156  {
157  case CLASS_PALADIN: spellId = 29196; break;
158  case CLASS_PRIEST: spellId = 29185; break;
159  case CLASS_SHAMAN: spellId = 29198; break;
160  case CLASS_DRUID: spellId = 29194; break;
161  default: break;
162  }
163  if (spellId != 0)
164  spell->GetCaster()->CastSpell(target, spellId, TRIGGERED_OLD_TRIGGERED, nullptr);
165  }
166  }
167  }
